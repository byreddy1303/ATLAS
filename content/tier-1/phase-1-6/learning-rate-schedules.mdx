export const schedulesCode = `# The four learning-rate schedules you'll see in modern DL:
# step, exponential, cosine, and cosine+linear warmup
import numpy as np, matplotlib.pyplot as plt

TOTAL = 10000       # total training steps
PEAK_LR = 3e-4      # peak learning rate

def step_schedule(t):
    "Step decay: drop by 10x at 3k and 6k steps."
    if t < 3000: return PEAK_LR
    if t < 6000: return PEAK_LR * 0.1
    return PEAK_LR * 0.01

def exp_schedule(t, gamma=0.9995):
    "Exponential: lr *= gamma each step."
    return PEAK_LR * gamma ** t

def cosine_schedule(t, warmup=0):
    "Cosine annealing (no warmup): lr = peak * 0.5 * (1 + cos(pi * t / T))."
    if t < warmup: return PEAK_LR * t / max(1, warmup)
    progress = (t - warmup) / (TOTAL - warmup)
    return PEAK_LR * 0.5 * (1 + np.cos(np.pi * progress))

def warmup_cosine(t, warmup=1000):
    "Linear warmup followed by cosine decay — the LLM default."
    return cosine_schedule(t, warmup=warmup)

ts = np.arange(TOTAL)
schedules = {
    "step":            [step_schedule(t) for t in ts],
    "exponential":     [exp_schedule(t) for t in ts],
    "cosine (no wm)":  [cosine_schedule(t) for t in ts],
    "warmup + cosine": [warmup_cosine(t) for t in ts],
}

plt.figure(figsize=(9, 4))
for name, lrs in schedules.items():
    plt.plot(ts, lrs, lw=2, label=name)
plt.xlabel("step"); plt.ylabel("learning rate"); plt.legend()
plt.title("Four learning-rate schedules")
plt.grid(alpha=0.3); plt.tight_layout()`;

export const warmupWhyCode = `# Why warmup matters — simulate Adam's early-training instability without it.
# When β2 = 0.999 (default), v estimate is poor for many steps.
# High initial learning rate + poor v estimate = unstable early updates.
import numpy as np

def adam_step(params, grads, m, v, t, lr, b1=0.9, b2=0.999, eps=1e-8):
    m = b1 * m + (1 - b1) * grads
    v = b2 * v + (1 - b2) * grads ** 2
    m_hat = m / (1 - b1 ** t); v_hat = v / (1 - b2 ** t)
    return params - lr * m_hat / (np.sqrt(v_hat) + eps), m, v

# Simulate a scenario: gradients are big and noisy in first 100 steps, then stabilize.
np.random.seed(0)
def synthetic_gradient(t):
    if t < 100: return np.random.normal(0, 10, size=3)     # 10x bigger, noisy
    return np.random.normal(0, 1, size=3)                   # normal after

# Without warmup — full lr from step 1
params = np.zeros(3); m = np.zeros(3); v = np.zeros(3)
max_update_no_warmup = 0
for t in range(1, 200):
    g = synthetic_gradient(t)
    new_params, m, v = adam_step(params, g, m, v, t, lr=3e-4)
    update = np.abs(new_params - params).max()
    max_update_no_warmup = max(max_update_no_warmup, update)
    params = new_params
print(f"NO warmup:   max update magnitude = {max_update_no_warmup:.4f}")

# With linear warmup over first 100 steps
params = np.zeros(3); m = np.zeros(3); v = np.zeros(3)
max_update_warmup = 0
for t in range(1, 200):
    g = synthetic_gradient(t)
    scale = min(1.0, t / 100)                               # linear ramp
    new_params, m, v = adam_step(params, g, m, v, t, lr=3e-4 * scale)
    update = np.abs(new_params - params).max()
    max_update_warmup = max(max_update_warmup, update)
    params = new_params
print(f"WITH warmup: max update magnitude = {max_update_warmup:.4f}")
print()
print("With warmup, the largest parameter update is much smaller —")
print("preventing early-training explosions when v hasn't converged.")`;

export const oneCycleCode = `# One-cycle schedule: warmup UP, then anneal DOWN, with peak lr in the middle.
# Popularized by Leslie Smith (fast.ai) for CNN training.
import numpy as np, matplotlib.pyplot as plt

def one_cycle(t, total_steps, peak_lr, start_lr=1e-6, end_lr=1e-8):
    half = total_steps // 2
    if t < half:
        # linear ramp up
        return start_lr + (peak_lr - start_lr) * t / half
    else:
        # cosine anneal down
        progress = (t - half) / (total_steps - half)
        return end_lr + (peak_lr - end_lr) * 0.5 * (1 + np.cos(np.pi * progress))

TOTAL = 10000
ts = np.arange(TOTAL)
lrs = [one_cycle(t, TOTAL, peak_lr=1e-2) for t in ts]

plt.figure(figsize=(7.5, 3.4))
plt.plot(ts, lrs, color="crimson", lw=2)
plt.xlabel("step"); plt.ylabel("learning rate")
plt.title("One-cycle policy: ramp up, then anneal down")
plt.grid(alpha=0.3); plt.tight_layout()

print("One-cycle schedule chosen values:")
for i in [0, 1000, 4000, 5000, 6000, 9000, TOTAL - 1]:
    print(f"  step {i:>5}: lr = {lrs[i]:.6f}")`;

<Motivation title="Every LLM training run uses linear warmup + cosine decay">

Every modern LLM pretraining paper reports something like: "Linear warmup for the first 2000 steps, then cosine decay to 10% of peak over the remaining 500k steps." That single sentence encodes a lot of hard-won knowledge:

- **Warmup** prevents early-training instability when the optimizer state (Adam's m and v) hasn't converged yet.
- **Cosine decay** smoothly reduces the learning rate over training, letting the model settle into a good minimum without hard "cliffs" that could destabilize.
- **The specific values** (2000 warmup steps, 10% floor) are the result of extensive empirical tuning at scale.

Why not just use a constant learning rate? Because:

1. **Early training** benefits from a small learning rate to avoid destabilizing the untrained model.
2. **Middle training** benefits from a high learning rate to make fast progress on the loss.
3. **Late training** benefits from a small learning rate to fine-tune into a sharp minimum.

A schedule captures all three phases in a single training run. This chapter shows the four schedules you'll actually see, explains why linear warmup is essential for transformer training, and gives you the vocabulary to read (and design) an LLM training config.

</Motivation>

<Intuition title="Learning rate is a knob you turn over time — warmup ramps it up, decay ramps it down">

The learning rate η controls the size of each parameter update. Too high and updates overshoot (loss diverges); too low and training crawls (loss barely moves).

**The ideal η changes over training:**

- **Steps 0-1000** (warmup phase): η starts at ~0 and ramps up linearly to a peak. Reason: at initialization the model produces near-random outputs; gradients are large and often unrepresentative of the eventual "true" gradients. Adam's variance estimate v is still noisy. Small η protects against exploding early updates.
- **Steps 1000-90% of training** (peak): η stays at its peak value. This is where most of the training happens and most of the loss decrease occurs.
- **Last 10-50% of training** (decay): η smoothly decays toward zero (or a small floor). Reason: as the model converges, small refinements matter; large steps would overshoot the minimum.

Cosine decay is the smoothest way to accomplish phases 2-3 in one continuous curve. Linear warmup handles phase 1. Together: **warmup + cosine** — the shape of every LLM training run's learning-rate curve.

</Intuition>

<Recall term="Learning rate">

The learning rate η is the scalar in the update rule θ ← θ − η · direction. It multiplies the update magnitude, controlling how big a step to take at each iteration. Too big → divergence; too small → slow convergence; just right → fast, stable descent. The "right" value depends on the loss landscape, the optimizer, and the current stage of training — hence the need for schedules.

</Recall>

## The four schedules you'll see

<RunnableCode title="Four learning-rate schedules on a common axis" code={schedulesCode} packages={["numpy", "matplotlib"]} captureFig={true} />

**1. Step decay** — drop the LR by a fixed factor (e.g., 10×) at pre-specified milestones. Original workhorse for image classification (ResNet trained for 100 epochs with drops at 30, 60, 90). Simple, effective, but has "cliffs" that can shock the optimizer.

**2. Exponential decay** — η_t = η_0 · γ^t for γ slightly less than 1 (e.g., 0.9995). Smooth, predictable, but requires tuning γ and can be too aggressive (goes to zero) or too gentle (barely decays).

**3. Cosine annealing** — η_t = η_min + 0.5 · (η_max − η_min) · (1 + cos(π · progress)). Popular in vision (SGDR paper), transformers, and many other setups. Smooth, one hyperparameter (min lr, often 0 or 10% of peak).

**4. Cosine + linear warmup** — start at 0, ramp linearly to peak over warmup_steps, then cosine decay from peak to floor over remaining_steps. **The default for transformer pretraining and fine-tuning.**

## Why warmup — the Adam+transformer story

**Warmup** is the linear ramp of learning rate from 0 to the peak over the first few hundred to few thousand steps. It's not decorative — it's essential for large-scale transformer training. Two reasons:

- **Adam's variance estimate v isn't warm at initialization.** With β2 = 0.999, v needs ~1000 steps to converge. Early v estimates are unreliable, so √v is an unreliable denominator, so the effective per-parameter learning rate can be wildly wrong. Damping the actual learning rate during this warmup period gives v time to stabilize.
- **The untrained model's gradients are large and non-representative.** At initialization, the model outputs are near-random; large loss values produce large gradients. Applying full learning rate immediately can destabilize weights before they even start converging to sensible values.

Empirically, transformer training WITHOUT warmup often diverges (loss → NaN) in the first 100-500 steps. Warmup fixes this reliably.

Standard warmup lengths for transformer training:

- **Small models (< 1B params)**: 500-1000 steps
- **Medium models (1B-10B)**: 1000-2000 steps
- **Large models (10B+)**: 2000-4000 steps
- **Fine-tuning (LoRA, RLHF)**: 100-500 steps (already-warmed optimizer state helps)

## Cosine decay — the smooth ramp-down

**Cosine annealing** is the standard decay schedule after warmup. The formula:

$$
\eta_t \;=\; \eta_{\min} + \frac{1}{2}\left(\eta_{\max} - \eta_{\min}\right)\left(1 + \cos\left(\pi \cdot \frac{t - \text{warmup}}{T - \text{warmup}}\right)\right)
$$

Properties:

- **Smooth**: no sudden drops (unlike step decay), so the optimizer's momentum and variance estimates don't get shocked.
- **Zero derivative at endpoints**: dη/dt = 0 at t = warmup and t = T. Smooth entry from warmup, smooth exit at the end.
- **Half-cycle by default**: cosine goes from 1 to −1 over π; we use half a cycle, so cos goes from 1 to −1 and (1 + cos)/2 goes from 1 to 0.

**η_min (the floor)** is usually 0 for training from scratch and 10% of peak for fine-tuning. A non-zero floor prevents "learning stops entirely" at the very end and can help with stability.

**Cosine with restarts (SGDR)**: instead of a single cosine over the full training, use multiple cosine cycles with warm restarts. Useful for iterative fine-tuning where you want to "shake" the model out of a local minimum periodically. Less common in LLM pretraining but appears in some vision setups.

## One-cycle schedule (Smith / fast.ai)

For shorter training runs (especially CNN training on datasets like CIFAR/ImageNet), **one-cycle** is a popular alternative: linear ramp up to a HIGH peak learning rate, then annealed decay back down.

<RunnableCode title="One-cycle schedule: ramp up then anneal down" code={oneCycleCode} packages={["numpy", "matplotlib"]} captureFig={true} />

The key insight (per Leslie Smith's papers): using a much larger peak LR than usual (e.g., 1e-2 vs 1e-3) works if you spend enough time ramping up to it and decaying from it. This can substantially reduce training time (measured in epochs) while achieving similar or better final accuracy — a phenomenon called "super-convergence."

One-cycle is dominant in some vision and short-training-run recipes. Less common in LLM pretraining because pretraining runs are longer and the "ramp UP high" idea doesn't extrapolate to million-step runs. But **fine-tuning** with one-cycle is a viable alternative to warmup+cosine.

## The traps that matter

**Warmup steps depend on total steps, not on epochs.** A 1000-step warmup is meaningful for a 100k-step run but overwhelms a 5000-step fine-tuning run. Rule of thumb: warmup should be 1-5% of total training steps for pretraining, or a fixed 100-500 for fine-tuning.

**Cosine decay expects to reach a specific end point.** If you cut training short (early stopping), the LR has only decayed partially — sometimes you're stopping at 30% of peak instead of the intended near-zero. Adjust `T` (total steps) to match your actual training length, not a wishful "if we had time" target.

**"Constant learning rate after warmup" is a valid option for some setups.** Some research reports use warmup followed by constant LR (no decay). This often works fine for shorter runs; longer runs benefit from decay to squeeze out the last percent of loss reduction.

**Learning rate scheduling interacts with batch size scheduling.** Larger batch sizes (per the linear scaling rule) permit larger learning rates. If you increase batch size mid-training, you may need to boost LR to match. Google's "batch scaling" recipes couple these two.

**Warmup helps other optimizers too, not just Adam.** SGD with momentum on very large models also benefits from warmup, though less dramatically. If in doubt, add warmup.

**LR schedule visualization is your friend.** Before starting a long training run, plot the LR schedule for the full expected training length. Confirm the peak, the warmup shape, the decay curve, and the final value all match your intentions. A schedule bug at step 0 is a wasted million-step run.

<RealWorldML title="Reading an LLM training config">

Here's a typical config for training a mid-size transformer (~7B params):

```
optimizer:
  name: AdamW
  lr: 3e-4                    # peak learning rate
  betas: [0.9, 0.95]
  weight_decay: 0.1
  eps: 1e-8

scheduler:
  name: cosine
  warmup_steps: 2000
  total_steps: 500_000
  min_lr_ratio: 0.1           # cosine anneals to 10% of peak, not 0

grad_clip: 1.0
```

Decoding the scheduler:

- **warmup_steps = 2000** out of 500,000 = **0.4% warmup**. Fits the 1-5% rule of thumb; the low end because Adam with β2 = 0.95 stabilizes faster than β2 = 0.999.
- **cosine** decay starts after step 2000 and continues to step 500,000. LR at step 250,000 (halfway through cosine phase, so cos = 0): (min_lr + (peak_lr − min_lr) * 0.5) = peak * (0.1 + 0.9 * 0.5) = peak * 0.55 ≈ 1.65e-4.
- **min_lr_ratio = 0.1** means the schedule anneals to 10% of peak (3e-5), not zero. This maintains a nonzero LR even at the end so the optimizer keeps making small adjustments — helps with stability at the tail.
- **grad_clip = 1.0** rescales the gradient if its norm exceeds 1.0. Independent of the LR schedule but complementary — prevents rare huge gradient events from blowing up the model despite the LR being reasonable.

Change the warmup steps to 0: the run will likely NaN in the first 500 steps. Change min_lr_ratio to 0: the run may plateau at a slightly worse loss because updates go to zero at the end. Change the schedule to constant: probably works, but you're leaving some final-loss improvement on the table.

Every value here is negotiable within a range, but the *shape* — warmup + cosine + grad clip — is what the field has converged on for LLM pretraining. Fine-tuning uses the same shape but shorter (500 warmup, cosine over ~10k steps).

</RealWorldML>

2. **Visualize different warmup lengths.** For a 10,000-step training run with peak LR 3e-4, plot the LR schedule for warmup = 100, 500, 1000, 2000 steps (all with cosine decay to 0 afterward). Notice how larger warmup shifts the peak later and makes the schedule "gentler" early on.

<Summary>

- **Learning rate schedules** vary η over the course of training: small at start (warmup), high in the middle, decay toward end.
- **Linear warmup** (0 → peak over first ~1-5% of steps): prevents early-training instability from unwarmed Adam variance estimate and extreme initial gradients. **Essential for transformer training.**
- **Cosine annealing** (peak → floor via cosine curve over remaining steps): smoothest way to decay LR; standard for LLM training. Formula: `η(t) = min_lr + (peak_lr - min_lr) * 0.5 * (1 + cos(π * progress))`.
- **Standard LLM recipe**: linear warmup + cosine decay to 10% of peak + gradient clipping at 1.0. Every serious pretraining run uses this shape.
- **One-cycle** (ramp UP to a HIGH peak, then anneal DOWN): alternative for shorter runs and CNN training; less common in LLM pretraining.
- **Step decay and exponential decay** are older, still used in some vision setups but largely replaced by cosine.
- **Warmup length rule of thumb**: 1-5% of total steps for pretraining; fixed 100-500 for fine-tuning.

</Summary>
