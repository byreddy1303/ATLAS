export const batchVsMiniVsStochasticCode = `# Three flavors of gradient descent on the same linear-regression problem.
# The exact same loss surface -- three different rules for computing the gradient.
import numpy as np, matplotlib.pyplot as plt

np.random.seed(0)
N = 500
X = np.random.randn(N, 1)
y = 3 * X.ravel() + 1 + 0.5 * np.random.randn(N)   # true w = 3, b = 1
w0 = np.array([-2.0, -2.0])    # start at (-2, -2)

def loss(w):
    return 0.5 * np.mean((X @ w[:1] + w[1] - y) ** 2)

def grad(w, idxs):
    "Gradient on the subset of rows given by idxs."
    Xi, yi = X[idxs], y[idxs]
    pred = Xi @ w[:1] + w[1]
    err  = pred - yi
    dw = (Xi * err[:, None]).mean(axis=0)
    db = err.mean()
    return np.array([dw[0], db])

def gd(kind, steps=60, lr=0.1, batch=16, seed=1):
    rng = np.random.default_rng(seed)
    w = w0.copy(); traj = [w.copy()]
    for _ in range(steps):
        if kind == "batch":     idxs = np.arange(N)                    # all N rows
        elif kind == "mini":    idxs = rng.choice(N, size=batch, replace=False)
        elif kind == "stoch":   idxs = np.array([rng.integers(0, N)])  # exactly 1 row
        w = w - lr * grad(w, idxs); traj.append(w.copy())
    return np.array(traj)

trajs = {"batch (all N)": gd("batch"), "mini-batch (16)": gd("mini"), "stochastic (1)": gd("stoch")}

# Plot contour + trajectories
ws = np.linspace(-2.5, 4, 60); bs = np.linspace(-2.5, 2.5, 60)
WW, BB = np.meshgrid(ws, bs)
Z = np.zeros_like(WW)
for i in range(WW.shape[0]):
    for j in range(WW.shape[1]):
        Z[i, j] = loss(np.array([WW[i, j], BB[i, j]]))

fig, axes = plt.subplots(1, 3, figsize=(13, 3.6))
for ax, (name, tr) in zip(axes, trajs.items()):
    ax.contour(WW, BB, Z, levels=15, colors="teal", alpha=0.35, linewidths=0.7)
    ax.plot(tr[:, 0], tr[:, 1], "-o", markersize=3, color="crimson", lw=1.3)
    ax.plot(3, 1, "*", color="gold", markersize=14, label="true (3, 1)")
    ax.set_title(f"{name}"); ax.set_xlabel("w"); ax.set_ylabel("b"); ax.grid(alpha=0.3)
plt.tight_layout()

for name, tr in trajs.items():
    print(f"{name:>17}: final (w, b) = ({tr[-1, 0]:+.3f}, {tr[-1, 1]:+.3f})   loss = {loss(tr[-1]):.4f}")`;

export const momentumVsSgdCode = `# Momentum vs plain SGD on a stretched quadratic -- the CANONICAL case where
# plain SGD wobbles down the narrow valley while momentum sails through.
import numpy as np, matplotlib.pyplot as plt

# Loss: 0.5 * (a * x^2 + y^2)   with a >> 1  -- long narrow valley in y
a = 20.0
def f(v):    return 0.5 * (a * v[0] ** 2 + v[1] ** 2)
def grad(v): return np.array([a * v[0], v[1]])

def sgd(x0, lr=0.05, steps=60):
    v = x0.copy(); traj = [v.copy()]
    for _ in range(steps):
        v = v - lr * grad(v); traj.append(v.copy())
    return np.array(traj)

def momentum(x0, lr=0.05, beta=0.9, steps=60):
    "v_t = beta * v_{t-1} + grad;   x_t = x_{t-1} - lr * v_t"
    v = np.zeros_like(x0)
    x = x0.copy(); traj = [x.copy()]
    for _ in range(steps):
        v = beta * v + grad(x)
        x = x - lr * v
        traj.append(x.copy())
    return np.array(traj)

def nesterov(x0, lr=0.05, beta=0.9, steps=60):
    "Nesterov look-ahead: evaluate gradient at (x - lr*beta*v) instead of at x."
    v = np.zeros_like(x0)
    x = x0.copy(); traj = [x.copy()]
    for _ in range(steps):
        lookahead = x - lr * beta * v
        v = beta * v + grad(lookahead)
        x = x - lr * v
        traj.append(x.copy())
    return np.array(traj)

x0 = np.array([1.0, 1.0])
trajs = {"plain SGD": sgd(x0), "momentum (b=0.9)": momentum(x0), "Nesterov": nesterov(x0)}

# Plot contours + each trajectory
xs, ys = np.meshgrid(np.linspace(-1.2, 1.2, 100), np.linspace(-1.2, 1.2, 100))
Z = 0.5 * (a * xs ** 2 + ys ** 2)

fig, axes = plt.subplots(1, 3, figsize=(13, 3.4))
for ax, (name, tr) in zip(axes, trajs.items()):
    ax.contour(xs, ys, Z, levels=15, colors="teal", alpha=0.4, linewidths=0.7)
    ax.plot(tr[:, 0], tr[:, 1], "-o", markersize=3, color="crimson", lw=1.3)
    ax.plot(0, 0, "*", color="gold", markersize=14)
    ax.set_title(f"{name} (final loss {f(tr[-1]):.4f})")
    ax.set_aspect("equal"); ax.grid(alpha=0.3); ax.set_xlim(-1.2, 1.2); ax.set_ylim(-1.2, 1.2)
plt.tight_layout()

for name, tr in trajs.items():
    print(f"{name:>17}: final loss = {f(tr[-1]):.6f}")`;

export const noiseAsFeatureCode = `# Two-well loss -- a shallow bad basin and a deeper good basin.
# Show that mini-batch NOISE occasionally hops from the shallow basin to the deep one.
import numpy as np

def f(x):
    "Two wells: shallow at x=-1 (depth 0.5), deep at x=+1.2 (depth 2.0)."
    left  = 0.5 * (x + 1.0) ** 2
    right = 2.0 * (x - 1.2) ** 2
    return np.minimum(left, right)

def dfdx(x):
    left_slope  = (x + 1.0)
    right_slope = 4.0 * (x - 1.2)
    return np.where(0.5 * (x + 1.0) ** 2 <= 2.0 * (x - 1.2) ** 2, left_slope, right_slope)

def noisy_gd(x0, sigma=0.0, lr=0.05, steps=400, seed=0):
    rng = np.random.default_rng(seed)
    x = x0
    for _ in range(steps):
        noise = rng.normal(0, sigma)                  # SGD-shaped gradient noise
        x = x - lr * (dfdx(x) + noise)
    return x

# Start in the SHALLOW well; is enough noise required to escape to the deep well?
outcomes = {"no noise": [], "moderate noise (sigma=0.5)": [], "high noise (sigma=1.5)": []}
for seed in range(50):
    outcomes["no noise"].append(noisy_gd(-0.9, sigma=0.0, seed=seed))
    outcomes["moderate noise (sigma=0.5)"].append(noisy_gd(-0.9, sigma=0.5, seed=seed))
    outcomes["high noise (sigma=1.5)"].append(noisy_gd(-0.9, sigma=1.5, seed=seed))

for name, finals in outcomes.items():
    escaped = sum(1 for x in finals if x > 0)
    print(f"{name:>28}:  {escaped:>2}/50 escaped to the deep well ({100*escaped/50:.0f}%)")
print()
print("SGD noise is a FEATURE, not a bug: it lets the optimizer escape shallow basins")

<Motivation title="Real neural nets are trained with mini-batch SGD + momentum">

Every deep-learning paper describes a training loop that basically looks like this:

```
for step in range(N):
    batch = sample_from_dataset(size=B)
    loss = forward(model, batch)
    grads = backward(loss)
    optimizer.step(grads)      # momentum or Adam, not plain GD
```

This is called **mini-batch stochastic gradient descent with momentum**. Long name, three ideas:

- **Mini-batch** = "use a small chunk of the data at a time, not the whole thing."
- **Stochastic** = "the gradient we compute is noisy because it's based on a random chunk, not everything."
- **Momentum** = "keep a running memory of past gradient directions."

Those three words hide four questions that shape every training run — how much data per step, how gradient estimates combine over time, how much noise you tolerate, and how learning rate and batch size interact.

For a GenAI engineer, this chapter answers:

1. **Why not use the whole dataset for every gradient step?** Full-batch gradient descent works nicely in theory, but each step is way too expensive on large datasets. Also, the lack of noise turns out to be a *bug*, not a feature, in high-dimensional non-convex problems.
2. **What does momentum actually do?** It speeds up progress along consistent directions and dampens oscillations in noisy or badly-scaled ones. The single biggest upgrade over plain SGD.
3. **How do batch size and learning rate interact?** A rule called the **linear scaling rule** tells you how to change the learning rate when you change batch size to keep training dynamics constant. Get this wrong and you can burn a lot of compute on a divergent run.
4. **Why does small-batch SGD sometimes generalize better than big-batch?** Because the noise in noisy gradients quietly nudges the optimizer toward flatter (more robust) minima — a lucky accident that turns out to be a feature.

Adam and its family (next chapter) are built on top of everything you learn here.

</Motivation>

<Intuition title="Batch = accuracy of the gradient estimate; momentum = memory of the direction we were going">

Imagine walking down a valley in fog. You can only feel the slope where you're standing. That's your **gradient** — a local direction of steepest descent.

**Batch size** is how much ground you probe before choosing which way to step.

- **Full-batch GD (B = N):** you carefully survey every square meter of the valley, average the slopes, then step. Slow but the direction is precise.
- **Mini-batch SGD (B ≈ 32–2048):** you probe a random sample, average their slopes, then step. Approximate direction, but a hundred times cheaper.
- **Pure SGD (B = 1):** you step based on the slope at literally one point. Very cheap, very noisy — often overshoots.

**Momentum** is the memory of which way you've been heading. Instead of only using the current slope, you also carry a **velocity** — an exponentially-weighted moving average of past gradients. When the current slope agrees with recent history, the velocity grows and you accelerate. When it disagrees (you're wobbling side-to-side in a narrow ravine), consecutive gradients cancel out in the velocity, and only the along-ravine component builds up. That's why momentum is dramatic on badly-conditioned loss surfaces: it turns wobbly zig-zags into a smooth glide.

</Intuition>

<Recall term="Gradient">

The gradient of a scalar function `f(x)` at a point `x` is the vector `∇f(x)` whose components are the partial derivatives ∂f/∂xᵢ. It points in the direction of steepest ascent — so `-∇f(x)` is the direction of steepest descent. Every gradient-based optimizer is a rule for computing an update from (a copy of, a smoothed version of, an adaptive rescaling of) the negative gradient.

</Recall>

## Batch, mini-batch, stochastic — the size spectrum

**Full-batch gradient descent (B = N).** Compute the loss and gradient on **all** training examples, then take one step:

$$
\theta \leftarrow \theta - \eta \cdot \nabla \frac{1}{N}\sum_{i=1}^N \ell(\theta; x_i, y_i)
$$

Each step is expensive but the gradient is the "true" one (up to numerical precision). Convergence is monotonic on convex problems.

**Pure stochastic gradient descent (B = 1).** Draw a single random example each step:

$$
\theta \leftarrow \theta - \eta \cdot \nabla \ell(\theta; x_i, y_i) \quad i \sim \text{Unif}(1..N)
$$

Cheap per step, but the direction is noisy. Convergence is not monotonic — the loss goes up and down as different examples pull in different directions.

**Mini-batch SGD (B in between, e.g., 16–2048).** Draw a random mini-batch of size B:

$$
\theta \leftarrow \theta - \eta \cdot \nabla \frac{1}{B}\sum_{i \in \mathcal{B}} \ell(\theta; x_i, y_i)
$$

This is what everyone uses in practice. The mini-batch gradient is an **unbiased estimate** of the full-batch gradient with variance `~ σ² / B`. Larger B ⇒ smaller variance ⇒ closer to full-batch behavior; smaller B ⇒ noisier but cheaper.

**Pitfall — B = 1 is rarely used in practice.** With B = 1 you get maximum noise, but you also get maximum wall-clock cost per example: modern GPUs are designed to process a batch in parallel, and B = 1 wastes all that hardware. The sweet spot for GPU utilization is usually B ≥ 32 — batches small enough to fit in memory and get useful noise, big enough to keep the accelerator busy.

## Momentum — memory of the direction we were going

The simplest momentum update:

$$
v_t = \beta \cdot v_{t-1} + \nabla f(\theta_{t-1})
$$
$$
\theta_t = \theta_{t-1} - \eta \cdot v_t
$$

with `β ∈ [0, 1)` — typically 0.9. Here `v_t` is a **velocity vector** that accumulates gradients over recent steps. When consecutive gradients agree, `v` grows and the effective step size along that direction gets large. When they disagree (bouncing side to side in a narrow valley), the disagreeing components cancel out in the sum and only the persistent direction survives.

Two ways to think about `β`:

- **Exponentially-weighted moving average (EWMA).** The velocity is a weighted average of recent gradients with weight `β^k` on the gradient from k steps ago. Effective "memory" length is roughly `1 / (1 − β) = 10` steps at β = 0.9.
- **Physical momentum.** In physics `p = m v`; the update is like Newtonian dynamics with friction `(1 − β)`. Higher `β` = lower friction = the ball keeps rolling further.

**Pitfall — the "SGD with momentum" of PyTorch is not the update above.** PyTorch's implementation is `v_t = β·v_{t-1} + ∇f(θ)`, and `θ ← θ − η·v_t` — but many textbooks write `v_t = β·v_{t-1} + (1−β)·∇f(θ)` (an EWMA with weight `1-β`). The two agree up to a factor of `1/(1−β)` on the effective learning rate. Always check whether your framework's momentum "corrects" the update magnitude or not — the effective LR you set is off by that factor if you swap between the two.

## Nesterov — look-ahead momentum

Nesterov's twist: evaluate the gradient at the **look-ahead** point `θ − η·β·v` rather than at `θ`:

$$
v_t = \beta \cdot v_{t-1} + \nabla f(\theta_{t-1} - \eta \beta v_{t-1})
$$
$$
\theta_t = \theta_{t-1} - \eta \cdot v_t
$$

The intuition: momentum will carry us to `θ − η·β·v` even before we compute a new gradient, so let's use the gradient **at that upcoming location** to correct our velocity. Empirically, Nesterov is often ~10-20% faster than plain momentum on convex problems and gives the celebrated `O(1/k²)` convergence rate — twice as fast in exponent as plain GD's `O(1/k)`.

On non-convex deep learning problems the theoretical advantage doesn't cleanly apply, but Nesterov is a standard option in PyTorch/JAX (`nesterov=True` on the SGD optimizer). Adam's success has meant that Nesterov is used less than 10 years ago, but it remains a solid choice for convex ML and CNN training.

## The generalization value of noise

A perhaps-surprising empirical fact: on nonconvex neural net losses, **smaller batch sizes often generalize better** than larger ones — the same final training loss can produce different test loss depending on batch size.

The current best explanation: gradient noise from small batches biases the optimizer toward **flat minima** — regions of the loss surface where the loss changes slowly in any direction. Flat minima have low **sharpness**, which correlates with better generalization: a slightly-perturbed model doesn't lose much accuracy. Sharp minima are precise fits to training data that break under any weight perturbation.

**Where large batches help.** For massively-parallel systems, larger batches make hardware more efficient — you spend less time on communication overhead. Large-batch training (B ≈ 32k, sometimes 1M+) is standard for LLM pretraining because compute is bought in GPU-hours, not in per-batch time. The **linear scaling rule** (Goyal et al. 2017) says: when you multiply batch size by k, multiply LR by k, and training dynamics stay similar. Coupled with linear warmup (LR schedules chapter), this is how modern very-large-batch training works.

## Choosing batch size — practical rules

Three factors set the mini-batch size in practice:

1. **Memory budget.** Batch × sequence-length × parameter-count must fit in GPU memory. You'll often see `B = 4` on a single GPU with `sequence_len = 8k`, then **gradient accumulation** simulate `B = 128` by summing gradients over 32 mini-forward-passes before the optimizer step.
2. **Throughput.** GPUs are amortized over batch dimension; too small a batch wastes cycles. The sweet spot is the largest B that fits in memory while leaving room for activations.
3. **Generalization.** For small models, small B (32–256) often gives better test loss. For LLM pretraining, generalization from batch-size differences is much less important than the compute efficiency of very large B (32k–1M+).

**Linear scaling rule.** If a batch-size-256 run uses LR 3e-4, then batch-size-2048 should use LR 3e-4 × 8 = 2.4e-3. Combine with linear warmup to avoid the extra-large early-step blowup.

<RealWorldML title="Batch size and gradient accumulation in an LLM training config">

Look at any modern LLM pretraining config (Llama, GPT, Mistral) and you'll see three related fields:

- **`per_device_train_batch_size`** — the batch that fits in one GPU's memory (e.g., 4 or 8).
- **`gradient_accumulation_steps`** — how many mini-batches to combine before an optimizer step (e.g., 32 or 128).
- **`num_devices`** — how many GPUs run in data-parallel (e.g., 8 or 1024).

The **effective batch size** is the product: `per_device × accumulation × num_devices`. A typical Llama-style config: `4 × 128 × 128 = 65,536` tokens per effective batch. That's the number the LR was tuned for and the linear scaling rule applies to.

**Why so large?**

- **Compute efficiency:** at LLM scale, the gradient is very high-dimensional; noise from very small batches is enough to dominate the signal. Big batches give clean signals that let LR schedules do their work.
- **Communication amortization:** in data-parallel training across 1000+ GPUs, gradient all-reduce is expensive per step. Larger batches mean fewer steps per epoch and less all-reduce overhead.

**Gradient accumulation is the "small hardware, big batch" trick.** If you only have 8 GPUs but the model was tuned for a 65k-token batch, use `accumulation = 128` to get 8 × 128 = 1024 mini-batches summed per step. Compute-equivalent to running on 1024 GPUs (just slower).

**Where small batches still win:** vision fine-tuning, non-LLM downstream tasks with small datasets, RL with actor-critic where trajectory variance dominates. In those regimes, B = 32-256 is standard because the noise vs efficiency trade-off tilts the other way.

</RealWorldML>

<Summary>

- **Batch spectrum:** full-batch GD (all N examples, precise but expensive) → mini-batch SGD (B in 16-2048, the practical default) → pure SGD (B=1, cheap but noisy).
- **Mini-batch gradient is an unbiased estimate** of the full-batch gradient with variance ~ 1/B.
- **Momentum** = velocity vector accumulating recent gradients: `v = β·v + ∇f`, `θ = θ − η·v`. Accelerates along consistent directions, damps oscillations.
- **Nesterov** = momentum with a look-ahead gradient: evaluate ∇f at the predicted-future position, not the current one. Provable O(1/k²) on convex problems.
- **Noise is a feature.** Small-batch SGD explores flatter minima that generalize better; noise helps escape shallow local pits and saddles.
- **Linear scaling rule:** batch × k means LR × k. Combined with warmup, this scales training to massive parallel systems.
- **Gradient accumulation** simulates a large batch on small hardware: sum k mini-batch gradients before an optimizer step.
- **In LLM pretraining, effective batch is 32k-1M+ tokens** — chosen for hardware efficiency and clean gradients, not the flat-minima effect.

</Summary>
