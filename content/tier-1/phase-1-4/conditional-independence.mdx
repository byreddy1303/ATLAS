export const conditionalCode = `# Conditional probability: P(A | B) = P(A and B) / P(B).
# Meaning: restrict the sample space to the outcomes where B is true,
# then measure how big A is inside that restricted world.
import numpy as np
rng = np.random.default_rng(0)

# Two fair dice.  Sample space = 36 ordered pairs.  Uniform measure.
n_trials = 200_000
d1 = rng.integers(1, 7, size=n_trials)
d2 = rng.integers(1, 7, size=n_trials)
S = d1 + d2

# Event B = "first die is 6".  Event A = "sum > 10".
B = d1 == 6
A = S > 10

p_A     = A.mean()
p_B     = B.mean()
p_AandB = (A & B).mean()
p_A_given_B = p_AandB / p_B

print(f"P(A)        = P(sum > 10)          = {p_A:.4f}   (analytical 3/36 = 0.0833)")
print(f"P(B)        = P(first die = 6)     = {p_B:.4f}   (analytical 1/6  = 0.1667)")
print(f"P(A and B)  = P(sum > 10 AND d1=6) = {p_AandB:.4f}   (analytical 2/36 = 0.0556)")
print(f"P(A | B)    = P(sum > 10 | d1=6)   = {p_A_given_B:.4f}   (analytical 2/6  = 0.3333)")

# Direct verification: among trials where d1 == 6, what fraction has S > 10?
p_direct = A[B].mean()
print(f"P(A | B) computed directly by restricting Omega to B: {p_direct:.4f}")
print("Both methods agree -- 'condition on B' means 'zoom in on the subset where B holds'.")`;

export const independenceCode = `# Two events A, B are independent iff  P(A and B) = P(A) * P(B).
# Equivalently, P(A | B) = P(A) -- knowing B tells you nothing about A.
import numpy as np
rng = np.random.default_rng(0)

# Case 1: two independent coin flips.
n = 200_000
f1 = rng.integers(0, 2, size=n)
f2 = rng.integers(0, 2, size=n)

A = f1 == 1
B = f2 == 1
p_A, p_B = A.mean(), B.mean()
p_AB = (A & B).mean()
print("Case 1 -- two independent flips:")
print(f"  P(A)*P(B) = {p_A * p_B:.4f}")
print(f"  P(A and B)= {p_AB:.4f}   (matches -- independent)")

# Case 2: dependent -- both flips share a hidden coin bias.
# Draw a bias Z; then both flips use it.  A and B are marginally identical to case 1
# but strongly correlated because they share Z.
Z = rng.beta(2, 2, size=n)         # random bias per trial
f1 = (rng.random(n) < Z).astype(int)
f2 = (rng.random(n) < Z).astype(int)

A = f1 == 1
B = f2 == 1
p_A, p_B = A.mean(), B.mean()
p_AB = (A & B).mean()
print("\\nCase 2 -- coupled flips (shared hidden bias Z):")
print(f"  P(A)*P(B) = {p_A * p_B:.4f}")
print(f"  P(A and B)= {p_AB:.4f}   (LARGER -- events are dependent)")

# Case 3: independence CAN return conditional on Z.  Given Z, the two flips are independent.
# This is 'conditional independence' -- the pattern behind Naive Bayes and hierarchical models.
mask = (rng.random(n) < 0.4)       # sub-sample where Z is roughly fixed
print("\\nCase 3 -- given (roughly fixed) Z, the same coupled flips look independent:")
sub_Z = Z[mask]
f1s = (rng.random(len(sub_Z)) < sub_Z).astype(int)
f2s = (rng.random(len(sub_Z)) < sub_Z).astype(int)
A = f1s == 1
B = f2s == 1
print(f"  P(A|Z)*P(B|Z) approx {A.mean() * B.mean():.4f}")
print(f"  P(A and B | Z)        {(A & B).mean():.4f}")`;

export const chainRuleCode = `# Chain rule for probability:
#   P(A, B, C) = P(A) * P(B | A) * P(C | A, B)
# This is how every autoregressive language model computes P(sequence).
import numpy as np

# Small hand-designed joint distribution over three binary variables A, B, C.
# We'll fill in a full joint table, then factor it via the chain rule and verify.
joint = np.zeros((2, 2, 2))
joint[0, 0, 0] = 0.10
joint[0, 0, 1] = 0.05
joint[0, 1, 0] = 0.15
joint[0, 1, 1] = 0.20
joint[1, 0, 0] = 0.05
joint[1, 0, 1] = 0.10
joint[1, 1, 0] = 0.15
joint[1, 1, 1] = 0.20
assert np.isclose(joint.sum(), 1.0)

# Marginals and conditionals from the joint.
p_A = joint.sum(axis=(1, 2))                 # P(A)
p_AB = joint.sum(axis=2)                     # P(A, B)
p_B_given_A = p_AB / p_A[:, None]            # P(B | A)
p_C_given_AB = joint / p_AB[:, :, None]      # P(C | A, B)

# Chain-rule reconstruction.
recon = p_A[:, None, None] * p_B_given_A[:, :, None] * p_C_given_AB
print("Max absolute error between joint and chain-rule reconstruction:",
      np.abs(recon - joint).max())
print("Chain rule verified -- any joint distribution factors this way, regardless of dependence.")

# Application: the joint probability of a length-T sequence x_1, ..., x_T factors as
#   P(x_1) * P(x_2 | x_1) * P(x_3 | x_1, x_2) * ... * P(x_T | x_1, ..., x_{T-1}).
# This is exactly what an autoregressive language model computes -- one conditional
# per token, given the entire preceding context.  Log-probabilities add:
#   log P(seq) = sum_t log P(x_t | x_<t).
print("\\nAutoregressive LMs = chain-rule factorization of the joint over tokens.")`;

export const montyHallCode = `# The Monty Hall paradox: three doors, one prize.  You pick a door.
# The host (who knows) opens a losing door from the two you didn't pick.
# Should you switch to the remaining door?
# Answer: YES.  Switching wins with probability 2/3; staying wins with probability 1/3.
import numpy as np
rng = np.random.default_rng(0)
n = 100_000

# Simulate.  The prize is behind a uniformly random door.  Your first pick is
# uniform.  The host opens a losing door among the two not picked.
prize = rng.integers(0, 3, size=n)
pick  = rng.integers(0, 3, size=n)

# For each trial, decide what "switch" gives.  If pick == prize, switching loses.
# If pick != prize, switching wins (host reveals the remaining loser, forcing you
# to switch to the prize door).
wins_switch = (pick != prize).mean()
wins_stay   = (pick == prize).mean()

print(f"Empirical P(win | switch) = {wins_switch:.4f}   (theory 2/3 = 0.6667)")
print(f"Empirical P(win | stay)   = {wins_stay:.4f}   (theory 1/3 = 0.3333)")

# Why: conditioning on 'host opened door H' doesn't reset the situation to
# 50/50.  Your original pick was correct with probability 1/3; that probability
# transfers to 'you were right, and switching loses'.  The remaining 2/3 mass
# concentrates on the un-opened door.  Conditional probability, correctly applied.`;

<Motivation title="Conditioning is how models use context — from GPT to RAG to diffusion">

**Conditioning** is a fancy word for something really simple: "given that I know X, what do I now believe about Y?" It's the operation of updating your view of the world when new information arrives.

Every "the model uses context" phrase you've heard in ML is really about conditioning. The three big ones:

- **Autoregressive language models** (like GPT). "Autoregressive" means "predict the next thing given the previous things." At every position, the model computes P(next token | all previous tokens). That's a **conditional probability** — the probability of the next word, given what came before. The whole sentence's probability is built up as a product:

$$P(\text{sentence}) = P(x_1) \cdot P(x_2 \mid x_1) \cdot P(x_3 \mid x_1, x_2) \cdot \dots$$

This is called the **chain rule** of probability. Every LLM you use is quietly computing this product.

- **Conditional generation.** Class-conditional image generation ("generate a cat"), prompt-conditioned text ("write a haiku about spring"), style-conditioned diffusion ("in Van Gogh style") — all of them are learning P(output | condition) instead of just P(output). The condition changes what the model produces.

- **Feature reasoning.** "This feature is independent of that one, given the class label" is the assumption behind Naive Bayes (a simple classifier). "These two documents are independent, given the query" is what a simple RAG system assumes. **Independence** — the topic of this chapter — is a really strong claim, and it's wrong more often than you'd hope. Knowing *when* it's OK to assume, and *why* it fails, is a huge part of ML engineering.

Two operators (conditioning, independence) and one identity (the chain rule) do more work in ML than any other three pieces of probability combined. This chapter is the payoff for the sample-space vocabulary from the previous chapter.

</Motivation>

<Intuition title="Conditioning = 'zoom in on the part where B is true, and re-measure'">

Grab the rectangle picture from the last chapter. Sample space Ω is a rectangle with area 1; event B is a shape inside it.

**Conditioning on B** means: erase everything outside B, then blow up what's left so it has area 1 again. That new smaller universe (which is B itself, rescaled) is now your world. Event A's conditional probability P(A | B) is the fraction of *this new universe* that A takes up.

Formally:

$$
P(A \mid B) \;=\; \frac{P(A \cap B)}{P(B)}
$$

The vertical bar "|" is read "given." So P(A | B) is "the probability of A, given B." The formula says:

- Numerator P(A ∩ B) — the overlap of A and B before rescaling. (The ∩ means "and," or "both.")
- Denominator P(B) — the total size of B before rescaling.
- Divide, and you get "the fraction of B that's also A."

One warning: if B has probability zero, conditioning on B is undefined — you can't zoom into an empty space. (Continuous distributions have a fix for this using densities.)

**Independence** is the special case where knowing B tells you nothing new about A. Mathematically: P(A | B) = P(A). Rearranging gives the multiplicative form:

$$P(A \cap B) = P(A) \cdot P(B).$$

Read that as "the probability of both A and B happening equals the two individual probabilities multiplied." This form is nicely symmetric — either A being independent of B, or B being independent of A, gives the same equation. Independence is a two-way street.

</Intuition>

<Recall term="Event">

An event is any subset of the sample space Ω. Every probability statement is about events. Conditioning is a way of turning one probability measure (on Ω) into another (on the subset B), and independence is a *relationship* between events under a given measure.

</Recall>

## Conditional probability — restrict, then renormalize

The definition again, this time with a small worked example. Roll two fair dice. Ω has 36 equally-likely outcomes. Let

- A = &ldquo;the sum is greater than 10&rdquo; = &#123;(5,6), (6,5), (6,6)&#125;. So P(A) = 3/36 = 1/12.
- B = &ldquo;the first die shows 6&rdquo; = 6 outcomes: (6,1), (6,2), …, (6,6). P(B) = 6/36 = 1/6.
- A ∩ B = &#123;(6,5), (6,6)&#125;. P(A ∩ B) = 2/36.

The definition gives P(A | B) = (2/36) / (6/36) = **2/6 = 1/3**. Way bigger than the unconditional P(A) = 1/12. Knowing the first die is a 6 makes &ldquo;sum > 10&rdquo; much more likely.

The direct-count check: among the six outcomes where the first die is 6 (i.e., inside B), exactly two of them have sum > 10 — (6,5) and (6,6). So the fraction of B that lies inside A is 2/6 = 1/3. Same answer. **Conditioning is literally &ldquo;fraction of the sub-universe.&rdquo;**

Three properties that follow immediately:

- **P( · | B) is itself a probability measure on B.** All three Kolmogorov axioms are inherited, with Ω replaced by B.
- **P(Ω | B) = 1.** Nothing outside B contributes, and B occupies all of the conditional world.
- **P(A | Ω) = P(A).** Conditioning on the whole universe changes nothing.

## Independence: when knowing B tells you nothing about A

Two events A, B are **independent** if learning B doesn&rsquo;t change your belief about A:

$$
P(A \mid B) \;=\; P(A) \qquad \Longleftrightarrow \qquad P(A \cap B) \;=\; P(A) \cdot P(B)
$$

The multiplicative form is symmetric in A and B, which is why we say &ldquo;A and B are independent&rdquo; rather than &ldquo;A is independent of B.&rdquo; Independence is a *property of the joint measure*, not of the events themselves — the same events could be independent under one measure and dependent under another.

**Examples that are independent.** Two flips of the same fair coin. A die roll and the color of a card drawn from a separate deck. Successive draws *with replacement* from a bag.

**Examples that look independent but aren&rsquo;t.**

- Two exam scores in a class. Both are affected by the difficulty of the exam and the quality of teaching — hidden common causes create correlation.
- Two flips of a coin whose bias is unknown. Marginally each flip is 50/50, but if you see 20 heads in a row, you should now believe heads is more likely on flip 21.
- The predictions of two different LLMs on the same prompt. Both are drawing from correlated training data; they're not independent.

**Conditional independence.** A and B may be dependent overall but *become* independent when conditioned on a third variable Z. Formally:

$$
P(A \cap B \mid Z) \;=\; P(A \mid Z) \cdot P(B \mid Z)
$$

This is the assumption behind **Naive Bayes** (words are conditionally independent given the class label) and **hidden Markov models** (observations are conditionally independent given the hidden state). It's a much weaker and much more useful assumption than unconditional independence.

## The chain rule — factor any joint distribution into a product of conditionals

For any three events A, B, C:

$$
P(A \cap B \cap C) \;=\; P(A) \cdot P(B \mid A) \cdot P(C \mid A \cap B)
$$

This is just the definition of conditional probability applied twice. It generalizes: for n events,

$$
P(A_1 \cap A_2 \cap \cdots \cap A_n) \;=\; P(A_1) \cdot P(A_2 \mid A_1) \cdot P(A_3 \mid A_1, A_2) \cdots P(A_n \mid A_1, \ldots, A_{n-1})
$$

The chain rule holds for *any* joint distribution — no independence assumption needed. It's the mathematical backbone of every autoregressive model.

**Autoregressive language models are chain-rule factorizations.** P(sentence) = P(word₁) · P(word₂ | word₁) · P(word₃ | word₁, word₂) · … The model is trained to estimate each conditional; teacher-forcing at training time gives it the correct context; the sequence probability is the product (or log-sum) of conditionals. GPT, Llama, Claude — every one of them is computing a chain-rule factorization of a joint over tokens.

**When independence buys you factorization for free.** If A₁, …, A_n are mutually independent, every conditional collapses to the marginal: P(A₂ | A₁) = P(A₂), etc. The joint becomes the product of marginals. This is why independence is such a strong assumption — it turns an exponentially large joint into a linearly small product. It's also why real-world data rarely satisfies it.

## The Monty Hall paradox — conditioning done right

Three doors. One has a prize. You pick door 1. The host, who knows what's behind each door, opens a losing door among the two you didn&rsquo;t pick — say door 3. Should you switch to door 2 or stay with door 1?

The naive answer &ldquo;50/50&rdquo; is wrong. **Switching wins with probability 2/3.**

Careful conditioning: your original pick was correct with probability 1/3, so P(prize is behind door 1) = 1/3. The two doors you didn&rsquo;t pick collectively contain the prize with probability 2/3. The host&rsquo;s action of opening one of them doesn&rsquo;t redistribute that 2/3 — it *concentrates it* on the remaining unopened door. So switching gives you probability 2/3.

The point isn&rsquo;t the paradox, it's the *lesson*: conditioning on a piece of information is different from redistributing probability uniformly. When someone reveals evidence, the update depends on *how* they revealed it, not just *what* was revealed. The host&rsquo;s knowledge (they never open the prize door) is the reason switching wins. Change the setup so the host opens a random door — including possibly the prize door — and switching becomes a wash.

## The traps that matter

**Conditioning is not commutative with intersection.** P(A ∩ B | C) is not the same as P(A | B ∩ C) or P(A | B) · P(B | C). Always write out the definition and check.

**Independence is a property of the measure, not the events.** Two coins may look independent under &ldquo;fair coin&rdquo; assumption but be dependent under &ldquo;unknown-bias&rdquo; assumption. Ask &ldquo;independent under which distribution?&rdquo;

**Pairwise independence doesn&rsquo;t imply mutual independence.** Three events can be pairwise independent (P(A ∩ B) = P(A)P(B), P(A ∩ C) = P(A)P(C), P(B ∩ C) = P(B)P(C)) and yet P(A ∩ B ∩ C) ≠ P(A)P(B)P(C). Classic example: A, B are independent coin flips; C = &ldquo;A and B agree.&rdquo; Any two of &#123;A, B, C&#125; are pairwise independent, but all three aren&rsquo;t.

**Conditioning on zero-probability events is undefined.** P(A | B) = P(A ∩ B) / P(B) with P(B) = 0 gives 0/0. Continuous distributions face this constantly — you have to condition on events like &ldquo;X is in a small neighborhood of x&rdquo; and take a limit, which is what leads to the *conditional density* concept.

**Beware the base-rate fallacy — conditioning correctly on tests results.** P(disease | positive test) is not the same as P(positive test | disease). The Bayes chapter has the full worked example; the point here is that conditional probabilities are directional and asymmetric. Always ask &ldquo;which way does the conditioning bar go?&rdquo;

**Selection effects induce spurious dependence.** If you only look at data where B has been selected (survivorship bias, publication bias), the observed correlations between other variables are conditional-on-B correlations — often very different from the true marginal correlations. In ML this shows up as data-leakage bugs and cherry-picking artifacts.

<RealWorldML title="Where independence quietly powers ML — and where it breaks">

Two case studies where "assume independence" makes ML work — and where reality bites back.

**Case 1: Naive Bayes classifier.** A Naive Bayes classifier decides which class (spam vs not-spam, sports vs politics) a document belongs to. Its formula:

$$
P(\text{class} \mid w_1, \ldots, w_n) \propto P(\text{class}) \prod_i P(w_i \mid \text{class}).
$$

That product form is only correct if the words are **conditionally independent given the class**. Which they aren't! Seeing "New" makes "York" hugely more likely — they're not independent. Naive Bayes cheerfully ignores this and multiplies anyway. Somehow it still works well as a *classifier* — the correct class is usually ranked highest even when the probabilities are wrong — but the actual probability numbers are often way off (usually way too confident).

**Case 2: RAG retrieval.** In a simple **RAG** (Retrieval-Augmented Generation) system, the retriever scores each document independently against the user's query. Ranking documents by that score assumes their relevance is independent given the query — which is fine in theory, but fails hard in practice when your top-5 documents are near-duplicates. Retrieving 5 versions of the same Wikipedia paragraph gives you no more information than retrieving 1.

That's why production RAG systems have a second stage called a **re-ranker** or **diversifier**. It looks at documents *jointly* — modeling something more like P(doc_i is useful | query, docs_1..i-1) — a chain-rule factorization that gives up the independence assumption for more accuracy at the cost of more compute.

The pattern to internalize: **independence assumptions buy you speed and simplicity; dropping them buys accuracy at a cost.** The engineering skill is knowing which independence assumptions are close to true, which are wrong but useful, and which need to be dropped. Every layer of a modern GenAI stack — tokenization, attention, retrieval, decoding — makes some independence assumption and pays a specific price for it.

</RealWorldML>

1. **Card-hand conditioning.** Deal 5 cards from a standard 52-card deck. Compute P(hand has at least one Ace | hand contains the King of Spades) by simulation (100,000 hands). <Summary>

- **Conditional probability** P(A | B) = P(A ∩ B) / P(B) — restrict the sample space to B, then measure A's fraction of the restricted space.
- **P( · | B) is itself a probability measure** on B, satisfying all three Kolmogorov axioms.
- **Independence** of A and B: P(A ∩ B) = P(A) · P(B). Equivalently, P(A | B) = P(A) — knowing B tells you nothing about A.
- **Conditional independence given Z**: P(A ∩ B | Z) = P(A | Z) · P(B | Z). A weaker and more useful assumption than unconditional independence; the foundation of Naive Bayes and HMMs.
- **Chain rule** P(A₁ ∩ … ∩ Aₙ) = P(A₁) · P(A₂ | A₁) · P(A₃ | A₁, A₂) · … Holds for any joint, no independence needed. The backbone of autoregressive language models.
- **Monty Hall**: conditioning correctly on how information is revealed matters as much as what is revealed. Switching wins with probability 2/3.
- **Naive Bayes and RAG re-rankers** are two sides of the same tradeoff — cheap independence assumptions vs expensive but accurate joint modeling.

</Summary>
