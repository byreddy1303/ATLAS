from itertools import product

omega = list(product(["H", "T"], repeat=2))
print("Sample space Omega =", omega)
print("|Omega| =", len(omega))

# Event A: "at least one head"

print("\\nEvent A (at least one H) =", A)

print("\\nA union B =", A_set | B_set, "  P =", len(A_set | B_set) / len(omega))
print("A intersect B =", A_set & B_set, "  P =", len(A_set & B_set) / len(omega))

export const axiomsCode = `# The three Kolmogorov axioms, verified numerically.
# (1) P(E) is non-negative for every event E.
# (2) P(Omega) = 1.
# (3) For disjoint events, P(E1 or E2) = P(E1) + P(E2).
import numpy as np

# Roll a fair 6-sided die: sample space is {1, 2, 3, 4, 5, 6}.
outcomes = np.arange(1, 7)
probs = np.full(6, 1/6)              # uniform measure

# Axiom 1: every P(E) >= 0.  Take any subset -- probability is a sum of non-negatives.
E1 = outcomes < 3                    # event "roll < 3", i.e., outcome is 1 or 2
p_E1 = probs[E1].sum()
print(f"P(roll < 3) = {p_E1:.4f}  (non-negative? {p_E1 >= 0})")

# Axiom 2: P(Omega) = 1.
p_omega = probs.sum()
print(f"P(Omega) = {p_omega:.4f}  (= 1? {np.isclose(p_omega, 1.0)})")

# Axiom 3: disjoint additivity.  Let E1 = 'roll < 3', E2 = 'roll >= 5'.  Disjoint.
E2 = outcomes >= 5
p_E2 = probs[E2].sum()
p_union = probs[E1 | E2].sum()
print(f"\\nP(E1) + P(E2) = {p_E1:.4f} + {p_E2:.4f} = {p_E1 + p_E2:.4f}")
print(f"P(E1 or E2)   = {p_union:.4f}   (matches sum for disjoint events)")

# For NON-disjoint events, use inclusion-exclusion.
E3 = outcomes % 2 == 0               # "even"
E4 = outcomes >= 4                   # "at least 4"
p_intersect = probs[E3 & E4].sum()
p_union_inc = probs[E3].sum() + probs[E4].sum() - p_intersect
p_union_direct = probs[E3 | E4].sum()
print(f"\\nNon-disjoint case: E3 = even, E4 = at least 4")
print(f"P(E3) + P(E4) - P(E3 and E4) = {p_union_inc:.4f}")
print(f"P(E3 or E4) directly         = {p_union_direct:.4f}")`;

export const simulationCode = `# When a sample space is huge, enumeration is impractical -- use Monte Carlo simulation.
# Question: draw 5 cards from a standard 52-card deck.  What is P(at least one Ace)?
import numpy as np
rng = np.random.default_rng(0)

# Analytical answer for comparison: P(no Ace) = C(48, 5) / C(52, 5)
from math import comb
p_no_ace  = comb(48, 5) / comb(52, 5)
p_at_least_one = 1 - p_no_ace
print(f"Analytical P(at least one Ace) = {p_at_least_one:.6f}")

# Now estimate by simulation.
deck = np.arange(52)             # cards 0..51; call 0..3 the Aces
n_trials = 100_000
hits = 0
for _ in range(n_trials):
    hand = rng.choice(deck, size=5, replace=False)
    if (hand < 4).any():
        hits += 1
p_hat = hits / n_trials
print(f"Empirical    P(at least one Ace) = {p_hat:.6f}   (from {n_trials:,} trials)")
print(f"Absolute error = {abs(p_hat - p_at_least_one):.5f}")

# Vectorized version -- 100x faster.
n_trials = 1_000_000
hands = np.array([rng.choice(deck, size=5, replace=False) for _ in range(n_trials)])
p_hat_vec = (hands < 4).any(axis=1).mean()
print(f"\\nVectorized   P(at least one Ace) = {p_hat_vec:.6f}   ({n_trials:,} trials)")`;

export const llmSamplingCode = `# LLM token sampling as a probability measure over a vocabulary sample space.
# The sample space Omega is the entire vocabulary; softmax gives a probability
# measure on Omega, and sampling strategies define events we choose to sample from.
import numpy as np

# Toy vocabulary and a set of logits from the model for the next token.
vocab = ["the", "a", "cat", "dog", "runs", "sleeps", "eats", "quickly", "<end>"]
logits = np.array([2.1, 1.8, 2.9, 1.3, 0.7, 0.4, -0.2, -0.8, 0.1])

# Softmax -> probability measure P on Omega = vocab (sums to 1, each P(w) >= 0).
def softmax(x, temperature=1.0):
    x = x / temperature
    x = x - x.max()              # log-sum-exp stability
    p = np.exp(x)
    return p / p.sum()

p = softmax(logits, temperature=1.0)
print("Probability measure over vocabulary (temperature=1.0):")
for w, pw in zip(vocab, p):
    print(f"  P({w!r:>10}) = {pw:.4f}")
print(f"  sum = {p.sum():.6f}  (axiom 2: P(Omega) = 1)")

# Event: "the next token is a noun" (subset of the sample space).
noun_indices = [i for i, w in enumerate(vocab) if w in {"cat", "dog"}]
p_noun = p[noun_indices].sum()
print(f"\\nEvent 'next token is a noun': P = {p_noun:.4f}")

# top-k sampling = restrict Omega to the k most probable tokens, renormalize.
def top_k(probs, k=3):
    idx = np.argsort(probs)[-k:]
    restricted = np.zeros_like(probs)
    restricted[idx] = probs[idx]
    return restricted / restricted.sum()

p_topk = top_k(p, k=3)
print("\\ntop-3 sampling (a new measure supported on the top-3 tokens):")
for w, pw in zip(vocab, p_topk):
    if pw > 0:
        print(f"  P({w!r:>10}) = {pw:.4f}")

# Temperature changes the measure: high T flattens, low T sharpens.
for T in [0.5, 1.0, 2.0]:
    p_T = softmax(logits, temperature=T)
    entropy = -np.sum(p_T * np.log(p_T + 1e-12))
    print(f"\\nT={T}: entropy = {entropy:.3f} bits-nats  (higher = flatter distribution)")`;

<Motivation title="Every probability question starts with the same question: 'what's the universe of possible outcomes?'">

Probability is the language your model speaks — but before we can talk about probabilities, we need to be crisp about *what we're taking probabilities of*. Every softmax output, every token an LLM samples, every train/test split, every "confidence score" from a retrieval system, every data augmentation — all of it is built on top of a **sample space** (the universe of possible outcomes) and a **probability measure** (a rule assigning weight to each outcome). Get those wrong, and everything downstream is guesswork.

Two vocabulary words up front so nothing sounds scary:

- **Sample space** (usually written **Ω**, capital Greek "omega") = the set of every possible outcome. For a coin flip, Ω = &#123;heads, tails&#125;.
- **Probability measure** (usually written **P**) = a rule that assigns a number between 0 and 1 to each subset of Ω, obeying a few simple rules we'll unpack below.

For a GenAI engineer, this chapter shows up in every day of work:

- **LLM token sampling.** The vocabulary (all possible tokens the model can output) is the sample space. A **softmax** over the model's raw scores ("logits") turns them into a probability measure on that space. Every decoding trick — **top-k** (only sample from the top k tokens), **top-p** (also called nucleus sampling — only sample from the smallest set summing to probability p), **temperature** (a knob that smooths or sharpens the distribution), **greedy** (just take the top token) — is a choice of *which measure to sample from*.
- **RAG retrieval.** In a **RAG** (Retrieval-Augmented Generation) system, the sample space is your document corpus. A similarity score defines an implicit measure over it. Picking the top-N documents to feed the LLM is picking a small subset.
- **Data splits.** Train/val/test are literally a partition of your dataset into three non-overlapping pieces. The math that says "the total probability across the three pieces equals 1" is exactly the additivity axiom.
- **Evaluation.** Test-set accuracy is an empirical estimate of a true probability — a **Monte Carlo estimator** (that's the name for "sample repeatedly, average the outcomes") of how well the model works on the full population.

You cannot skip this vocabulary. Two lines of math — "Ω is the set of outcomes; P assigns a number to every subset" — do more downstream lifting than any fancy technique that comes later. The rest of Phase 1.4 sits on top of this chapter.

</Motivation>

<Intuition title="Probability = 'how much of the possible-outcome universe does this cover?'">

Here's the mental picture that makes everything click.

Draw a rectangle with area 1 on a piece of paper. That's the sample space Ω — every possible outcome of your experiment lives somewhere inside it. Now draw a shape inside the rectangle. That shape is an **event** — a subset of outcomes you care about (like "the coin lands heads" or "the roll is even"). The **probability** of that event is just the area of the shape.

From this one picture, everything else falls out:

- Areas are never negative → probabilities are never negative.
- The whole rectangle has area 1 → the probability of "something happens" is 1.
- Two shapes that don't overlap: their combined area is the sum of each area → the probability of "either event A or event B, where they don't overlap" is P(A) + P(B).

Those three facts have a fancy name — **Kolmogorov's axioms** (Andrey Kolmogorov, 1933) — but you already know them. Every probability rule you'll ever meet in ML is either these three or a direct consequence.

Whether it's Naive Bayes, transformer attention weights, or a diffusion model's noise schedule, the same picture applies. The only differences are (a) what the sample space is and (b) which measure you've put on it.

</Intuition>

<Recall term="Set operations">

Given subsets A, B of a universe Ω:

- **Union** A ∪ B — outcomes in A OR B (or both).
- **Intersection** A ∩ B — outcomes in A AND B.
- **Complement** Aᶜ — outcomes NOT in A. Equals Ω \\ A.
- **Disjoint** — A ∩ B = ∅ (empty set); the two share no outcomes.
- **Partition** — a collection of disjoint sets whose union is Ω.

Every event operation in probability reduces to one of these set operations on the sample space.

</Recall>

## Sample space and events

The **sample space** Ω is the set of all possible outcomes of an experiment. Every question begins by choosing Ω. Examples:

- Single coin flip: $\Omega = \\{H, T\\}$ — two outcomes.
- Two coin flips: $\Omega = \\{HH, HT, TH, TT\\}$ — four outcomes.
- One roll of a six-sided die: $\Omega = \\{1, 2, 3, 4, 5, 6\\}$.
- Draw one card: $\Omega$ = the 52 cards. |Ω| = 52.
- Sampling a next token from an LLM: $\Omega$ = the vocabulary. |Ω| = 32,000 (Llama) or ~200,000 (GPT-4 class).
- Position of a physical particle in a room: $\Omega$ = a 3-D continuous set — Ω is uncountably infinite.

An **event** is any subset of Ω. If the outcome of the experiment lands inside the subset, we say "the event occurred." Some events have names (rolling an even number, drawing an Ace) but formally an event is just a set of outcomes.

- $A = \\{2, 4, 6\\}$ is the event "the die shows even."
- $B$ = "at least one head in two flips" $= \\{HH, HT, TH\\}$.
- $C$ = "the LLM's next token is a punctuation mark" is the subset of the vocabulary that is punctuation.

For finite sample spaces where every outcome is equally likely, the probability of an event is just a counting ratio: P(A) = |A| / |Ω|. This is the "classical" definition of probability that Laplace worked with; it's still how most textbook problems are solved.

## The three rules probability has to obey

Andrey Kolmogorov (1933) formalized these three "obvious" facts and called them **axioms**. That's just a fancy word for "rules we accept without further proof, from which everything else is derived." Any function P that assigns numbers to events *and* obeys all three is a valid probability measure. Here they are, in English then in symbols:

**(1) Probabilities are never negative.**
$$P(A) \geq 0 \text{ for every event } A.$$

**(2) The probability that "something happens" is 1.**
$$P(\Omega) = 1.$$

**(3) For non-overlapping events, "or" means "add."**
$$P(A_1 \cup A_2 \cup \cdots) = \sum_i P(A_i) \text{ if the } A_i \text{ don't share any outcomes.}$$

That third one is called **countable additivity**. "Countable" just means "you can list them one by one, even if the list is infinite." The **pairwise disjoint** condition means "no two of them overlap." The union symbol ∪ means "or" — outcomes in this event *or* that one.

Everything else you know about probability follows from these three. Two consequences you'll use constantly:

**Complement rule.** For any event A: $P(A^c) = 1 - P(A)$, where $A^c$ ("A-complement") means "not A." The proof is one line: A and "not A" are disjoint and together cover Ω, so by axiom 3, P(A) + P(A^c) = P(Ω) = 1.

**Inclusion–exclusion for two events.** $P(A \cup B) = P(A) + P(B) - P(A \cap B)$. In English: "the probability of A-or-B is the probability of A plus the probability of B minus the overlap (so we don't count it twice)." The overlap A ∩ B means "outcomes in both A *and* B."

Those two identities plus the axioms will solve about 90% of the probability problems you'll meet.

## Computing probabilities: enumeration vs simulation

There are two ways to compute the probability of an event:

**Enumeration.** List Ω, list the event A, compute |A| / |Ω| (equal-outcome case) or sum P(ω) over ω in A. Feasible when Ω is small — coin flips, single die rolls, small card problems.

**Simulation (Monte Carlo).** Sample many outcomes from Ω according to the measure P, count how often the event occurs, divide. The **law of large numbers** (a later chapter) guarantees this estimate converges to the true probability as the number of samples grows. Feasible when Ω is astronomically large but you can draw samples cheaply.

For a GenAI engineer, simulation is the dominant tool: token sequences, image latents, dialog trajectories — none of these can be enumerated, but they can be sampled. Monte Carlo estimates back much of the evaluation and diagnostic tooling.

The relative error of a Monte Carlo estimate shrinks as 1/√N — quadruple the samples for half the error. This is a fundamental scaling law you meet everywhere: A/B test sample sizes, SGD mini-batch noise, Bayesian posterior sampling, RL policy evaluation.

## Two ways to read probability: frequentist vs Bayesian

The axioms don't care how you *interpret* probability. But there are two competing philosophies out there, and they'll come up whether you want them to or not:

**Frequentist.** "P(A) is the long-run fraction of times A would happen, if you could repeat the experiment infinitely many times." This is the intuitive interpretation for repeatable experiments — coin flips, dice rolls, benchmark evaluations. When a paper says "the model achieves 87% accuracy," that's a frequentist statement: it's a long-run rate over held-out samples.

**Bayesian.** "P(A) is a *degree of belief* in A, given the information you have right now." This lets you put probabilities on one-off things ("what's the probability my hypothesis is true?"), because probability is treated as a subjective belief that updates as new evidence arrives. Updates use Bayes' theorem (next chapter). Dominant in Bayesian ML, decision theory, and modern LLM calibration research.

Both interpretations obey the same three axioms. The fight is only about *which events you're willing to assign probabilities to*. Practical ML happily mixes both — cross-validation feels frequentist, Bayesian optimization is Bayesian by design. Same math, different mindset.

## Things that trip people up

**"Write down Ω first, always."** Half of probability mistakes come from being fuzzy about the sample space. A classic puzzle: "A family has two children, at least one is a boy. What's the probability both are boys?" The answer is 1/3 or 1/2 depending on whether Ω is &#123;BB, BG, GB, GG&#125; or "families that have at least one boy." Neither is wrong — but you have to be explicit about which universe you're in.

**"Equally-likely is an assumption, not the default."** Assuming P(ω) = 1/|Ω| for every outcome is convenient but not automatic. Real experiments rarely have exactly uniform outcomes; you need reasoning or data to justify uniformity.

**"Zero probability doesn't always mean 'impossible.'"** In continuous sample spaces (Ω = the real line, for example), every single point has probability exactly zero — but the point still exists, and can still be observed. That's why continuous probability uses **densities** (probability *per unit interval*) instead of raw point probabilities. Details in the random-variables chapter.

**"Simulation estimates come with wobble."** A Monte Carlo estimate is itself random — run the simulation twice, get slightly different numbers. Always report an error bar. The standard error of a probability estimate over N samples is √(p(1−p)/N), which shrinks as 1/√N. Quadrupling samples cuts the error in half.

<RealWorldML title="LLM sampling: every decoding trick is a choice of probability measure">

Here's the payoff. When an LLM writes text, it produces one token at a time. Each step is a mini probability problem with these ingredients:

- **Sample space Ω** = the **vocabulary** (all the tokens the model can output — around 32,000 for Llama, roughly 200,000 for a GPT-4-class model). A **token** is a word or word-piece, roughly.
- **Probability measure P** at that step = softmax(logits / T). Here **logits** are the model's raw output scores for each token (unnormalized), **T** is the **temperature** (a knob), and **softmax** turns the scores into probabilities that sum to 1. This measure obeys all three of Kolmogorov's rules: non-negative, sums to 1, "or means add" for non-overlapping token subsets.
- **Events** = subsets of the vocabulary you might care about: "the next token is a noun," "the next token starts a new sentence," "the next token is the special `end-of-text` token."
- **Sampling strategies** = choices of *which measure* to actually sample from. All the fancy names you've heard are just different measures on the same Ω:
  - **Greedy** = pick the top-scoring token, always. This is a point-mass measure: probability 1 on the top token, 0 on everything else.
  - **Top-k** = restrict to the top k tokens (say k = 40), zero out the rest, renormalize so it sums to 1. Now you're sampling from a smaller sample space.
  - **Top-p** (also called **nucleus sampling**) = restrict to the smallest set of top tokens whose combined probability is at least p (say p = 0.9), then renormalize. This adapts the size of the restricted set to the situation.
  - **Temperature** = the T in `softmax(logits / T)`. High T (like 1.5) flattens the distribution — the model becomes more random. Low T (like 0.3) sharpens it — the model becomes more decisive. T = 0 is greedy.

Big picture: every decoding trick is a choice about *which measure on Ω to sample from*. Once you see it this way, common LLM problems get much easier to reason about:

- **Mode collapse** — the model keeps saying the same thing → your measure is too sharp; try higher T or nucleus sampling.
- **Repetitive text** — the model loops → your measure isn't accounting for recent history; add a repetition penalty.
- **Hallucinations** — the model confidently says false things → your measure is placing weight on tokens that lead the model down bad paths; try lower T or top-p to concentrate on more confident options.

Decoding isn't magic. It's measure design.

</RealWorldML>

- **Event** = any subset of Ω. "The die shows even" is the event &#123;2, 4, 6&#125;.

- **Frequentist vs Bayesian** interpretations both obey the same axioms; the difference is which events you're willing to assign probabilities to.
- Every LLM decoding strategy is a *choice of probability measure* on the vocabulary sample space.
