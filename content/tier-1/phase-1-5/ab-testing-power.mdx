export const powerCurveCode = `# Power = P(reject H0 | H1 true).  It depends on effect size, sample size, and alpha.
# Visualize power as a function of n for a fixed effect size.
import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm
rng = np.random.default_rng(0)

def power_for_two_sample_z(n_per_arm, delta, sigma, alpha=0.05):
    """
    Power of a two-sided two-sample z-test for a difference of means.
    delta = true difference between arms; sigma = pooled std dev per arm.
    """
    se = sigma * np.sqrt(2 / n_per_arm)
    z_alpha = norm.ppf(1 - alpha / 2)
    # Under H1, the test statistic Z ~ N(delta / se, 1); we reject if |Z| > z_alpha.
    z_shift = delta / se
    power = norm.cdf(z_shift - z_alpha) + norm.cdf(-z_shift - z_alpha)
    return power

sigma = 1.0
delta = 0.2   # detect a 0.2 unit lift

# Plot power vs sample size per arm.
ns = np.arange(20, 1500, 20)
powers = [power_for_two_sample_z(n, delta, sigma) for n in ns]

fig, ax = plt.subplots(figsize=(6.4, 3.6), facecolor="none")
ax.set_facecolor("#0f0f14")
ax.plot(ns, powers, color="#c4f04b", lw=2)
ax.axhline(0.8, color="#f5cf3d", ls="--", label="80% power target")
ax.axhline(0.05, color="#8a8a82", ls=":", label="alpha (type I error)")
ax.set_xlabel("n per arm", color="#dcdcd6")
ax.set_ylabel("power", color="#dcdcd6")
ax.set_title(f"Power curve for detecting delta = {delta} at alpha = 0.05", color="#dcdcd6")
ax.legend(facecolor="#181820", edgecolor="#2a2a35", labelcolor="#dcdcd6")
ax.tick_params(colors="#dcdcd6")
for spine in ax.spines.values():
    spine.set_color("#2a2a35")
plt.tight_layout()
plt.show()

# Find the smallest n giving 80% power.
target = 0.80
n_needed = next(n for n, p in zip(ns, powers) if p >= target)
print(f"Smallest n per arm giving >= 80% power: {n_needed}")`;

export const sampleSizeCode = `# Sample-size formula for two-sample z-test:
#   n per arm ~ (z_alpha/2 + z_beta)^2 * (sigma_C^2 + sigma_T^2) / delta^2
# For pooled equal variances sigma^2: n ~ 2 * sigma^2 * (z_alpha/2 + z_beta)^2 / delta^2.
import numpy as np
from scipy.stats import norm

def n_per_arm(delta, sigma, alpha=0.05, power=0.80, two_sided=True):
    z_a = norm.ppf(1 - alpha / 2) if two_sided else norm.ppf(1 - alpha)
    z_b = norm.ppf(power)
    n = ((z_a + z_b) ** 2) * 2 * sigma ** 2 / delta ** 2
    return int(np.ceil(n))

# Various effect sizes for the same variance.
sigma = 1.0
print(f"{'Delta':>10}  {'n per arm (80% power, 5% alpha)':>35}")
for delta in [0.10, 0.20, 0.30, 0.50, 1.00]:
    n = n_per_arm(delta, sigma)
    print(f"{delta:>10.2f}  {n:>35}")

# For proportion tests (Bernoulli), sigma^2 = p*(1-p).  Use p = baseline for planning.
print("\\nProportion test (baseline p = 0.10, target detect absolute lift delta_p):")
p0 = 0.10
for delta_p in [0.005, 0.01, 0.02, 0.05]:
    var_per_arm = p0 * (1 - p0)   # rough conservative estimate; more accurate uses both p0 and p1
    n = int(np.ceil((norm.ppf(0.975) + norm.ppf(0.80)) ** 2 * 2 * var_per_arm / delta_p ** 2))
    print(f"  delta_p = {delta_p}: n per arm = {n}")

print("\\nRule of thumb: halving delta requires 4x more samples (n scales as 1/delta^2).")`;

export const peekingCode = `# Peeking: repeatedly looking at the running A/B test and stopping when p < alpha.
# This dramatically inflates the Type I error rate.  Simulate to see.
import numpy as np
from scipy.stats import norm
rng = np.random.default_rng(0)

alpha = 0.05
z_alpha = norm.ppf(1 - alpha / 2)

# Scenario: control and treatment are actually identical (H0 true).  Draw one sample at a time
# and, after each new sample, run a z-test.  Stop as soon as p < 0.05.
def one_peeking_experiment(max_n=5000):
    """Return True if the experiment ever crossed the significance threshold."""
    z_score = 0.0
    n_c, n_t = 0, 0
    sum_c, sum_t = 0.0, 0.0
    sq_c, sq_t = 0.0, 0.0
    for i in range(1, max_n + 1):
        # Draw one from each arm.
        x_c = rng.normal(0, 1)
        x_t = rng.normal(0, 1)   # SAME distribution -- H0 true
        n_c += 1; n_t += 1
        sum_c += x_c; sum_t += x_t
        sq_c += x_c ** 2; sq_t += x_t ** 2
        if i < 30:   # skip early where estimates are noisy
            continue
        # Compute two-sample z-statistic.
        m_c = sum_c / n_c; m_t = sum_t / n_t
        v_c = (sq_c - n_c * m_c ** 2) / (n_c - 1)
        v_t = (sq_t - n_t * m_t ** 2) / (n_t - 1)
        se = np.sqrt(v_c / n_c + v_t / n_t)
        z = (m_t - m_c) / se
        if abs(z) > z_alpha:
            return True
    return False

# How often does peeking find "significance" even under H0?
n_experiments = 500
false_pos = sum(one_peeking_experiment() for _ in range(n_experiments))
print(f"Under H0, peeking-based tests found significance in {false_pos} / {n_experiments} experiments")
print(f"Effective false-positive rate: {false_pos / n_experiments * 100:.1f}%")
print(f"Nominal alpha: {alpha * 100:.0f}%")
print("\\nPeeking inflates the FP rate from 5% to 30-40% or worse.")
print("Fixes: fixed-horizon test, group-sequential methods, or Bayesian monitoring.")`;

export const bayesianABCode = `# Bayesian A/B test: place a Beta prior on each arm's conversion rate, update to
# posterior Beta after data, and compute P(treatment > control) directly.
# No p-value peeking issue because probabilities monotonically update.
import numpy as np
from scipy.stats import beta as beta_dist
import matplotlib.pyplot as plt
rng = np.random.default_rng(0)

# True conversion rates.
true_p_c = 0.10
true_p_t = 0.12    # 2 percentage-point absolute lift (20% relative)

# Prior: Beta(1, 1) = Uniform.  Weakly informative.
alpha_c, beta_c = 1, 1
alpha_t, beta_t = 1, 1

# Simulate the experiment; check P(treatment > control) at each checkpoint.
checkpoints = [50, 200, 500, 2000, 5000]
n_max = max(checkpoints)
control_hits, treatment_hits = 0, 0

for n in range(1, n_max + 1):
    if rng.binomial(1, true_p_c): control_hits += 1
    if rng.binomial(1, true_p_t): treatment_hits += 1
    if n in checkpoints:
        # Posteriors.
        alpha_c_post = alpha_c + control_hits;   beta_c_post = beta_c + (n - control_hits)
        alpha_t_post = alpha_t + treatment_hits; beta_t_post = beta_t + (n - treatment_hits)
        # P(p_t > p_c) via Monte Carlo from the two posteriors.
        n_mc = 10_000
        samples_c = beta_dist.rvs(alpha_c_post, beta_c_post, size=n_mc)
        samples_t = beta_dist.rvs(alpha_t_post, beta_t_post, size=n_mc)
        p_t_gt_c = (samples_t > samples_c).mean()
        # 95% credible intervals on each arm.
        lo_c, hi_c = beta_dist.ppf([0.025, 0.975], alpha_c_post, beta_c_post)
        lo_t, hi_t = beta_dist.ppf([0.025, 0.975], alpha_t_post, beta_t_post)
        print(f"n per arm = {n:>5}: control = [{lo_c:.4f}, {hi_c:.4f}], treatment = [{lo_t:.4f}, {hi_t:.4f}], P(t > c) = {p_t_gt_c:.4f}")

print("\\nP(treatment > control) directly answers the business question.")
print("You can stop whenever it crosses your decision threshold (e.g., 95%) -- no correction needed.")`;

<Motivation title="A/B testing — where stats meets shipping (and where you can ship noise if you're not careful)">

Every time you compare two versions of something and pick a winner based on data, you're doing an **A/B test**. Version A is the current thing (the "control"); version B is the new thing you want to try (the "treatment"). You send some users to each, measure a metric, and check if B is really better than A or if the difference is just noise.

This shows up everywhere in GenAI work:

- Old prompt vs new prompt on user satisfaction.
- Baseline LLM vs fine-tuned LLM on internal accuracy.
- Current retrieval strategy vs new retrieval strategy on answer quality.

The math underneath is just two-sample hypothesis testing (from the previous chapter). But the *practice* has several traps that regularly ship noise as real improvements:

- **Sample-size planning.** How many users do you actually need to tell A and B apart? If you don't compute this before starting, you'll either waste traffic or draw conclusions from too little data.
- **Power.** Is your test big enough that a real difference would actually show up? Low-power tests miss real effects, so "not significant" tells you nothing.
- **Peeking.** Looking at the running results every day and stopping the moment "p < 0.05" is much worse than it sounds — it dramatically inflates your false-positive rate.
- **Bayesian alternatives.** For teams that want to peek daily (which is basically everyone), Bayesian A/B tests are cleaner because they don't have the peeking problem.

Three real GenAI-engineer scenarios you'll hit:

- **Should we ship this prompt change?** Prompt A is the current one, prompt B is a candidate. You want enough traffic to say "B is better with high confidence" before flipping the switch.
- **Which of these two models is actually better?** Model A gets 82.1% on 500 eval examples, model B gets 82.4%. Is that 0.3% real or just noise? Sample-size analysis tells you if 500 examples is even enough.
- **How much RLHF data do we need?** If you're training a reward model on human preferences, you need enough pairs to tell a genuinely-better model from noise.

This chapter builds the practical A/B testing toolkit: power curves, sample-size formulas, peeking-safe designs, and Bayesian alternatives that dodge the peeking problem entirely.

</Motivation>

<Intuition title="Power = 'if the effect is real, do I actually catch it?'">

Here's the one word that trips people up more than any other in A/B testing: **power**.

Power is really simple. It's the probability that, *if there truly is an effect*, your test will actually notice it. Written as a formula: power = P(reject H₀ | H₁ is true) — but you don't need the formula, you need the concept. Just remember: **power = chance of catching a real effect**.

Three knobs move power:

- **More users → higher power.** Bigger sample size shrinks noise, so the real signal shows up more clearly.
- **Bigger true effect → higher power.** A 10% lift is obvious; a 0.1% lift is drowned in noise.
- **More permissive threshold (larger α) → higher power.** But this comes at a cost — more false positives.

The industry-standard target is **80% power**. That means: "if my treatment really does what I think it does, I'll catch it 80% of the time." If your test only has 30% power and you get a "not significant" result, that tells you almost nothing — you probably just missed the effect because your sample was too small.

**Sample-size planning** is the practical version of this. You start with your goals — "I want to detect a 2% lift, at 5% significance, with 80% power" — and work backward to compute how many users you need. Then you commit to that sample size *before* running the experiment. Adjusting after the fact is a form of cheating (it inflates false positives).

**The peeking problem.** This one is worth pausing on because everyone does it. Say you plan a 5,000-user A/B test. Every morning you check the running p-value. If it drops below 0.05 one day, you stop and declare victory. Seems reasonable, right?

It's not. Each peek is a fresh chance for the running statistic to wander into "significant" territory purely by luck. Peek enough times and you'll definitely find a false positive. Continuous peeking can turn your intended 5% false-positive rate into a real-world 30-40% rate. Six to eight times inflated. This is one of the biggest sources of "our A/B test showed a lift but the improvement never materialized in production" — you shipped noise.

There are proper fixes for this (which we'll cover), but the shortest one is: switch to a Bayesian A/B test, which doesn't have this problem at all.

</Intuition>

<Recall term="Type I / Type II errors">

**Type I error** = crying wolf. Rejecting the null hypothesis (H₀) when it's actually true — declaring a real effect when there isn't one. The rate is called **α (alpha)** and you get to pick it (usually 0.05). **Type II error** = missing the wolf. Failing to reject H₀ when it's actually false — missing a real effect. The rate is called **β (beta)** and it depends on how big the true effect is and how much data you have. **Power** = 1 − β = your chance of catching a real effect.

</Recall>

<Recall term="Effect size Δ (delta)">

**Effect size** is the "how much better" number you care about. If you're testing conversion rates, effect size Δ might be "2 percentage points." If you're testing user satisfaction 1-5 stars, effect size might be "0.1 stars." Bigger effect = easier to detect. You have to pick a target effect size *before* the test — the smallest lift worth detecting — because that determines your required sample size.

</Recall>

## Power, in one picture

Let's connect the intuition to numbers. Say you're comparing control (rate μ_C) and treatment (rate μ_T), and the true difference between them is Δ = μ_T − μ_C. Both arms have the same variance σ² and n users per arm. The test statistic (a number that measures "how surprising is this data") is:

$$
Z = \frac{\bar X_T - \bar X_C}{\text{SE}}, \quad \text{SE} = \sigma \cdot \sqrt{\frac{2}{n}}.
$$

**Standard error (SE)** is the wobble of the estimate — that √2/n factor is from combining wobble across both arms. Under the "there really is an effect" scenario, Z is bell-curve-shaped centered at Δ/SE. You reject H₀ if |Z| > 1.96 (the 95% threshold).

Power is the probability that Z ends up past that threshold:

$$
\text{Power} \;=\; \Phi(z_{\text{shift}} - z_{\alpha/2}) + \Phi(-z_{\text{shift}} - z_{\alpha/2})
$$

where z_shift = Δ · √(n / (2σ²)) is essentially "signal-to-noise, scaled up by sample size." The Φ (capital phi) is just "look up on a standard bell curve." As n grows, z_shift grows, and power heads toward 1. As Δ grows, same story. As α shrinks, the threshold gets stricter and power drops.

Bottom line: **power curves rise as either sample size or true effect gets bigger.** The industry target is 80% at some pre-committed effect size.

## Sample-size formulas — plan before you run

Here's the formula you'll actually memorize. For a two-sample test with equal variance σ² per arm:

$$
n_{\text{per arm}} \;\approx\; \frac{(z_{\alpha/2} + z_\beta)^2 \cdot 2\sigma^2}{\Delta^2}
$$

Plug in standard numbers (α = 0.05 gives z_(α/2) ≈ 1.96, target 80% power gives z_β ≈ 0.84):

$$
n_{\text{per arm}} \;\approx\; \frac{(1.96 + 0.84)^2 \cdot 2\sigma^2}{\Delta^2} \;\approx\; \frac{15.7 \sigma^2}{\Delta^2}.
$$

Rule of thumb: **for α = 0.05, 80% power, use ~16σ²/Δ² per arm.** Simple and close enough.

**Big consequence.** Halving the target effect Δ **quadruples** the required sample size, because n depends on 1/Δ². Detecting a 1% lift at 10% baseline conversion needs about 15,700 users per arm. Detecting a 0.5% lift at the same baseline? 62,800 users per arm. Small effects are expensive.

For conversion-rate tests (each user either converts or doesn't — a Bernoulli), σ² = p(1 − p), so:

$$
n_{\text{per arm}} \;\approx\; \frac{16 \cdot p(1-p)}{\Delta^2}
$$

Everyone in industry uses this or a slight variant. Once you internalize "detecting smaller effects quadratically costs more traffic," you'll stop being surprised by A/B tests that take weeks.

## The peeking trap — and how to avoid it

Say you plan an A/B test for n = 5,000 users per arm, expecting to run for a week. But you check the dashboard every morning. On day 3, the running p-value drops below 0.05. Ship it, right?

Nope. This is one of the most common mistakes in the industry, and it's not subtle — it's a math error.

Here's why it fails. Every time you check the p-value, you're taking another chance to reject the null. Even when there's genuinely no effect, the running test statistic (which is basically summing random noise) will occasionally wander into "significant" territory just by luck. Give yourself enough peeks and it's almost certain to happen at least once. Stop the first time it does and you'll declare "victory" over what's actually pure noise.

The numbers are ugly. Continuous peeking under the null hypothesis inflates your effective false-positive rate from the intended 5% to somewhere around 30–40%. That's a 6-8× inflation. Ship 10 features this way and you'll ship 3-4 fake improvements.

<RunnableCode title="Peeking under H0: nominal 5% alpha becomes ~30% empirical FP rate" code={peekingCode} packages={["numpy", "scipy"]} />

**Ways to fix peeking:**

- **Fixed-horizon test.** Commit to a sample size up front. Look at the results *once*, at the end. Boring but correct.
- **Group-sequential methods.** Pre-agree on a few checkpoint times (say, at 25%, 50%, 75%, 100% of the planned traffic) and use tighter thresholds at each checkpoint. Two named designs you'll hear: **O'Brien–Fleming** (very strict early, gets easier as you approach the end) and **Pocock** (same threshold across all checks). Both keep the overall false-positive rate at your intended α.
- **Always-valid p-values / e-values.** Newer methods (from Howard, Ramdas, and their collaborators) that give you valid statistical inference even when you peek continuously. Netflix and LinkedIn use these in production.
- **Bayesian monitoring.** Just switch to a Bayesian A/B test. No peeking problem, no corrections needed. Covered next.

## Bayesian A/B — posteriors that don't care about peeking

Instead of computing p-values, we frame the test Bayesianly. Recipe for a conversion-rate A/B (from the MAP chapter):

- Put a **Beta(α, β) prior** on each arm's conversion rate. Beta(1, 1) is uniform ("no opinion"). Beta(2, 10) says "I think it's probably low."
- After you observe the data, update to the **posterior**: control's posterior is Beta(α + successes_C, β + failures_C). Same for treatment.
- Compute **P(treatment better than control | data)** by drawing many samples from each posterior and counting how often treatment > control. This is a simple Monte Carlo calculation.
- **Decision rule:** ship treatment if P(treatment better) > 95%. (Or 90%, or whatever your business decides — the threshold is up to you.)

**Why doesn't peeking break this?** Because the Bayesian posterior updates *monotonically* as data comes in — each new observation shifts it one honest step. Every time you look at "P(treatment better)," that number is a valid probability given the data you have *right now*. There's no "multiple looks inflating the error rate" — the number is what it is at every moment.

So you can peek every hour, every day, forever. When P(treatment better) crosses your threshold, you can stop. When it stays below, keep collecting. No corrections needed.

**Trade-off:** the Bayesian answer depends on the prior. A strong skeptical prior (say Beta(20, 100)) will take more data to move; a flat prior (Beta(1, 1)) shifts fast but is more sensitive to noise. In practice, teams pick lightly-informative priors calibrated to their historical baseline conversion rates.

## Beyond A/B — designs you'll meet in production

The plain A/B is the simplest randomized experiment. Real production experimentation often needs more:

- **A/B/C tests** (multiple treatments at once). Each additional arm is another chance to falsely reject H₀, so you have to apply a **multiple-comparison correction** — Bonferroni (simple: divide α by the number of arms) or Benjamini-Hochberg FDR (smarter: controls the *fraction* of false discoveries). Both are covered in the hypothesis-testing chapter.
- **Factorial designs.** Test multiple things at once — prompt A vs B **and** retrieval X vs Y — in a compact 2×2 grid. Lets you spot interactions ("prompt B is better *only* with retrieval Y") efficiently.
- **Multi-armed bandits.** Instead of splitting traffic evenly, adaptively route more traffic to arms that look promising while still exploring the others. Two well-known algorithms: **Thompson sampling** (Bayesian) and **UCB** (Upper Confidence Bound). Used when you're optimizing continuously rather than "picking one winner and shipping."
- **Interleaving.** In search ranking, instead of user A getting one ranker and user B another, mix the two rankers' results for the *same* user. Higher statistical power because each user contributes to both arms.
- **Segmentation.** Some users respond, others don't. Averaged across everyone, the effect might look zero — but the treatment might be great for one segment. Break results down by user type, geography, device, etc., to see heterogeneous effects.

Every serious ML deployment platform (Netflix XP, Meta's experimentation stack, Split, LaunchDarkly) has infrastructure for one or more of these. The stats underneath is the same as plain A/B; the twist is in how you design and analyze.

## Things that trip people up

**"Design first, analyze second."** Compute your required sample size *before* running the test. Don't fudge it later.

**"'80% power' is meaningless without an effect size."** Power depends on the target effect. "80% power at 2% lift" is meaningful; "80% power" alone is not. Always pick a **minimum detectable effect** upfront.

**"Peeking without correction is a bug, not a shortcut."** If you're going to monitor daily and stop early, use group-sequential methods, always-valid p-values, or just switch to Bayesian A/B.

**"Novelty effects."** Users behave differently on day 1 (curious, exploring) than on day 5 (steady state). If you cut the experiment short, you might be measuring novelty rather than long-term impact. Run long enough to see steady behavior.

**"External validity."** Your A/B might have run on 1% of users during a specific week. That doesn't guarantee the result generalizes to 100% of users during the busy holiday season. Sanity-check with segmentation and phased rollouts.

**"Simpson's paradox."** The aggregate result can flip direction when you break it down by segment. Example: treatment looks better overall, but it's *worse* for every individual customer type (because the mix of customer types shifted between arms). Always break results down.

**"Ratio metrics are sneaky."** Metrics like "clicks per user" are ratios of two random quantities, not raw counts. Their standard errors need the **delta method** (a Taylor-series trick for the SE of a ratio) or a bootstrap. Running a naive t-test on ratios can give wildly wrong CIs.

**"Sample Ratio Mismatch (SRM)."** If you set up a 50/50 split but the observed split is 51/49 with high confidence (chi-squared test says so), something is broken — probably a bug in the bucketing code, or a filter that isn't random. **Check SRM first**, before analyzing anything downstream. If SRM is off, throw the results away and fix the plumbing.

<RealWorldML title="Five ways A/B testing shows up in real GenAI work">

**1. Shipping an LLM prompt change.** You tried a new system prompt on your internal eval; helpfulness looks about 3% better. To actually ship, you run a production A/B: 50,000 users each arm, primary metric is a 1–5 star rating. Quick sample-size check: with rating standard deviation σ ≈ 1 and target lift Δ = 0.05 stars, you need about 16·1²/0.05² ≈ 6,400 users per arm — trivial at production scale. After a week, a t-test gives p = 0.003 with lift 0.07 ± 0.02. Ship it.

**2. Comparing two LLMs on a static benchmark.** Model A got 82.1% on 500 examples, model B got 82.4% on the same 500 examples. Real improvement? Almost certainly not — the standard error on 500 Bernoulli examples at 82% accuracy is about √(0.82·0.18/500) ≈ 1.7%, so any lift under ~2% is well within noise. Get a bigger eval set, or admit "within noise" honestly.

**3. RLHF preference collection.** To train a reward model on human preferences, you collect pairwise comparisons like "which response is better?" If you want to distinguish a candidate model with a 5% win-rate lift from a near-tied baseline (50%), you need approximately 16·0.5·0.5/0.05² ≈ 1,600 preference pairs per candidate to hit 80% power.

**4. Continuous monitoring** (not an A/B test, but relevant). Suppose you're monitoring the rate of prompt-injection attacks on your API. This isn't a designed A/B — you're just watching a stream and want to notice when something changes. Use **SPC** (**Statistical Process Control**, a set of control-chart methods from manufacturing) or **online change-point detection** instead of a fixed-horizon test.

**5. Feature flags and gradual rollouts.** Big-tech deployment goes like this: flip a feature flag to 1% of users, then 10%, then 100%, checking metrics at each step. Every one of those steps is implicitly an A/B against the prior state. Modern experimentation platforms (Netflix XP, Meta's stack, Split, LaunchDarkly) automate the analysis. Understanding what's happening under the hood keeps you from getting fooled by the auto-generated dashboard.

**A note on Bayesian A/B in industry.** Netflix, Booking.com, Shopify, and many others use variants of Bayesian A/B testing in production. Reasons:

- It handles peeking without corrections.
- The output ("95% probability treatment is better") is *far* easier to explain to stakeholders than "p = 0.03 on a two-sample z-test with Welch correction."
- It naturally accommodates hierarchical priors (e.g., "each region has its own baseline conversion rate, but they share some structure").

If you're going to do A/B testing at scale in a real product, learn both frameworks. Pick whichever fits your team's cadence and stakeholder communication style.

<CodeItYourself title="Three drills — sample size, peeking simulation, Bayesian A/B on a small dataset">

1. **Sample size for a Bernoulli A/B.** You want to detect a 1% absolute lift in a 5% baseline conversion rate at α = 0.05, 80% power. Compute n per arm using the formula n ≈ 16 · p(1 − p) / Δ² (with p = baseline). Then simulate 500 experiments at that n where the true lift is exactly 1%. Empirical power should be about 80%.

2. **Peeking under H₀.** Simulate two-sample z-tests where control and treatment have IDENTICAL distributions (H₀ true). At each step from n = 30 to n = 2000, compute the running p-value. Stop when p &lt; 0.05 (or reach n = 2000 without rejecting). Repeat 500 times. Fraction of experiments that ever crossed p &lt; 0.05 should be much larger than 5% — showing peeking&rsquo;s Type I inflation.

3. **Bayesian A/B on a small experiment.** Control: 20 conversions in 200 users. Treatment: 30 conversions in 200 users. Compute posterior Beta distributions (Beta(1,1) priors). Sample 10,000 points from each posterior; compute the fraction where treatment sample &gt; control sample. This is your posterior P(treatment better). Compare to a frequentist two-proportion z-test on the same data.

</CodeItYourself>

<Summary>

- **A/B testing** decides whether a change is worth shipping. Power / sample size / peeking are the operational concerns.
- **Power** = 1 − β = P(reject H₀ | H₁ true). Standard target 0.80 at a specified effect size.
- **Sample size formula**: n per arm ≈ 16 · σ² / Δ² for α = 0.05, 80% power. Halving Δ requires 4× more data.
- **Peeking** with stop-on-significance inflates FP rate from 5% to 30–40%. Fixes: fixed-horizon, group-sequential, always-valid p-values, or Bayesian monitoring.
- **Bayesian A/B**: prior + posterior gives P(treatment better | data), updates monotonically, safe under peeking.
- **Advanced designs**: A/B/C, factorial, bandits, interleaving, segmentation — all rest on the same statistical foundations.
- **Practical traps**: SRM (sample ratio mismatch), novelty effects, Simpson&rsquo;s paradox, ratio metrics needing delta method or bootstrap.

</Summary>
