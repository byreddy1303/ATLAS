export const pValueCode = `# p-value: probability of observing data at least as extreme as what we saw, IF the null hypothesis is true.
# Small p means "if the null were true, our observation would be very surprising".
import numpy as np
from scipy.stats import norm
rng = np.random.default_rng(0)

# Setting: we suspect the population mean is > 0.  Sample n = 50 observations.
# Null hypothesis H0: mu = 0.  Alternative H1: mu > 0.
n = 50
sample = rng.normal(loc=0.3, scale=1.0, size=n)  # true mu = 0.3
sample_mean = sample.mean()
sample_se   = sample.std(ddof=1) / np.sqrt(n)

# Under H0, sample_mean ~ N(0, sigma^2/n).  Compute the z-statistic.
z = (sample_mean - 0.0) / sample_se

# One-sided p-value (H1: mu > 0).
p_one_sided = 1 - norm.cdf(z)
p_two_sided = 2 * (1 - norm.cdf(abs(z)))    # for H1: mu != 0

print(f"Sample: mean = {sample_mean:.4f}, SE = {sample_se:.4f}")
print(f"z-statistic          = {z:.4f}")
print(f"One-sided p-value (mu > 0): {p_one_sided:.4f}")
print(f"Two-sided p-value (mu != 0): {p_two_sided:.4f}")

# Repeat under H0 to visualize the p-value distribution.
n_repeats = 5000
p_values_under_null = []
for _ in range(n_repeats):
    s = rng.normal(loc=0.0, scale=1.0, size=n)   # H0 true
    m = s.mean()
    se = s.std(ddof=1) / np.sqrt(n)
    p_values_under_null.append(2 * (1 - norm.cdf(abs(m / se))))
p_values_under_null = np.array(p_values_under_null)

# KEY FACT: under H0, p-values are Uniform(0, 1).  Verify.
print(f"\\nUnder H0, fraction with p < 0.05 = {(p_values_under_null < 0.05).mean():.4f}   (expected 0.05)")
print(f"Under H0, fraction with p < 0.01 = {(p_values_under_null < 0.01).mean():.4f}   (expected 0.01)")
print("The Type I error rate = alpha; you WILL reject H0 falsely alpha fraction of the time.")`;

export const tTestCode = `# Two-sample t-test: compare the means of two independent groups.
# Common use: A/B testing.  Is the average lift statistically different from zero?
import numpy as np
from scipy.stats import ttest_ind
rng = np.random.default_rng(0)

# Control group: 200 samples with mean 3.0, std 1.0.
# Treatment group: 200 samples with mean 3.15 (5% lift), std 1.0.
n_c, n_t = 200, 200
control   = rng.normal(3.0, 1.0, size=n_c)
treatment = rng.normal(3.15, 1.0, size=n_t)

diff = treatment.mean() - control.mean()
print(f"Control mean:   {control.mean():.4f}")
print(f"Treatment mean: {treatment.mean():.4f}")
print(f"Observed lift:  {diff:+.4f}")

# Standard error of the difference.
se_diff = np.sqrt(control.var(ddof=1)/n_c + treatment.var(ddof=1)/n_t)
t_stat = diff / se_diff
print(f"\\nSE of difference: {se_diff:.4f}")
print(f"t-statistic:      {t_stat:.4f}")

# scipy version -- Welch's t-test allows unequal variances.
result = ttest_ind(treatment, control, equal_var=False)
print(f"\\nWelch's t-test:")
print(f"  t-statistic: {result.statistic:.4f}")
print(f"  p-value:     {result.pvalue:.4f}")

if result.pvalue < 0.05:
    print("  Reject H0: the two means differ at the 5% level.")
else:
    print("  Fail to reject H0: the difference is not statistically significant at 5%.")

# Confidence interval on the difference of means.
from scipy.stats import t as t_dist
df = n_c + n_t - 2   # simplified pooled df
t_crit = t_dist.ppf(0.975, df)
ci_low  = diff - t_crit * se_diff
ci_high = diff + t_crit * se_diff
print(f"\\n95% CI for the lift: ({ci_low:+.4f}, {ci_high:+.4f})")`;

export const permutationTestCode = `# Permutation test: non-parametric.  Combine the two groups, shuffle labels many times,
# compute the statistic on each shuffled dataset.  The observed statistic's rank in this
# empirical distribution IS a p-value -- no distributional assumption.
import numpy as np
rng = np.random.default_rng(0)

# Skewed data where t-test's Gaussian assumption is questionable.
n_c, n_t = 60, 60
control   = rng.exponential(scale=2.0, size=n_c)
treatment = rng.exponential(scale=2.2, size=n_t)  # ~10% mean lift

# Statistic: difference in means (could equally use median, quantile, etc.)
observed_diff = treatment.mean() - control.mean()
print(f"Observed difference (treatment - control): {observed_diff:+.4f}")

# Permutation test: combine samples, shuffle labels, recompute the difference.
combined = np.concatenate([control, treatment])
n_c_total = len(control)

n_perm = 5000
perm_diffs = np.zeros(n_perm)
for p in range(n_perm):
    shuffled = rng.permutation(combined)
    new_control   = shuffled[:n_c_total]
    new_treatment = shuffled[n_c_total:]
    perm_diffs[p] = new_treatment.mean() - new_control.mean()

# Two-sided p-value: how often is the shuffled difference at least as extreme as observed.
p_value = (np.abs(perm_diffs) >= abs(observed_diff)).mean()
print(f"Permutation p-value: {p_value:.4f}")

# Compare to Welch's t-test on the same data.
from scipy.stats import ttest_ind
tt = ttest_ind(treatment, control, equal_var=False)
print(f"Welch's t-test p-value: {tt.pvalue:.4f}")

print("\\nBoth p-values usually agree closely, even for skewed data.")
print("Permutation is exact under the null hypothesis of exchangeability -- no CLT needed.")
print("Prefer permutation for small samples, non-Gaussian statistics, or when you're paranoid about assumptions.")`;

export const multipleComparisonsCode = `# Multiple comparisons: running k tests at level alpha will produce ~alpha*k false positives
# under the null.  Two standard corrections: Bonferroni (family-wise error) and Benjamini-Hochberg (FDR).
import numpy as np
from statsmodels.stats.multitest import multipletests
rng = np.random.default_rng(0)

# Simulate 20 hypotheses.  5 are truly different from 0 (effect size 0.5).  15 have effect 0.
n_hyp = 20
n_true = 5
true_effects = np.zeros(n_hyp)
true_effects[:n_true] = 0.5

# For each hypothesis, generate n = 100 samples and compute a one-sample z-test p-value.
n = 100
from scipy.stats import norm
p_values = np.zeros(n_hyp)
for i in range(n_hyp):
    s = rng.normal(true_effects[i], 1.0, size=n)
    z = s.mean() / (s.std(ddof=1) / np.sqrt(n))
    p_values[i] = 2 * (1 - norm.cdf(abs(z)))

print("Raw p-values (first 10):")
for i in range(10):
    truth = "* TRULY DIFFERENT" if true_effects[i] != 0 else ""
    print(f"  Hypothesis {i:>2}: p = {p_values[i]:.4f}  {truth}")

# Uncorrected: how many with p < 0.05?
raw_significant = (p_values < 0.05).sum()
print(f"\\nRaw significant (p < 0.05): {raw_significant} / {n_hyp}")

# Bonferroni: reject H_i iff p_i < alpha / m.  Very conservative.
_, p_bonf, _, _ = multipletests(p_values, alpha=0.05, method='bonferroni')
bonf_significant = (p_bonf < 0.05).sum()
print(f"Bonferroni significant:      {bonf_significant} / {n_hyp}   (alpha_effective = {0.05/n_hyp:.4f})")

# Benjamini-Hochberg: controls false discovery rate (fraction of rejections that are false).
_, p_bh, _, _ = multipletests(p_values, alpha=0.05, method='fdr_bh')
bh_significant = (p_bh < 0.05).sum()
print(f"BH-FDR significant:          {bh_significant} / {n_hyp}   (controls FDR at 5%)")

print("\\nBonferroni is safer if you must avoid ANY false positives.")
print("BH-FDR is more powerful if you can tolerate a small fraction of false positives among your rejections.")`;

<Motivation title="Was the pattern I saw real, or a fluke?">

You run an experiment and see a pattern: "the new prompt got 4.2 stars on average, the old one got 3.9." Is that a real improvement, or noise that would swing the other way if you re-ran the test tomorrow? **Hypothesis testing** is the formal machinery for answering exactly that question. It gives you a discipline for saying "unlikely to be a fluke" without hand-waving.

The idea in one sentence: **assume nothing interesting is happening, ask "would my observation look surprising under that boring assumption?", and only claim a real effect if the answer is yes.** The "boring assumption" is called the **null hypothesis** (written H₀ and read "H-null" or "H-nought"). Everything in this chapter is a way to make "surprising" precise.

Two places you will meet this daily as a GenAI engineer:

- **A/B testing** — the mechanism by which you decide whether a model change ships. Suppose you tweak the system prompt and observe a 1.2% lift in **CTR** (click-through rate — the fraction of impressions that got clicked). Is that lift real or noise? Without a proper test you cannot tell, and you may ship changes that do nothing (or worse, hurt). Sample-size planning, the "peeking" problem (repeatedly checking a running test and stopping when it looks good — which secretly cheats), and multiple-testing corrections all live here.
- **Model comparisons** — "model A got 85.1% on the benchmark, model B got 85.3%; therefore B is better" is not a valid claim on its own. That 0.2-point gap could easily be inside the run-to-run noise of the evaluation. Mature ML labs report **standard errors** (a measure of the noise in an estimate — how much it would jiggle if you re-ran the eval), **confidence intervals** (a range of plausible true values, defined later in this chapter), and adjustments for the fact that they compared many models.

A serious warning up front: hypothesis testing has spectacular ways to fail. **p-hacking** (see NewConcepts) — running many tests until one looks significant, or picking your test after peeking at the data — invalidates the whole framework. Misreading a non-significant result as "no effect exists" throws away real discoveries. Confusing statistical significance with practical importance leads to shipping fake wins. This chapter will teach the machinery, but equal weight to the failure modes, because in practice the failure modes are where most damage happens.

Roadmap of the chapter: (1) the H₀/H₁ framework, (2) the *precise* definition of a p-value (the single most misinterpreted quantity in statistics), (3) the three most-used tests you will actually reach for — the **t-test** (for comparing means), the **chi-squared test** (for count / categorical data), and the **permutation test** (for anything else) — and (4) the corrections needed when you run many tests at once.

</Motivation>

<Intuition title="Picture: assume 'nothing interesting is happening', simulate what data would look like, and check if your real data is weirdly out at the edge.">

Here is the whole test in one mental picture.

1. **Assume the boring story is true.** Pretend, for the sake of argument, that the two groups you are comparing are identical (H₀: "no effect"). Under that pretence, ask: **what values would my summary of the data take, purely from random noise?**
2. **Simulate — literally or by formula — the shape of that noise distribution.** Called the **null distribution**: the distribution the summary statistic would follow if H₀ were true. For a difference of averages this is (approximately) a bell curve centred at zero.
3. **Look up where your actual observation lands.** If your real number is *right in the middle of the noise distribution*, the boring story explains it fine — you keep believing H₀. If your real number is *way out at one edge* (a "tail"), the boring story is a bad fit — you **reject H₀**.

"Way out in the tail" is made numerical by the **p-value**: the probability, computed under the assumption that H₀ is true, of observing a statistic at least as extreme as the one you actually got. Small p (say 0.001) = "my data would only occur one time in a thousand if the boring story were true" → strong evidence against the boring story. Big p (say 0.6) = "my data is exactly the sort of thing the boring story would produce" → nothing to see.

The threshold you commit to in advance is called **α** (Greek "alpha"), the **significance level**, usually 0.05. If p &lt; α you "reject H₀"; otherwise you "fail to reject." α is your *tolerance* for making a **Type I error** — declaring an effect when in fact H₀ was true. Setting α = 0.05 means "I am willing to be wrong (falsely claim an effect) 5% of the time when there is really nothing there."

Two consequences of taking this seriously — both important, both counter-intuitive at first:

1. **A 95% confidence interval is exactly the set of null values you would fail to reject at α = 0.05.** (A **confidence interval**, defined properly in the previous chapter but recapped in a Recall below, is a range of values that plausibly contain the true parameter.) Hypothesis tests and confidence intervals are two views of the same object.
2. **"Not finding significance" is NOT the same as "there is no effect."** You may just not have looked hard enough — small sample size, weak measurement. This is why the good practice is to report the observed **effect size** (the size of the difference you saw, in real units — e.g. "0.18 stars") and the confidence interval, not just the p-value.

</Intuition>

<Recall term="Null hypothesis H₀ and alternative H₁">

**H₀** ("H-null") is the default, boring story — usually "no effect", "no difference between groups", or "the parameter equals some baseline value." **H₁** ("H-alt") is the interesting claim you would like to argue *for*. Hypothesis testing sets up H₀ as a strawman and asks the data to *knock it down*.

Crucial subtlety: a hypothesis test does **not** compute the probability that H₀ is true. It computes `P(data | H₀)` — read "probability of the data, given H₀" — which is the reverse of what most people intuitively expect. The confusion is understandable but consequential; a whole section below is dedicated to it.

</Recall>

<Recall term="Statistic and sampling distribution">

A **statistic** is any number you compute from your data — the sample mean, the difference of two group averages, the correlation between two variables, etc. A **sampling distribution** is the distribution of that number *if you were to redo the experiment many times with fresh samples*. Every test in this chapter works by comparing the *observed* value of some statistic to its sampling distribution under H₀ (the "null distribution").

</Recall>

<Recall term="Confidence interval (CI)">

A **95% confidence interval** for a parameter is a range constructed from your data such that, if you repeated the experiment many times and rebuilt the interval each time, 95% of those intervals would contain the true parameter value. Rough intuition (not exactly right, but useful): "a range of values that are plausibly the true parameter given what you saw." Built in detail in the previous chapter.

</Recall>

## The null-alternative framework

Recipe, four steps, always the same shape.

**Step 1: Write down the claim.** You want to argue for something (H₁: "the new prompt works better"). Its negation is the boring default (H₀: "the new prompt is no different from the old one"). Write both down before looking at the data. The choice of H₀ is often obvious but not always — pick the version that is *conservative*: the story you would be embarrassed to abandon on shaky evidence.

**Step 2: Pick a test statistic.** A **test statistic** T = T(data) is any number computed from the data whose distribution *under H₀* you can either derive by formula (parametric tests) or simulate (permutation tests). Common test statistics: the difference of two group means, standardised by their combined noise; a count-mismatch score for tables; a rank sum for non-parametric setups. Compute T from your observed data — call that number `t_obs`.

**Step 3: Compute the p-value.** The p-value is the probability, under H₀, of getting a test statistic *at least as extreme* as the one you observed. "Extreme" is directional. There are two flavours depending on H₁:

- **Two-sided** ("H₁ is: any difference"): extreme = far from zero in either direction. Formula:

$$
p \;=\; P(|T| \geq |t_{\text{obs}}| \mid H_0)
$$

Read the vertical bar `|·|` as absolute value; the `|` inside the probability means "given" (conditional probability).

- **One-sided** ("H₁ is: the effect goes in this specific direction, say positive"): extreme = far in that direction only.

$$
p \;=\; P(T \geq t_{\text{obs}} \mid H_0)
$$

Rule of thumb: use two-sided unless you have a compelling directional prior — many analysts default to two-sided to avoid the appearance of cherry-picking.

**Step 4: Compare p to α and decide.** If p &lt; α, reject H₀. Otherwise fail to reject. Report the number, not just the verdict.

**A fundamental fact worth memorising**: under H₀, **p-values themselves are Uniform(0, 1)** distributed — every value in [0, 1] is equally likely. This is not an approximation, it is a definitional property. It has one very concrete consequence: if you reject whenever p &lt; α, and H₀ actually is true, then you falsely reject exactly α fraction of the time. Setting α = 0.05 buys you a guaranteed 5% false-positive rate — no more, no less.

Under H₁ (the alternative — some real effect exists), p-values are **stochastically smaller** than uniform — they pile up near 0. The technical name for "how much they pile up near 0" is **power**, defined as

$$
\text{power} \;=\; P(\text{reject } H_0 \mid H_1 \text{ true}).
$$

Higher power = more likely to correctly detect a real effect. Power grows with (a) sample size n, (b) the true effect size, and (c) α (a laxer threshold rejects more easily). The next chapter (ab-testing-power) works out the arithmetic.

## What p-values are (and, especially, are NOT)

The p-value is the most-misinterpreted number in all of statistics. Six wrong readings, in rough order of how often people commit them, each followed by why it is wrong.

1. **"p is the probability that H₀ is true."** — NO. That would be `P(H₀ | data)`, a **posterior probability** (a Bayesian object that requires assuming a **prior** — a probability distribution over the hypothesis *before* seeing data; introduced in the MAP chapter). The p-value is `P(data | H₀)`. Swapping the conditioning order changes the meaning entirely. Think of it like `P(rain | umbrella)` vs `P(umbrella | rain)` — related, not equal.

2. **"p is the probability that H₁ is false."** — NO. Same mistake as #1 with the hypothesis flipped.

3. **"p = 0.03 means 3% chance of a false positive on this specific test."** — NO. Your Type I error rate is α, set at the design stage, and is a *long-run* frequency across many hypothetical repeats of the experiment — not a probability tied to any single test result.

4. **"p &gt; 0.05 means H₀ is true."** — NO. Absence of evidence is not evidence of absence. You may have too small a sample (low power) to detect a real effect. Failing to reject H₀ is a decision to withhold judgement, not a positive assertion of the null.

5. **"p is the size / importance of the effect."** — NO. p tells you *whether* the pattern is noise, not *how big* it is. With a huge sample size (n = 10 million) you can achieve p &lt; 0.001 for a completely negligible 0.001% lift. With a small sample (n = 20) you can have a genuinely important effect and still see p = 0.3.

6. **"A p-value from a study with n = 10⁶ is comparable to one from a study with n = 20."** — NO. Same nominal p-value, wildly different amounts of evidence about the *practical* size of the effect. Always look at n and effect size together with p.

What p **is**: the probability of observing data at least as extreme as yours *if H₀ were true*. That is it. It is a tool for calibrating evidence against one specific null hypothesis, no more.

Best practice, always: **report the p-value alongside the observed effect size and its confidence interval.** The p-value answers "is this pattern noise-explainable?"; the effect size answers "how big is the difference?"; the CI answers "how certain am I about that size?". Any one of the three on its own can mislead. All three together give a defensible picture.

## Type I and Type II errors, and power

Any binary decision (reject vs fail-to-reject) against a binary truth (H₀ true vs false) has exactly four possible outcomes. Two are correct, two are errors, and the errors have their own names:

| Reality ↓ · Decision → | Reject H₀ | Fail to reject H₀ |
|---|---|---|
| **H₀ true** | **Type I error** — false alarm (rate α) | ✓ correct |
| **H₀ false** | ✓ correct | **Type II error** — missed effect (rate β) |

Definitions in plain English:

- **α** = the false-alarm rate you tolerate = `P(reject H₀ | H₀ actually true)`. Also called the **significance level**. You choose it before running the test (traditionally 0.05, but 0.01 or 0.001 for higher-stakes decisions). Setting α smaller means fewer false alarms, at the cost of missing more real effects.
- **β** = the miss rate = `P(fail to reject H₀ | H₀ actually false)`. Unlike α, β is *not* something you set directly — it emerges from your sample size, the noise in your data, and the true effect size.
- **Power** = `1 − β` = the probability of *catching* a real effect when it exists. Higher power = better detector. Power of 0.8 means "if the effect I care about is really there, my test will notice it 80% of the time."

You control power by pulling three levers, for a fixed α:

1. **More data** (increase n). Extra samples shrink the noise in your statistic, so real effects poke through the noise more easily. Power grows roughly like √n.
2. **Reduce noise** (better measurement). If you can lower the standard deviation of your observations (cleaner metric, more careful measurement, or **stratification** — analysing within homogeneous subgroups — introduced in later chapters), power rises.
3. **Loosen α**. Higher α = easier to reject → catches more real effects, but at the price of more false alarms. This is the fundamental trade-off; you cannot reduce both α and β simultaneously without more data.

The industry-standard target is **power ≥ 0.80 at your intended effect size, with α = 0.05**. That is: "if the effect I want to detect really exists at the size I care about, I have at least an 80% chance of detecting it, while tolerating a 5% false-alarm rate." A test with less than 0.5 power is nearly useless — it will miss real effects most of the time, so a non-significant result tells you almost nothing. The next chapter (ab-testing-power) derives the sample size formula that follows from this target.

## Common tests

Three tests cover ~95% of what you will need. Each is the same four-step recipe with different specifics.

### One-sample t-test — "does the mean equal some target?"

**Use when.** You have one group of numeric measurements and you want to check whether their true average equals some pre-specified value μ₀ ("mu-nought"). Example: "does our chatbot average at least 4.0 stars?" → H₀: μ = 4.0.

**Data.** X_1, …, X_n — n numeric observations from a single group.

**Statistic.** Standardise the deviation of the observed average X̄ ("x-bar", the sample mean) from the target μ₀ by the estimated noise:

$$
T = \frac{\bar X - \mu_0}{s / \sqrt{n}}
$$

- `s` = the sample standard deviation (the usual noise estimate — how much observations typically deviate from the mean).
- `s / √n` = the **standard error of the mean** — the noise in `X̄` itself, which is √n times smaller than the noise in a single observation.

**Null distribution.** Under H₀, T follows a **Student's t distribution** with `n − 1` **degrees of freedom** (a "degree of freedom" here is just a parameter of the t distribution shape; for large n the t distribution looks essentially like a standard normal 𝒩(0, 1), and for small n it has fatter tails to compensate for the noisy s). Use `scipy.stats.ttest_1samp` in code.

### Two-sample (Welch's) t-test — "do two groups have the same mean?"

**Use when.** You have two independent groups and want to compare their averages. **This is the workhorse of A/B testing for any continuous metric.**

**Data.** Control group: `X_1^C, …, X_{n_C}^C`. Treatment group: `X_1^T, …, X_{n_T}^T`.

**Statistic.** Difference of averages, standardised by the combined noise:

$$
T = \frac{\bar X_T - \bar X_C}{\sqrt{s_C^2 / n_C + s_T^2 / n_T}}
$$

The denominator is the **standard error of the difference** — you get it by summing the variances of the two group means (since they are independent) and taking the square root.

**Null distribution.** Assuming the group averages are approximately normal (which by the Central Limit Theorem — see clt-lln — holds for any n ≥ 30 or so), T follows a t distribution. The exact "degrees of freedom" is a slightly ugly formula called **Welch–Satterthwaite** — you never need to compute it by hand, `scipy.stats.ttest_ind(..., equal_var=False)` does it for you. `equal_var=False` says "don't assume the two groups have equal variances" — safer default in real data.

### Chi-squared test — "are two categorical variables independent?"

**Use when.** Your data is a **contingency table** — a table of counts cross-classified by two categorical variables. Example: rows = "user segment" (new / returning / power); columns = "clicked or not"; each cell contains the number of users in that segment × click combo. H₀: the two classifications are independent (segment and click behaviour are unrelated).

**Statistic.** For each cell (i, j), compare the observed count `O_ij` to the count `E_ij` you would *expect* if H₀ (independence) held. The expected count is `E_ij = (row i total) × (column j total) / (grand total)` — the multiplication rule for independent events. Sum the standardised squared errors:

$$
\chi^2 \;=\; \sum_{i, j} \frac{(O_{ij} - E_{ij})^2}{E_{ij}}
$$

Read `χ²` as "chi-squared" (Greek chi). Each term measures how surprised you should be by that cell.

**Null distribution.** Under H₀, χ² follows a **chi-squared distribution** with `(r − 1)(c − 1)` degrees of freedom, where r and c are the number of rows and columns. Use `scipy.stats.chi2_contingency`. Rule of thumb: this test is only reliable when every expected count is at least 5 — for smaller tables use **Fisher's exact test** (see NewConcepts).

### Permutation (randomisation) test — "compare anything, no assumptions"

**Use when.** You do not want to assume any particular distribution for your data (small n, heavily skewed, or you are comparing a weird custom statistic that has no closed-form null distribution). This is the **non-parametric** option — "non-parametric" meaning "does not assume a specific parametric family like Gaussian or Poisson."

**Recipe.**

1. Compute your statistic on the real data — call it `t_obs`.
2. Merge the two groups into one big pile of observations.
3. Randomly shuffle the group labels many times (say 5000). For each shuffle, compute the same statistic. This gives a **null distribution** built by simulation.
4. The p-value is the fraction of shuffled statistics that are at least as extreme as `t_obs`.

**Why it works.** Under H₀ of no difference between groups, the group labels are meaningless — any assignment of labels to observations is equally plausible. So the distribution of your statistic across all possible label shuffles *is* the null distribution, exactly. The formal name for this assumption is **exchangeability** ("swapping observations across groups does not change the joint distribution of the data").

**Trade-offs.**

- **Pros**: no distributional assumption; works with *any* statistic (median difference, correlation, custom score, F1); exact under H₀.
- **Cons**: computationally expensive if the statistic is slow to compute; exchangeability can fail in structured data (paired observations, time series).

## Multiple comparisons

Here is the problem in one sentence: **if you run many tests at α = 0.05, you will get false positives even when every null is true, purely by chance.**

Concretely: k independent tests at α = 0.05 give you an expected `k × 0.05` false positives under all nulls. Twenty tests → one expected false positive. A hundred tests → five. And the probability of at least *one* false positive is `1 − (1 − α)^k`, which climbs fast: for k = 20 it is already 64%. This is called the **multiple comparisons problem**, and it is why you should treat any benchmark leaderboard that reports many pairwise comparisons without correction with heavy skepticism — the top model is often the *luckiest*, not the best.

There are two standard fixes. They target different guarantees; pick the one whose guarantee you want.

### Bonferroni — controls the family-wise error rate (FWER)

The **family-wise error rate (FWER)** is the probability of making *at least one* false rejection across the whole family of k tests. The Bonferroni recipe is trivially simple:

- **Rule**: reject `H_i` if and only if `p_i < α / k`.

That is, tighten the threshold from α to α/k. Setting k = 20 and α = 0.05 means each individual test has to hit p &lt; 0.0025 to count.

**Guarantee**: `P(any false rejection | all nulls true) ≤ α`. Exactly what FWER means.

**Downside**: extremely conservative. When many tests actually contain real effects, Bonferroni loses power — it will miss most of them because it demands very small p-values from each.

**Use when**: even a single false positive is costly. Medical decisions, high-stakes deployment gates, regulatory settings.

### Benjamini–Hochberg — controls the false discovery rate (FDR)

The **false discovery rate (FDR)** is a different, softer guarantee: the *expected fraction* of your rejections that are false. If you make 20 rejections and 3 are false, your realised false-discovery-fraction is 3/20 = 15%. BH controls the expected value of this fraction at some level q (typically 0.05 or 0.10).

**Recipe (BH):**

1. Sort your k p-values ascending: `p_(1) ≤ p_(2) ≤ … ≤ p_(k)`.
2. Find the largest `i` such that `p_(i) ≤ (i / k) · q`.
3. Reject every hypothesis with `p ≤ p_(i)`.

**Guarantee**: `E[false discoveries / total discoveries] ≤ q`. Note: an *expected fraction*, not a probability of any false positive.

**Trade-off vs Bonferroni**: BH is more powerful (rejects more real effects) but allows a small fraction of your discoveries to be false. In genomics, where you might be testing 20,000 genes at once, BH is standard because Bonferroni would be so strict as to reject almost nothing. In ML benchmark evaluation, BH is increasingly the norm for multi-metric leaderboards.

**Use when**: some false discoveries among many claims are tolerable. Genomics, exploratory data analysis, large-scale benchmark comparisons.

Rule of thumb: **Bonferroni for high-stakes single-shot decisions; BH-FDR for exploratory science where you expect many real effects among many tests.**

## The traps that matter

**Trap 1: `p ≠ P(H₀ is true)`.** Say this aloud until it is second nature. The p-value is the probability of the *data* given the hypothesis, not of the *hypothesis* given the data. If you want `P(H₀ | data)`, you need a Bayesian analysis with an explicit prior (see the map-and-priors chapter).

**Trap 2: statistical significance ≠ practical significance.** With enough data, arbitrarily tiny effects reach any nominal significance level. A p &lt; 0.001 on a lift of 0.02% with n = 10 million tells you the pattern is real, but not that anyone should care. Always translate the effect into the real-world units that matter to your problem, and ask "would we ship for a change this size?" *before* looking at the p-value.

**Trap 3: p-hacking.** Any of the following invalidates the frequentist guarantee that α is your false-positive rate:

- Running many alternate tests and reporting whichever one hit p &lt; 0.05.
- Trying alternate metrics (accuracy, F1, MRR, …) and reporting the best.
- Slicing the data by subgroup and reporting the significant slice.
- Picking your test statistic or one-sided vs two-sided *after* seeing the data.
- Stopping when p first goes below α (see Trap 4).

The umbrella term is **p-hacking** (also called "the garden of forking paths"). The fix is **preregistration** (see NewConcepts) — commit in writing to your primary metric, test, sample size, and analysis plan *before* looking at the data.

**Trap 4: peeking.** In an online A/B test the temptation is to check the dashboard every few hours and stop the moment you see p &lt; 0.05. This wrecks the false-positive guarantee: each peek is another chance to falsely reject, and the probabilities compound. Five naïve peeks at α = 0.05 give you an actual FWER of ~23%, not 5%.

Fixes:

- **Fixed-horizon test**: commit to n in advance, look exactly once, at the end. Traditional and correct but slow.
- **Group-sequential methods** (e.g., **O'Brien–Fleming**, **Pocock** — pre-specified peek schedules with tighter thresholds at each interim look). Maintain FWER while allowing interim analyses.
- **Always-valid p-values / confidence sequences** (Howard, Ramdas, and collaborators, 2018+): modern methods that allow *arbitrary* continuous monitoring while maintaining valid Type I control.
- **Bayesian A/B testing** — monitor `P(treatment better | data)` directly; no equivalent of a peeking penalty. Covered in ab-testing-power.

**Trap 5: multiple comparisons ignored.** Any workflow that runs many tests at α = 0.05 without correction — leaderboard-style pairwise comparisons across models, per-metric evaluation on many benchmarks, subgroup analyses — will produce false positives that fail to replicate. Apply Bonferroni (strict) or BH-FDR (softer).

**Trap 6: small-n tests are underpowered.** Failing to reject with n = 20 tells you almost nothing — you likely lacked the power to detect a real effect. Compute power at your design stage; if it is below 0.5, either redesign the study or collect more data before running.

**Trap 7: "reject vs fail to reject" as a hard binary.** The α = 0.05 line is a bureaucratic bookkeeping convention, not a real-world threshold. p = 0.049 and p = 0.051 carry essentially identical evidence, yet one "rejects" and one does not. The American Statistical Association (in a 2016 statement) explicitly warned against binary p &lt; 0.05 thinking. Modern practice: report the actual p-value, effect size, and CI; let the reader decide.

**Trap 8: wrong test for the data.** A t-test on heavily skewed data with small n can give inaccurate p-values (violates the near-normality assumption). A chi-squared test on a contingency table with expected counts below 5 is unreliable (use **Fisher's exact test** — see NewConcepts — instead). A permutation test on 10 million observations may be too slow for real-time A/B analytics. Match the tool to the data.

<RealWorldML title="Statistical significance in ML: releases, benchmarks, RLHF, and evaluation">

Four places you will actually apply this material as a GenAI engineer. Each maps a real workflow to the machinery above, with the jargon unpacked in place.

**1. A/B testing an LLM prompt change.**

You update the system prompt for your chatbot and want to know whether it improved user satisfaction. The metric is a 1-to-5 star rating averaged per user over a session. Recipe:

- Randomly assign 5000 users to "control" (old prompt) and 5000 to "treatment" (new prompt).
- Compute the average rating in each group and its standard error.
- Run a two-sample Welch's t-test on the difference.
- If p &lt; 0.05 AND the effect size is above your business-decided threshold (say ≥ 0.05 stars), ship the new prompt.

Two things to watch for. First, **peeking**: if you look at daily results and stop when p first crosses 0.05, you inflate the false-positive rate — a shipping decision you regret later. Second, **effect size vs significance**: with 5000 users on each side you may achieve significance for a 0.02-star lift that is completely invisible to users.

For teams doing this at industrial scale, **Bayesian A/B testing** is often preferred — you monitor `P(treatment better than control | data)` directly, and there is no peeking penalty. Two common patterns: a **Beta-Binomial** model for binary metrics like conversion (Beta = a distribution on probabilities, Binomial = counts of successes; the pair is a **conjugate** pairing meaning the posterior stays Beta as data arrives), and a **hierarchical Gaussian** for continuous metrics (a Gaussian on each user's mean with a Gaussian prior over the group mean). The next chapter covers the design side of both frequentist and Bayesian A/B tests.

**2. Benchmark leaderboards.**

Every large-language-model benchmark you have heard of — **MMLU** (57-subject multiple-choice knowledge test), **GSM8K** (grade-school math word problems), **HumanEval** (code-generation from docstrings), **LMSYS Chatbot Arena** (an online platform where humans pairwise-rank two anonymous model answers, aggregated into an **Elo rating** — the same rating system used in chess) — reports accuracy numbers. When one model beats another by 0.3% on MMLU, the community's first question is "is that inside the noise of the evaluation?"

Answer: compute a **bootstrap confidence interval** on the accuracy (the bootstrap = a resampling technique for CIs, covered in the confidence-bootstrap chapter). If the two CIs overlap heavily, the 0.3% "improvement" is likely just noise. Modern eval papers report this; older ones frequently did not.

**3. RLHF preference significance.**

**RLHF** (Reinforcement Learning from Human Feedback) trains an LLM to prefer outputs humans like. The pipeline collects **pairwise preferences**: annotators see two completions from the model and pick the better one. To decide which checkpoint to ship, you often do a paired comparison — for the same prompt, annotators vote on completion A vs completion B, and you count wins.

The right test for "did A beat B significantly?" on paired vote data is **McNemar's test** (see NewConcepts — a chi-squared variant for paired 2×2 tables). The relevant twist for RLHF: because you compare many checkpoints against many prompts, you must apply a **multiple-comparison correction** (BH-FDR is the usual choice). Without it, the "best" checkpoint across many comparisons is often the luckiest one.

**4. LLM eval multi-metric bookkeeping.**

Serious LLM evaluation uses many metrics at once — MMLU, GSM8K, HumanEval, MATH (competition-level math), GPQA (graduate-level physics/biology/chemistry questions), LMArena Elo, plus internal benchmarks. Running a t-test on each pairwise comparison without correction produces many spurious "significant" wins. Apply BH-FDR across the whole family of tests, and report "k of these m metrics show a genuine improvement at FDR q = 0.05" — a defensible headline instead of a fragile per-metric claim.

**5. Continuous monitoring (the peeking fix again).**

If your production dashboard shows a live p-value on a running A/B test, you *will* be tempted to stop when it dips below 0.05. Use either group-sequential methods (O'Brien–Fleming, Pocock) with pre-specified interim looks, or modern "always-valid" methods (confidence sequences) that maintain valid Type I control at every glance. Both approaches are what large experimentation platforms (Netflix, Meta, LinkedIn) run under the hood.

<Summary>

- **Hypothesis test** = &ldquo;is my observation too unusual under H₀ to keep believing H₀?&rdquo;
- **p-value** = P(data at least this extreme | H₀). Small p → evidence against H₀. NOT P(H₀ is true).
- **Under H₀, p-values are Uniform(0, 1)**; rejecting at α gives false-positive rate α.
- **Type I** = reject when H₀ true (rate α). **Type II** = miss when H₀ false (rate β). **Power** = 1 − β.
- **t-test** (means), **chi-squared** (contingency), **permutation** (non-parametric) are the three most-used tests.
- **Multiple comparisons**: k tests at α gives ~α·k false positives. Correct with **Bonferroni** (FWER) or **BH-FDR** (FDR).
- **Report p-value + effect size + CI**. Statistical significance is not practical significance. Non-significant is not &ldquo;no effect.&rdquo;
- **Traps**: p-hacking, peeking, multiple-comparison ignorance, tiny-effect-with-huge-N.
