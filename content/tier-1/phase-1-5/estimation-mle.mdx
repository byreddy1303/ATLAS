export const estimatorCode = `# An estimator is a function of the data that produces a guess for a parameter.
# Compare two estimators of the mean of iid samples from a distribution.
import numpy as np
rng = np.random.default_rng(0)

# True population: N(mu = 5, sigma^2 = 4).
true_mu = 5.0
n = 30

# Estimator 1: sample mean.  Unbiased, consistent.
# Estimator 2: median.  Also unbiased and consistent for a symmetric distribution,
#              but with LARGER variance than the mean for Gaussian data.
n_reps = 5000
means, medians = [], []
for _ in range(n_reps):
    x = rng.normal(true_mu, 2.0, size=n)
    means.append(x.mean())
    medians.append(np.median(x))

print(f"Estimator: sample mean")
print(f"  mean of estimator across {n_reps} experiments = {np.mean(means):.4f}  (should equal true_mu = 5)")
print(f"  variance of estimator                                    = {np.var(means):.4f}")

print(f"\\nEstimator: sample median")
print(f"  mean of estimator                                        = {np.mean(medians):.4f}")
print(f"  variance of estimator                                    = {np.var(medians):.4f}")

# Efficiency = variance ratio.  For Gaussian data, mean is more efficient than median.
efficiency = np.var(means) / np.var(medians)
print(f"\\nVariance ratio (mean / median) = {efficiency:.4f}")
print("For Gaussian data, the sample mean has variance ~2/pi ~= 0.637 of the median's variance.")
print("=> The mean is a more EFFICIENT estimator (smaller variance) for Gaussian populations.")`;

# For iid data, L(theta) = product of per-sample densities.

# Observed data: 20 samples from N(mu = 3, sigma = 1).

export const mleBernoulliCode = `# MLE for Bernoulli(p): derive by hand, then verify numerically.
# Data: n coin flips, k of which are heads (successes).
# L(p) = product of p^(x_i) * (1-p)^(1-x_i) = p^k * (1-p)^(n-k)
# log L(p) = k * log(p) + (n - k) * log(1 - p)
# d/dp log L = k/p - (n-k)/(1-p) = 0
# =>  k(1-p) = (n-k)p  =>  k - kp = np - kp  =>  k = np  =>  p = k / n.
# MLE for a Bernoulli's success probability is just the empirical frequency.
import numpy as np
from scipy.optimize import minimize_scalar
rng = np.random.default_rng(0)

# Simulate a biased coin with true p = 0.35, observe 100 flips.
true_p = 0.35
n = 100
data = rng.binomial(1, true_p, size=n)
k = data.sum()

# Closed-form MLE.
p_mle = k / n
print(f"Data: {k} heads in {n} flips.")
print(f"Closed-form MLE: p_hat = k/n = {p_mle:.4f}")

# Verify by numerical optimization of the negative log-likelihood.
def neg_log_likelihood(p):
    if p <= 0 or p >= 1:
        return np.inf
    return -(k * np.log(p) + (n - k) * np.log(1 - p))

result = minimize_scalar(neg_log_likelihood, bounds=(1e-6, 1 - 1e-6), method="bounded")
print(f"Numerical MLE (via scipy):                {result.x:.4f}")
print(f"True parameter (would be unknown in practice): {true_p}")

# The MLE is a random variable -- its value depends on the sample.
# For large n, MLE for Bernoulli has variance p*(1-p)/n (Cramer-Rao bound).
CRB_var = true_p * (1 - true_p) / n
print(f"\\nCramer-Rao lower bound for Var(p_hat): {CRB_var:.6f}")

# Verify empirically by repeating the experiment 5000 times.
mles = [rng.binomial(1, true_p, size=n).sum() / n for _ in range(5000)]
print(f"Empirical Var(p_hat) over 5000 experiments: {np.var(mles):.6f}   (matches CRB up to sampling noise)")`;

# log L(mu, sigma^2) = sum_i log( (1 / (sigma * sqrt(2*pi))) * exp(-(x_i - mu)^2 / (2 sigma^2)) )
#                    = -n/2 * log(2*pi*sigma^2) - sum_i (x_i - mu)^2 / (2 sigma^2)

#

# For large N the difference vanishes; for small N it matters.
import numpy as np
rng = np.random.default_rng(0)

true_mu, true_sigma = 3.0, 2.0
n = 50
data = rng.normal(true_mu, true_sigma, size=n)

mu_mle = data.mean()
sigma2_mle_uses_N = ((data - mu_mle) ** 2).mean()          # MLE -- N in denominator
sigma2_unbiased  = ((data - mu_mle) ** 2).sum() / (n - 1)   # Bessel-corrected

print(f"True: mu = {true_mu}, sigma^2 = {true_sigma**2}")
print(f"MLE for mu:     mu_hat = {mu_mle:.4f}")
print(f"MLE for sigma^2:        {sigma2_mle_uses_N:.4f}   (uses N, biased-low)")
print(f"Unbiased sigma^2:       {sigma2_unbiased:.4f}     (uses N-1)")

# MSE loss in ML regression = MLE for Gaussian noise with fixed sigma.
# The connection: minimizing sum_i (y_i - f(x_i))^2 is equivalent to maximizing
# log-likelihood under y_i ~ N(f(x_i), sigma^2) with sigma fixed.

w_ls, *_ = np.linalg.lstsq(X, y, rcond=None)

<Motivation title="Every ML loss function is the same idea in a costume">

Here's a wild fact. Almost every "loss function" you'll ever meet in machine learning — the number the model tries to make smaller during training — is secretly the same idea wearing different costumes. That idea has a name: **maximum likelihood**, or MLE for short (the "E" is for "estimation"). Once you see it, huge chunks of ML stop looking like a pile of unrelated recipes and start looking like variations on one theme.

Here's the honest list of what turns out to be MLE in disguise:

- **Mean-squared-error regression** (predicting a number and measuring how far off you were). This is MLE if you assume the true value is your prediction plus a bit of bell-curve-shaped random noise.
- **Cross-entropy classification** (predicting which class something belongs to). This is MLE if you assume the model's output is a probability over classes and the true label came from that distribution.
- **VAEs and diffusion models** (the image generators). Both train by pushing up a number that's essentially "how probable is the real data under my model" — that's likelihood.
- **Language model pretraining** (how GPT learns from text). At every position, GPT predicts the next word as a probability distribution over the vocabulary, and the training loss is `−sum log P(next word | previous words)` — again, likelihood.
- **Contrastive losses** like SimCLR (self-supervised learning). Reformulate them and they turn into "which of these pairs actually go together?" — a classification likelihood.

Two quick vocabulary unpacks before we go further, so nothing feels scary:

- **Distribution**: a rule that assigns probability (or density) to every possible outcome. Rolling a fair die is a distribution over &#123;1,…,6&#125;.
- **Parameter**: the "knob" that shapes the distribution. For a coin, the knob is the probability of heads. For a bell curve, the knobs are the center (μ, "mu") and the width (σ, "sigma").

Understanding MLE means understanding *why* all those losses are the right objectives — not just accepting them as recipes handed down from on high. You also get three properties for free: with enough data the estimator lands on the truth, it uses the data about as well as any estimator possibly could, and its errors follow a predictable bell-curve shape so you can put error bars on your answer.

This chapter derives MLE by hand for the two most-used distributions (coin flip and bell curve), shows why every gradient step your model takes is really climbing a likelihood hill, and gives you the vocabulary to spot MLE hiding inside a paper you'll read next month.

</Motivation>

<Intuition title="MLE = 'pick the setting that makes what you actually saw the least surprising'">

Let's start with a real question, no math. You flip a coin 10 times and get 7 heads. What's a reasonable guess for the coin's true probability of heads?

Try a few candidates and just see which fits best.

- If the coin were fair (p = 0.5), the probability of seeing 7-out-of-10 heads is about 12%.
- If the coin were biased toward heads at p = 0.7, the same 7-out-of-10 outcome has probability about 27%. More than twice as likely.
- If p = 0.9, the outcome has probability about 6% — less likely, because 0.9 usually gives even more heads than 7.

So of the three, p = 0.7 makes the thing you actually saw feel least surprising. If we sweep across every p from 0 to 1, the winner is exactly p̂ = 7/10 = 0.7 — the "hat" (^) just means "estimated from data." That's the whole idea of MLE in one line: **pick the knob setting that gives your actual observation the highest probability**.

Zooming out. You've picked a family of distributions — coin flips, bell curves, whatever — and it has a knob called θ (theta, the Greek letter for "the parameter to figure out"). Different settings of θ give the data D (that's just "the data you saw") different amounts of probability. MLE says: pick the θ that assigns the highest probability to your data.

We write this as:

$$
\hat\theta_{\text{MLE}} \;=\; \arg\max_{\theta}\; L(\theta) \;=\; \arg\max_\theta \sum_i \log p(x_i \mid \theta).
$$

Two symbols to unpack so this doesn't look scary:

- **L(θ)** is the **likelihood** — a fancy name for "probability of the data, as a function of the knob θ." Different θ, different likelihood.
- **arg max** just means "the value of θ that makes L biggest." (arg max = "argument that maximizes.")

You'll notice we switched to summing the **log** of probabilities. That's a computational trick, not a conceptual one. Multiplying thousands of tiny numbers (like 0.001 × 0.002 × …) crashes to zero on a computer; adding their logs stays sane. And the log doesn't move the peak — whatever θ was biggest for L is still biggest for log L. So we almost always work with log-likelihood.

Here's the mental picture. Plot L(θ) as a curve over θ. There's a peak somewhere — MLE is you pointing at the peak. How *sharp* that peak is tells you how certain you are: a sharp peak = "the data pinned θ down tightly"; a flat peak = "many θ values look about equally good." The sharpness has a name (Fisher information, discussed later) but you don't need it right now.

</Intuition>

<Recall term="Estimator">

An **estimator** is any rule that turns raw data into a guess about some hidden number. If you have samples X₁, …, Xₙ and you want to guess the population's true mean, "just compute the average" is an estimator. So is "pick the middle value" (median). Because the data you got is random, your guess is random too — get a different sample, get a different guess. The pattern of guesses you'd get across many hypothetical resamples is called the estimator's **sampling distribution**. Three numbers describe how good the estimator is: **bias** (how far off on average from the truth), **variance** (how jumpy the guess is), and **mean squared error** (bias² + variance — total badness).

</Recall>

<Recall term="Population vs sample">

**Population** = the whole thing you want to know about (e.g., "all users of the app"). **Sample** = the small random slice you actually observed (e.g., "the 200 users we surveyed"). The population has a true parameter (like a true average revenue per user) — we can't see it directly. All estimation is about using the sample to guess about the population.

</Recall>

## What makes a good estimator?

An **estimator** is just a rule for turning data into a guess. "Compute the mean" is an estimator. "Compute the median" is a different estimator. Both can be aimed at the same target (say, the true average of a population), but they behave differently. Five things people care about:

- **Bias.** If you ran the same estimator on infinitely many independent samples and averaged the answers, would you land on the truth? If yes, the estimator is *unbiased*. The sample mean is unbiased for the population mean. Sample variance with N in the denominator is slightly *biased low* — it tends to underestimate the true spread. That's why textbooks use N − 1 for the "unbiased" version (this is called **Bessel's correction**, just a name for the "use N − 1" trick).

- **Variance.** How jittery is the estimator? If your sample changes a bit, does your guess bounce around a lot or barely move? Lower variance = more reliable.

- **Consistency.** As your sample gets bigger and bigger, does the estimator get closer and closer to the truth? Any useful estimator has to be consistent. If it doesn't shrink onto the truth with unlimited data, it's broken.

- **Efficiency.** Given the sample you have, is your estimator wringing out as much information as theoretically possible? There's a lower bound on how small any estimator's variance can be (called the **Cramér–Rao lower bound**, or CRB for short — that's just its historical name, don't be scared by it). An estimator that hits this bound is called *efficient*. Good news: MLE hits this bound when you have lots of data.

- **Robustness.** If a few data points are garbage (a sensor glitched, someone typed in a wrong number), how badly does that mess up your estimate? The median is robust — a few outliers barely move it. The mean is not — one huge outlier drags it a lot.

**A key identity you'll use forever**: MSE = Bias² + Variance. Total error breaks cleanly into "how far off on average" plus "how jittery." You can often trade one for the other. Adding a tiny bias can shrink variance so much that MSE drops — this is exactly what regularization does (next chapter), and it's why real-world estimators are often deliberately biased.

## The likelihood function, unpacked

Now let's make "likelihood" concrete instead of scary. Pick a family of distributions — say bell curves 𝒩(μ, σ²), or coin flips Bernoulli(p) — and imagine you got data D = (x₁, …, xₙ). Two words to unpack up front:

- **iid** is a three-letter acronym for **independent and identically distributed**. "Independent" = one sample doesn't affect the next. "Identically distributed" = every sample came from the same underlying process. Basically: n clean draws from the same jar.
- **Parametric family** is fancy for "a distribution with a knob." Coin flips have one knob (probability of heads). Bell curves have two knobs (center and width).

Given a knob setting θ, the **likelihood** is just the probability the model assigns to the data you actually saw:

$$
L(\theta) \;=\; p(D \mid \theta) \;=\; \prod_{i=1}^n p(x_i \mid \theta).
$$

Read this out loud: "L of θ equals the product, over all your samples, of the probability of each sample given θ." The product is there because with iid samples, joint probability = product of individual probabilities.

One warning before we go further. When your data is continuous (like weights, heights, pixels), p(xᵢ | θ) is a **density** — the height of a probability curve — not a probability. Densities can be bigger than 1 (they only need to integrate to 1). That's fine. All you care about is *comparing* L(θ) across different θ, not the raw number.

The **log-likelihood** is what we actually optimize:

$$
\ell(\theta) \;=\; \log L(\theta) \;=\; \sum_{i=1}^n \log p(x_i \mid \theta).
$$

The lowercase ℓ (script "l") is just "log-likelihood." The reason to take the log is practical, not deep:

1. **Numbers stop crashing to zero.** Multiplying a million probabilities of 0.01 gives you 10⁻²⁰⁰⁰, which any computer rounds to zero. Adding a million values of log(0.01) gives you a big negative number — sane and finite.
2. **Derivatives get easier.** log of a product = sum of logs. Sum of logs → sum of derivatives → much cleaner algebra.
3. **The math matches how we train models.** Log-likelihood is a sum over samples, which is exactly how mini-batch training works (you sum the loss over a batch and take a step).

## Doing the math: MLE for a coin flip

Time to actually turn the crank. Data: n coin flips, each xᵢ is either 0 (tails) or 1 (heads). Let k = x₁ + … + xₙ = the total number of heads. Model: each flip is a **Bernoulli(p)** — a fancy name for "coin flip with heads-probability p." So p(x=1) = p and p(x=0) = 1 − p, or written more compactly: p(x) = p^x · (1−p)^(1−x).

The likelihood is the product of these per-flip probabilities:

$$
L(p) \;=\; \prod_{i=1}^n p^{x_i} (1-p)^{1-x_i} \;=\; p^k (1-p)^{n-k}.
$$

That collapse in the middle happens because Σ xᵢ = k (heads) and n − k = tails.

Take the log to make life easier:

$$
\ell(p) \;=\; k \log p + (n - k) \log(1 - p).
$$

Now we find the peak. **Differentiate** (compute the slope) with respect to p and set it to zero — because the top of a hill has zero slope:

$$
\frac{d \ell}{d p} \;=\; \frac{k}{p} - \frac{n - k}{1 - p} \;=\; 0
$$

Move things around to solve: k(1 − p) = (n − k)p, then expand: k − kp = np − kp, then the kp's cancel: k = np, so:

$$
\boxed{\;\hat p_{\text{MLE}} \;=\; \frac{k}{n}\;}
$$

— just the fraction of heads you observed. Which is what any beginner would guess before knowing calculus. But now you know *why* that guess is optimal, not just intuitive.

Why does this matter? Because this exact same formula shows up all over ML:

- **Naive Bayes classifier** (a simple spam filter): each word's per-class probability is estimated as "how often that word appears in class-emails / total class-emails" — that's k/n.
- **N-gram language models** (predicting the next word from raw counts): P(word | previous words) = "how many times we saw this word after that context / total times we saw that context" — again k/n.

Different application, same MLE.

## Doing the math: MLE for a bell curve (and why MSE loss exists)

Same routine, harder distribution. The **Gaussian** 𝒩(μ, σ²) is the bell curve: μ (mu) is the center where the peak sits, σ² (sigma-squared, the "variance") is how wide the curve is. Data: n samples drawn iid from this bell curve. The log-likelihood works out to:

$$
\ell(\mu, \sigma^2) \;=\; -\frac{n}{2} \log(2\pi \sigma^2) - \frac{1}{2\sigma^2} \sum_i (x_i - \mu)^2.
$$

Don't panic at the expression — it comes from taking log of the standard bell-curve density formula. The 2π constant is just the constant inside the density.

We need two derivatives now, one for each knob. When you have multiple knobs, "partial derivative" is the fancy term: take the derivative with respect to one knob while pretending the other is frozen. Notation: ∂ instead of d.

**Finding μ̂:** take ∂ℓ/∂μ, set to zero.

$$
\frac{\partial \ell}{\partial \mu} \;=\; \frac{1}{\sigma^2} \sum_i (x_i - \mu) \;=\; 0 \;\Rightarrow\; \boxed{\;\hat \mu_{\text{MLE}} \;=\; \bar x\;}
$$

That last symbol x̄ (x-bar) is just the sample mean — average all your x's. Neat detail: the answer doesn't involve σ² at all. Even if you don't know how wide the bell is, the best guess for its center is still just the average.

**Finding σ̂²:** take ∂ℓ/∂σ², set to zero.

$$
\frac{\partial \ell}{\partial \sigma^2} \;=\; -\frac{n}{2\sigma^2} + \frac{1}{2\sigma^4} \sum_i (x_i - \hat\mu)^2 \;=\; 0
$$

Solving:

$$
\boxed{\;\hat \sigma^2_{\text{MLE}} \;=\; \frac{1}{n} \sum_i (x_i - \hat \mu)^2\;}
$$

In English: the MLE for spread is the average squared distance from the sample mean. Reasonable.

**Watch out for one thing.** That formula divides by n. The classical "unbiased" formula divides by n − 1 (that's Bessel's correction we mentioned earlier). So MLE for variance is biased-low — it slightly underestimates how spread out the data really is. For small samples this matters; for large ones it washes out.

**And here's the payoff.** If you fit a line (or any function) to data by "least squares" — minimizing the sum of squared errors — you are secretly doing MLE, assuming the true value is your prediction plus bell-curve noise. Here's the math: if yᵢ = f(xᵢ, w) + εᵢ where εᵢ is small bell-curve-shaped random noise, the log-likelihood is:

$$
\ell(w) = -\frac{n}{2} \log(2\pi\sigma^2) - \frac{1}{2\sigma^2} \sum_i (y_i - f(x_i, w))^2.
$$

Maximizing this over the weights w is the same as *minimizing* Σ (yᵢ − f(xᵢ, w))² — the squared-error loss. So every time you write MSE loss in PyTorch, you are quietly telling the model "I assume your errors are bell-curve-shaped." Often that's fine. When it isn't (e.g., heavy-tailed errors), you might want a different loss.

## What MLE gives you for free when N is big

The word **asymptotic** just means "in the limit of lots of data" — think "what happens when n gets huge." MLE has three lovely asymptotic properties. These are why it's the default in statistics and ML:

1. **Consistency** — with enough data, MLE lands on the truth. Formally: as n grows, θ̂ gets arbitrarily close to the real θ. So more data = better answer, always.

2. **Bell-curve error** — the estimate's error follows a predictable bell curve. Specifically: √n · (θ̂ − θ) becomes bell-curve-shaped with width 1/√(Fisher info). Translation: your error bars shrink as 1/√n, and you can compute them.

3. **Best-possible precision** — for large n, MLE hits the Cramér–Rao lower bound, meaning no other consistent estimator can do better. You are extracting all the information the data has.

Two vocabulary unpacks:

- **Fisher information** I(θ) is the "sharpness of the peak" of the log-likelihood at the true θ. Sharp peak → high Fisher info → low variance in your estimate (you're pinning θ down). Flat peak → low Fisher info → high variance (θ is hard to pin down from this data). The formal formula is `I(θ) = −E[∂²ℓ/∂θ²]` — the expected second derivative (curvature) of the log-likelihood — but you don't need to compute it by hand day-to-day.
- **Cramér–Rao lower bound** (CRB): the theoretical floor for how small an unbiased estimator's variance can be. Reaching this floor means "you couldn't possibly have done better."

For a practicing engineer, all of this justifies two habits: (1) use MLE (or an approximation of it, like SGD on a loss function) as your default; (2) if you need error bars, get them from the shape of the log-likelihood near the peak (or from bootstrap resampling — see the next chapter). Every optimizer in `scipy.optimize` and every training loop in PyTorch is quietly hunting for the top of a likelihood hill.

## Things that trip people up

**"Likelihood is not the same as probability."** L(θ) tells you how well each θ explains the data — but it is *not* the probability that θ is correct. You cannot say "there's a 90% chance the true p is 0.5" from L alone. Doing that requires adding a prior belief and switching to Bayesian language (that's the next chapter, on MAP and priors).

**"MLE assumes the model shape is right."** If you assume Gaussian noise but reality is heavy-tailed (occasional huge errors), MLE will do its best under the wrong assumption. It won't yell — it'll just quietly give you a biased answer. So always sanity-check: do the residuals look bell-curve-ish? If not, either change the model or accept the caveat.

**"Zero-probability observations blow everything up."** If your model says "this outcome is impossible" but you observed it once, log-likelihood becomes log(0) = −∞, and the whole optimization crashes. Real fix: **additive smoothing** (also called Laplace smoothing) — pretend you saw every outcome one extra time. In language models this shows up as "add α to every count." In deep learning, softmax outputs never hit zero, which is one reason it's used.

**"MLE for spread is biased low."** The formula we derived divides by n, but the classical "unbiased" version divides by n − 1. For small samples this matters (say n &lt; 30). For big deep-learning batches nobody cares — n is huge, the difference vanishes.

**"MLE alone can overfit."** With flexible models (like deep networks), MLE will happily contort the model until it memorizes the training data — including the noise. That's called **overfitting**. Fix: add a **prior** (see the MAP chapter next) or use **regularization** (a soft penalty on how weird the weights get). Both are ways to say "MLE, but stay reasonable."

**"Log-space isn't optional."** Whenever you see `log_softmax` + `nll_loss` in PyTorch (instead of `softmax` + `log`), or `logsumexp` instead of `log(sum(exp(...)))`, that's the library saving you from underflow. Use the built-ins.

<RealWorldML title="Three ML losses that are secretly MLE">

Let's take three losses you'll actually meet in ML and show that each is MLE in disguise. If you get this, you can decode almost any paper you read for the rest of your career.

**1. Cross-entropy classification.** Your model outputs π(x, w) — a probability distribution over K classes, produced by softmax on the logits. **Softmax** is just "turn a vector of raw scores into probabilities that sum to 1." **Logits** are the raw scores before softmax. Given labeled data (xᵢ, yᵢ) — inputs with their true class — the likelihood of your labels under the model is:

$$
L(w) = \prod_i \pi_{y_i}(x_i, w).
$$

That subscript π_(y_i) means "the probability the model assigned to the *true* class for example i." Take log, get:

$$
\ell(w) = \sum_i \log \pi_{y_i}(x_i, w).
$$

Flip the sign, get the cross-entropy loss you've seen a hundred times. Minimizing cross-entropy *is* doing MLE, assuming your model outputs a valid probability distribution over classes.

**2. Language model pretraining (how GPT trains).** The model looks at the tokens so far and predicts the next one. A **token** is roughly a word or word-piece. The **vocabulary** V is all possible tokens. For each sentence x = (x₁, …, x_T):

$$
L(w) = \prod_t P(x_t \mid x_{<t}, w).
$$

The subscript `<t` means "everything before position t." Log it, get log-likelihood = Σ_t log P(x_t | x_&lt;t, w). Negate, get the token-level cross-entropy — the exact objective GPT and every other decoder-only LLM trains on. So when you hear "GPT is trained to predict the next token," what's really happening is MLE on trillions of tokens.

**3. Diffusion models (image generators).** Diffusion models learn to un-noise images. During training, you take a clean image, add a controlled amount of random noise (call it ε, "epsilon"), and train a network εθ to predict what noise was added. The loss is E[||ε − εθ(noisy image, timestep)||²] — squared error between true and predicted noise. That's the same MSE-as-MLE-under-Gaussian-noise story from earlier: minimizing this loss is MLE, assuming the noise is bell-curve-shaped (which it is, by construction).

Once you can spot MLE in a loss function, three things get easier:

- You can predict properties of the trained model (with enough data, it lands on the truth; error bars shrink as 1/√n).
- You understand *why* the loss looks the way it does (it's the log-probability of the data under some assumed model).
- You know what breaks if the assumption is wrong (heavy tails → don't use MSE; extremely imbalanced classes → cross-entropy might need reweighting).

That's the difference between "here's the loss function the paper handed me" and "here's the probabilistic assumption that loss encodes."

</RealWorldML>

<Summary>

- **Estimator** = a function of the data that guesses a population parameter. Its distribution over resamples is its sampling distribution; bias, variance, and MSE describe its quality.
- **Likelihood** L(θ) = P(D | θ) is a function of θ with data fixed. **Log-likelihood** ℓ(θ) = Σ log p(x_i | θ) is what we actually optimize.
- **Maximum likelihood estimator** θ̂_MLE = argmax_θ ℓ(θ). Take derivatives, set to zero, solve.
- **Bernoulli MLE**: p̂ = k/n (empirical frequency).
- **Gaussian MLE**: μ̂ = sample mean; σ̂² = (1/n) Σ (x_i − μ̂)² (uses N, biased-low — the unbiased estimator uses N − 1).
- **Every ML loss is MLE**: MSE ↔ Gaussian, cross-entropy ↔ Categorical, LM pretraining ↔ chain-rule Categorical.
- **MLE properties**: consistent, asymptotically normal (variance 1/(n · Fisher info)), asymptotically efficient.
- **Traps**: likelihood ≠ probability; zero-density observations kill ℓ; MLE overfits without regularization (see next chapter: MAP + priors).

</Summary>
