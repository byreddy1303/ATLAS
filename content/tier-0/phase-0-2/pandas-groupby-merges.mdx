export const splitCode = `import pandas as pd

df = pd.DataFrame({
    "city":  ["Austin", "Dallas", "Austin", "Houston", "Dallas", "Austin"],
    "sales": [120, 90, 150, 200, 110, 95],
})

# Step 1: SPLIT — groupby returns a *plan* for piles, not the piles themselves
g = df.groupby("city")
print(type(g).__name__)          # DataFrameGroupBy — a lazy object
print()
print(g.groups)                  # {'Austin': [0,2,5], 'Dallas': [1,4], 'Houston': [3]}
print()                          # ^ just row positions per pile — no data moved yet

# you can walk the piles yourself (rarely needed, great for intuition)
for name, pile in g:
    print(f"--- {name} ({len(pile)} rows) ---")
    print(pile)

# Steps 2+3: APPLY a reduction to each pile, COMBINE into one result
print()
print(g["sales"].mean())         # one number per city, indexed by city
#
# The whole pattern in one line:
# df.groupby("city")["sales"].mean()
#        SPLIT         APPLY+COMBINE`;

export const handCode = `import pandas as pd

df = pd.DataFrame({
    "plan":  ["free", "pro", "free", "team", "pro", "free"],
    "spend": [0, 30, 0, 99, 25, 10],
})

# By hand: three piles
#   free -> [0, 0, 10]   sum=10, count=3, mean=3.33
#   pro  -> [30, 25]     sum=55, count=2, mean=27.5

export const verbsCode = `import pandas as pd

df = pd.DataFrame({
    "plan":  ["free", "pro", "free", "team", "pro", "free", "pro"],
    "spend": [0.0, 29.99, 0.0, 99.0, 29.99, 0.0, 24.99],
    "tickets": [3, 1, 0, 2, 0, 5, 1],
})

# ---- VERB 1: agg — one row PER GROUP (table shrinks) --------------------
summary = df.groupby("plan").agg(
    users=("plan", "size"),          # named aggregations:
    spend_mean=("spend", "mean"),    #   new_col=(source_col, how)
    spend_max=("spend", "max"),
    tickets_total=("tickets", "sum"),
)
print(summary)
print()

# ---- VERB 2: transform — same shape as INPUT (broadcast back) -----------
# each row gets ITS OWN group's statistic — perfect for features
df["plan_avg_spend"] = df.groupby("plan")["spend"].transform("mean")
df["spend_vs_plan"]  = df["spend"] - df["plan_avg_spend"]
print(df)
print()

# ---- VERB 3: filter — keep or drop WHOLE groups --------------------------
big_plans = df.groupby("plan").filter(lambda g: len(g) >= 3)
print(big_plans["plan"].unique())    # only plans with 3+ rows survive
#
# VERB 4: apply — arbitrary Python per group. The escape hatch.
# Slow (a real Python loop over groups). Reach for it LAST.`;

export const mergeCode = `import pandas as pd

users = pd.DataFrame({
    "user_id": [1, 2, 3, 4],
    "plan":    ["pro", "free", "team", "pro"],
})
orders = pd.DataFrame({
    "user_id": [1, 1, 3, 5],          # user 5 has no users row; users 2,4 have no orders
    "amount":  [50, 20, 300, 75],
})

# INNER: only keys present on BOTH sides survive
print(pd.merge(users, orders, on="user_id", how="inner"))
print()

# LEFT: every users row survives; missing order data becomes NaN
print(pd.merge(users, orders, on="user_id", how="left"))
print()

# OUTER + indicator: keep everything and LABEL where each row came from
full = pd.merge(users, orders, on="user_id", how="outer", indicator=True)
print(full)
print()
print(full["_merge"].value_counts())   # both / left_only / right_only — the audit trail`;

users = pd.DataFrame({"user_id": [1, 2, 3], "plan": ["pro", "free", "team"]})
profile = pd.DataFrame({"user_id": [1, 2, 2, 3], "age": [30, 25, 26, 41]})

export const pivotCode = `import pandas as pd

# LONG format: one row per observation — the shape computers love
long = pd.DataFrame({
    "city":  ["Austin", "Austin", "Dallas", "Dallas", "Austin", "Dallas"],
    "month": ["Jan", "Feb", "Jan", "Feb", "Jan", "Jan"],
    "sales": [100, 120, 90, 95, 40, 30],
})
print(long)
print()

# pivot_table: LONG -> WIDE. It is a groupby wearing a new layout:
# rows = index groups, columns = column groups, cells = aggfunc of values
wide = long.pivot_table(index="city", columns="month",
                        values="sales", aggfunc="sum", fill_value=0)
print(wide)
print()
# ^ note Austin/Jan had TWO rows (100 and 40) -> aggfunc summed them to 140.
#   pivot_table never errors on duplicates — it AGGREGATES them. That's the
#   difference from .pivot(), which raises on duplicate index/column pairs.

# melt: WIDE -> LONG. The inverse move.
back = wide.reset_index().melt(id_vars="city",
                               var_name="month", value_name="sales")
print(back.sort_values(["city", "month"]).reset_index(drop=True))`;

export const rollingCode = `import pandas as pd
import numpy as np

# 30 days of daily revenue with a weekly rhythm + noise
rng = np.random.default_rng(3)
days = pd.date_range("2026-06-01", periods=30, freq="D")
rev = 100 + 20*np.sin(np.arange(30) * 2*np.pi/7) + rng.normal(0, 8, 30)
s = pd.Series(rev.round(1), index=days, name="revenue")

# resample: groupby over TIME BUCKETS — calendar-aware
print(s.resample("W").sum().round(1))     # weekly totals (weeks end on Sunday)
print()

# rolling: a sliding window ENDING at each row — trailing view of the past
roll = s.rolling(window=7).mean()          # 7-day trailing average
print(pd.DataFrame({"revenue": s, "roll7": roll.round(1)}).head(10))
print()
# ^ first 6 rows are NaN: a 7-wide window isn't full yet.
#   min_periods=1 would emit partial averages instead:
print(s.rolling(7, min_periods=1).mean().round(1).head(3))
print()

# WHY trailing windows matter for ML: at day t you may only use days <= t.
# rolling() looks backwards by construction — leakage-safe by default.
# resample answers "how did each week do?"; rolling answers
# "as of TODAY, what was the recent trend?" — the feature-engineering question.`;

export const realWorldCode = `import pandas as pd
import numpy as np

# ---- Scenario: turn a raw EVENT LOG into per-user features, then join labels ----
events = pd.DataFrame({
    "user_id": [1,1,1, 2,2, 3, 4,4,4,4, 5,5],
    "kind":    ["login","ticket","login", "login","ticket", "login",
                "login","login","ticket","ticket", "login","login"],
    "minutes": [12, 0, 30,  5, 0,  45,  8, 22, 0, 0,  60, 15],
})
users = pd.DataFrame({
    "user_id": [1, 2, 3, 4, 5],
    "plan":    ["pro", "free", "team", "free", "pro"],
    "churned": [0, 1, 0, 1, 0],
})

# 1) SPLIT-APPLY-COMBINE: one feature row per user from the raw log
feats = events.groupby("user_id").agg(
    n_events=("kind", "size"),
    n_tickets=("kind", lambda k: (k == "ticket").sum()),
    total_minutes=("minutes", "sum"),
)
# a cleaner no-lambda version of n_tickets, built BEFORE grouping:
feats["ticket_share"] = (
    events.assign(is_ticket=events["kind"].eq("ticket"))
          .groupby("user_id")["is_ticket"].mean()
)
print(feats)
print()

# 2) MERGE features onto the label table — stating every assumption
model_df = pd.merge(
    users, feats,
    left_on="user_id", right_index=True,   # feats is indexed by user_id
    how="left",                            # every labeled user must survive
    validate="one_to_one",                 # 1 user = 1 feature row, or ERROR
)
assert len(model_df) == len(users)
print(model_df)
print()

# 3) users with NO events would be NaN — make that an explicit zero
count_cols = ["n_events", "n_tickets", "total_minutes", "ticket_share"]
model_df[count_cols] = model_df[count_cols].fillna(0)

# 4) the familiar handoff
X = pd.get_dummies(model_df.drop(columns=["user_id", "churned"]),
                   columns=["plan"], dtype="int64").to_numpy(dtype="float64")
y = model_df["churned"].to_numpy()
print("X:", X.shape, "| y:", y.shape, "| churn rate:", y.mean())`;

<Motivation title="Per-row answers are trivia — per-group answers are insight">

Almost no real question is about one row. Real questions compare **groups**: does the free plan churn more than pro? Which signup month produced the best customers? Every one has the same shape — split rows into piles by a key, compute something per pile, report one line per pile — and pandas compresses the whole pattern into one expression: `df.groupby("plan")["spend"].mean()`. This is **split-apply-combine**, and it's pandas' version of SQL's `GROUP BY`.

The second skill exists because real data arrives as *several* tables — a `users` table, an `events` log, a `payments` table — and answering anything means **merging** rows across them by a shared key. Merging is where quiet catastrophes live: an inner join silently deletes every user who never filed a ticket; a duplicated key silently multiplies rows and doubles revenue in the report. Nothing errors — the numbers are just confidently wrong. So this chapter is two skills braided together: the groupby machine and **join discipline** (`how=`, `validate=`, `indicator=`, row-count asserts).

For an ML engineer the payoff is one sentence: **a feature table is a groupby merged onto a label table.** That sentence is the first mile of every churn model, fraud detector, and eval pipeline (model × prompt × score tables) you will ever build.

</Motivation>

<Intuition title="Sort the mail into piles; weigh each pile; staple the tables by key">

**GroupBy is sorting mail.** Envelopes (rows) land on your desk, each addressed to a city. *Split*: one pile per city — nothing opened yet. *Apply*: weigh each pile (or count, or average). *Combine*: one line per city. `df.groupby("city")` plans the piles (lazily); `.mean()` does apply-and-combine in one breath. There are exactly two output shapes: **`agg` shrinks** (one row per pile — the report shape) and **`transform` keeps the original shape** (every row receives *its own pile's* answer — the feature shape).

**Merge is stapling two tables by a shared key.** For every left row, find right rows with a matching key and staple them side by side. Three questions decide everything: *what counts as a match?* (`on="user_id"`), *what happens to non-matches?* (`how=`: inner drops them, left keeps every left row with NaN padding, outer keeps all), and *how many matches per key?* — the cardinality. One-to-one is a clean staple; one-to-many fans out legitimately (one user, three orders → three rows); many-to-many *multiplies* and is almost always a dirty-key accident.

Keep the two verbs distinct: **merge matches by key; concat just stacks** (same columns, more rows — January atop February, no matching logic).

</Intuition>

## Split → apply → combine

<RunnableCode title="watch the piles form, then collapse to one line per group" code={splitCode} />

`df.groupby("city")` computes **nothing** — it returns a lazy plan (peek at `.groups`: a dict from key to row positions). The `for name, pile in g:` loop makes the piles tangible — walk it once for intuition, then never write it in real work (it's the slow path). The payoff line `g["sales"].mean()` applies the reduction per pile and combines results into one Series whose **index is the group keys** — the column's *values* became the result's *labels*, last chapter's index-as-identity principle resurfacing. Under the hood, built-in reductions run as a single compiled pass using integer group codes (which is also why `category` dtype speeds up groupbys).

<RunnableCode title="three piles you can check on your fingers" code={handCode} />

Hand-check the machine once: free is `[0, 0, 10]` → mean 3.33; pro is `[30, 25]` → 27.5; team's lone row is its own everything. Passing a *list* to `.agg(["count", "sum", "mean"])` computes several statistics in one call, and `reset_index()` flattens the labeled result back to ordinary columns — which you'll want whenever the summary is itself about to be merged or plotted. (Multi-key groupbys return a MultiIndex; `as_index=False` or `reset_index()` is the beginner-safe escape.)

Two pitfalls: **groupby sorts keys alphabetically by default** (`sort=False` preserves appearance order), and **rows with a NaN key are silently dropped from every pile** — your "sales by region" report simply omits unlabeled rows and won't reconcile with `df["sales"].sum()`. Audit `df["region"].isna().sum()` before grouping, or pass `dropna=False`.

## The verbs — agg, transform, filter (and the escape hatch)

<RunnableCode title="the daily groupby vocabulary on one small table" code={verbsCode} />

Choosing a verb is one question — **what shape is the answer?**

- **`agg` — the report verb (shrinks).** One row per group. Use *named aggregation*: `.agg(spend_mean=("spend", "mean"), users=("plan", "size"))` — each keyword names an output column from a `(source_col, how)` pair. Note `"size"` counts rows (NaN included); `"count"` counts non-null values — a classic off-by-some.
- **`transform` — the feature verb (same shape).** Runs the per-pile computation, then broadcasts each pile's answer back onto that pile's rows — so the result aligns row-for-row and can be assigned as a column. `df["spend"] - df.groupby("plan")["spend"].transform("mean")` is "how does this row compare to its group?" — the single most useful feature template in tabular ML (group z-scores, group shares, fill-with-group-mean: all transform). Prefer string names (`"mean"`) over lambdas — strings take the fast compiled path.
- **`filter` — the bouncer.** The lambda sees each pile, returns True/False, and whole groups stay or leave: "only plans with 3+ users" in one line.
- **`apply` — arbitrary Python per pile, any shape, slowly** (it *is* a Python loop over groups). Reach for it only after checking that no `agg`/`transform`/mask combination expresses the idea — nine times out of ten, one does.

One reporting habit that prevents embarrassment: **no rate without its n.** A plan with 2 users showing 99% retention is noise, not insight — always carry a `size` column alongside any per-group rate.

## Merge — state what should happen to non-matches

<RunnableCode title="inner, left, outer — and the audit trail" code={mergeCode} />

Study what each `how=` does to users 2 and 4 (no orders) and to order-owner 5 (no user row). **`inner`** keeps only keys on both sides — 2, 4, 5 all vanish. **`left`** keeps every `users` row, NaN-padding where nothing matched — the default posture when the left table is your population of record. **`outer`** keeps everything and is mostly diagnostic, especially with **`indicator=True`**: the `_merge` column (`both` / `left_only` / `right_only`) plus `.value_counts()` is a one-line reconciliation report between any two tables.

The pitfall that corrupts analyses: **an inner join is a deletion wearing a join's clothes.** Ask "do churners file more tickets?" after inner-joining users to tickets and you've excluded every zero-ticket user — the calmest customers — from both sides of the comparison. Habits: check `len()` before and after every join; when the left table is your population, write `how="left"` and let the NaNs *show you* who didn't match. Two more silent failure modes: a missing key can never match (NaN ≠ NaN), and an int64 `user_id` joined against a string `"101"` matches **nothing** — zero rows, no error. First move when a join returns suspiciously few rows: print both sides' `df["key"].dtype`. (Also: shared non-key column names get renamed `_x`/`_y` behind your back — pass `suffixes=` deliberately.)

## validate= — turning silent corruption into a loud error

<RunnableCode title="the m:m explosion, caught at the door" code={validateCode} />

**`validate="one_to_one"`** (or `"one_to_many"` / `"many_to_one"`) declares the cardinality you believe; pandas raises a `MergeError` the moment reality disagrees — as here, where user 2 has two profile rows. Without it, the merge *succeeds* and quietly duplicates user 2 everywhere downstream. There is no reason to write an important join without it.

The m:m demo shows the stakes: 3 rows × 2 rows sharing a key → 6 rows, every pairing materialized (a join emits the Cartesian product of matches *within each key*). On a million-row table with an accidentally-duplicated key that's an out-of-memory alarm — or a silently doubled revenue number. Belt and suspenders: `assert len(enriched) == len(users)` after any left join.

The last block re-draws the merge/concat line: **`concat` stacks tables sharing columns** — more rows, no matching. Pass `ignore_index=True`, or the frames' 0,1 indexes both survive and you own duplicate labels (last chapter's trap, self-inflicted). And build row-wise data with *one* `pd.concat(list_of_frames)`, never appends in a loop.

## pivot_table and melt — wide for humans, long for machines

<RunnableCode title="pivot_table (long → wide) and melt (wide → long)" code={pivotCode} />

The same data has two shapes. **Long** — one row per observation (`city / month / sales`) — is what machines want: groupby, merge, matplotlib, scikit-learn all eat long. **Wide** — cities down the side, months across the top — is what humans read. **`pivot_table` goes long → wide and is a groupby in a costume**: group by (city, month), aggregate with `aggfunc`, lay the result out as a grid. Proof in the output: Austin/Jan had two rows (100 and 40) → the cell reads 140. That aggregation is the difference from plain `.pivot()`, which does no aggregation and *raises* on duplicate pairs. **`melt` is the inverse** — `id_vars` stays as identity, every other column collapses into variable/value pairs. Spreadsheets from humans arrive wide; your first move is melting them long. The workflow: **melt what humans hand you, compute in long, pivot_table what you hand back.**

## Time is a group — resample buckets, rolling windows

<RunnableCode title="weekly buckets vs 7-day trailing windows on one revenue series" code={rollingCode} />

With a datetime index, two time-aware groupers unlock. **`resample` is groupby over calendar buckets** — `s.resample("W").sum()` piles rows by week and sums each (calendar-aware: `"D"`, `"W"`, `"MS"` all work). One row per bucket: the time version of `agg`, answering *report* questions. **`rolling` is a sliding window ending at each row** — `s.rolling(7).mean()` gives every row the average of itself and the 6 before: same length as input, the time version of `transform`. The first 6 values are NaN (window not full yet); `min_periods=1` emits partial answers instead — an explicit, honest choice.

Why production feature pipelines run on `rolling`: **at prediction time you only possess the past.** This is your first meeting with **leakage** — using future information to predict the past, the cardinal sin of ML features. A trailing window at row *t* contains only days ≤ *t* by construction, so it's structurally leakage-safe (its evil twin `center=True` is not — fine for smoothing plots, forbidden for features). The full production template adds a shift to exclude even today: `df.groupby("user_id")["spend"].transform(lambda s: s.shift(1).rolling(7).mean())` — close to half of every fraud/churn/demand model's feature list. The reflex starts here: **features look backwards.**

<RealWorldML title="Event log → per-user features → labeled training table: the first mile of every tabular model">

<RunnableCode title="groupby the log, merge onto labels, validate, hand off" code={realWorldCode} />

The setting: **facts live at the wrong granularity.** The `events` log has one row per *action*; the `users` table has one row per *user*, including the label (`churned`). A model needs one row per user — so the log must be summarized per user (groupby) and attached to the labels (merge). Line by line:

- `events.groupby("user_id").agg(n_events=("kind","size"), ...)` — split the log into per-user piles, reduce to named features. The lambda `(k == "ticket").sum()` builds a boolean mask inside each pile and sums the Trues — mask-counting, once per group.
- `events.assign(is_ticket=...).groupby(...)["is_ticket"].mean()` — the cleaner, faster version: build the mask column *once* on the whole log, then let the built-in `"mean"` compute the ticket *share* per pile (True=1, False=0, so a mask's mean is the fraction that match). Pre-computing columns so built-ins apply keeps you on the compiled fast path.
- Look at `feats`: **indexed by `user_id`** (the machine's signature), and note who's *missing* — a user with zero events has no rows in the log, hence no pile, hence no feature row at all. Silent absence; the merge must handle it.
- `pd.merge(users, feats, left_on="user_id", right_index=True, how="left", validate="one_to_one")` — every claim stated. Keys: users' column vs feats' *index*. `how="left"`: labeled users are the population of record; a user with no events must **survive** as NaNs, because "never did anything" is signal — an inner join would build a model that has never seen an inactive user, then face them constantly in production. `validate=` raises instead of duplicating if the features ever gain duplicate keys; the `assert` on row count is the belt to its suspenders.
- `model_df[count_cols].fillna(0)` — turn the join's NaNs into the statement they mean: zero events, zero minutes. This is a *modeling decision*, not reflex — for "average session length," 0 would be a lie (they have *no* sessions, not short ones) and a missing-indicator column would be better.
- `pd.get_dummies(...).to_numpy(dtype="float64")` — last chapter's handoff: one-hot the plan, drop identifiers, exit label-world into the `(5, k)` float matrix, `y` alongside.

The punchline to carry forward: **groupby set the granularity, merge attached the truth, validate guarded the door.** Six lines of pandas — very nearly verbatim the first six lines of every tabular ML project you will ship.

<Summary>

- **Split-apply-combine** is the machine: pile rows by key, reduce each pile, label answers with the keys. `groupby(...)` builds a *lazy* plan; the key column's values become the result's *index*.
- **The verbs by output shape:** `agg` → one row per group (reports; use *named aggregation*); `transform` → same shape as input (features: group means, z-scores, shares); `filter` → whole groups in or out; `apply` → anything, slowly — last resort.
- **Merge = match by key, with three loud declarations:** `on=` (the key), `how=` (inner drops non-matches, left keeps the population of record, outer + `indicator=True` audits), `validate=` (cardinality, or a MergeError). Assert row counts after; check key dtypes before; `concat` merely *stacks* (with `ignore_index=True`).
- **The failures are silent:** inner joins delete, duplicate keys multiply (m:m = Cartesian product per key), NaN keys match nothing, and groupby drops NaN keys from piles. The safety kit exists because no error will save you.
- **Wide ↔ long:** `pivot_table` (a groupby printed as a grid — aggregates duplicates, unlike `.pivot()`) for humans; `melt` back to long for machines. Compute long, present wide.
- **Time:** `resample` = groupby over calendar buckets (reports); `rolling` = trailing window per row (features) — backwards-looking by construction, the foundation of leakage-safe features (with `shift(1)` for full hygiene).
- **The sentence that runs applied ML:** groupby the log into features, merge onto labels with `how="left", validate="one_to_one"`, fill NaNs *deliberately*, hand `X, y` to the model.

</Summary>
