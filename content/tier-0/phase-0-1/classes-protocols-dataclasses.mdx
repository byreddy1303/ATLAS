export const dataclassCode = `# @dataclass — Python writes __init__, __repr__, __eq__ for you
from dataclasses import dataclass, field

@dataclass(frozen=True, slots=True)
class Point:
    x: float
    y: float
    label: str = "?"                    # default value

    # regular methods still work
    def dist_to(self, other: "Point") -> float:
        return ((self.x - other.x) ** 2 + (self.y - other.y) ** 2) ** 0.5

a = Point(1, 2, "origin-ish")
b = Point(4, 6)

print(a)                                # Point(x=1, y=2, label='origin-ish')
print(a.dist_to(b))                     # 5.0
print(a == Point(1, 2, "origin-ish"))   # True — __eq__ auto-generated

# frozen=True makes Point immutable AND hashable
locations = {a, b, a}
print(len(locations))                   # 2 — the duplicate 'a' is deduped

# slots=True saves memory AND blocks typos on attribute assignment
try:
    a.x = 999                           # AttributeError — frozen
except Exception as e:
    print("frozen catches:", type(e).__name__)`;

export const protocolCode = `# Protocol — structural typing: 'this walks like a duck'
from typing import Protocol, runtime_checkable

@runtime_checkable
class HasLen(Protocol):
    def __len__(self) -> int: ...       # signature only — no implementation

def print_length(x: HasLen) -> None:
    print(f"length of {type(x).__name__}: {len(x)}")

# Notice: none of these classes INHERIT from HasLen — they just implement __len__
class Team:
    def __init__(self, members):
        self.members = members
    def __len__(self):
        return len(self.members)

print_length([1, 2, 3])                 # length of list: 3
print_length("hello")                   # length of str: 5
print_length(Team(["a", "b"]))          # length of Team: 2

# runtime_checkable enables isinstance()
print(isinstance(Team([]), HasLen))     # True — Team walks like HasLen
print(isinstance(42, HasLen))           # False — int has no __len__`;

export const dunderCode = `# Implement dunders to make your class feel native
class Vector:
    def __init__(self, *xs):
        self.xs = tuple(xs)

    def __repr__(self):
        return f"Vector{self.xs}"

    def __add__(self, other):           # v1 + v2
        return Vector(*(a + b for a, b in zip(self.xs, other.xs)))

    def __mul__(self, k):               # v * k  (right-mul scalar)
        return Vector(*(a * k for a in self.xs))

    def __rmul__(self, k):              # k * v  (scalar on the left)
        return self * k

    def __iter__(self):                 # for x in v
        return iter(self.xs)

    def __len__(self):                  # len(v)
        return len(self.xs)

    def __eq__(self, other):
        return isinstance(other, Vector) and self.xs == other.xs

    def __hash__(self):
        return hash(self.xs)

a = Vector(1, 2, 3)
b = Vector(4, 5, 6)
print(a + b)             # Vector(5, 7, 9)
print(a * 3)             # Vector(3, 6, 9)
print(2 * a)             # Vector(2, 4, 6) — uses __rmul__
print(len(a), list(a))   # 3 [1, 2, 3]
print({a, Vector(1, 2, 3)})   # deduped — thanks to __eq__ and __hash__`;

export const compositionCode = `# Composition beats inheritance for reading, testing, and swapping parts
from dataclasses import dataclass, field
from typing import Protocol
import time

# 1) Define small protocols — what does a 'model' need to do?
class Trainable(Protocol):
    def fit(self, X, y) -> None: ...
    def predict(self, X): ...

class Logger(Protocol):
    def info(self, msg: str) -> None: ...

# 2) A concrete logger
class ConsoleLogger:
    def info(self, msg: str) -> None:
        print(f"[{time.strftime('%H:%M:%S')}] {msg}")

# 3) A model — inherits from NOTHING, HAS-A logger, HAS-A model
class TrainingRun:
    def __init__(self, model: Trainable, logger: Logger):
        self.model = model
        self.logger = logger

    def train(self, X, y):
        self.logger.info(f"training on {len(X)} rows")
        self.model.fit(X, y)
        self.logger.info("done")
        return self.model.predict(X)

# 4) Test-only fake for the model
class ConstantPredictor:
    def __init__(self, val): self.val = val
    def fit(self, X, y): pass
    def predict(self, X): return [self.val] * len(X)

run = TrainingRun(model=ConstantPredictor(0.5), logger=ConsoleLogger())
X = [[1], [2], [3]]
y = [0, 1, 0]
print("preds:", run.train(X, y))`;

export const modelZooCode = `# A tiny 'model zoo' — protocol + composition + dataclass, together
from dataclasses import dataclass, field
from typing import Protocol, runtime_checkable
import statistics

@runtime_checkable
class Regressor(Protocol):
    def fit(self, X, y) -> None: ...
    def predict(self, X): ...

class MeanRegressor:
    """Predicts the mean of the training y for every X."""
    def fit(self, X, y):
        self.mean = statistics.mean(y)
    def predict(self, X):
        return [self.mean] * len(X)

class MedianRegressor:
    """Predicts the median of the training y — more robust to outliers."""
    def fit(self, X, y):
        self.med = statistics.median(y)
    def predict(self, X):
        return [self.med] * len(X)

@dataclass
class Benchmark:
    dataset_name: str
    scores: dict[str, float] = field(default_factory=dict)

    def score(self, model: Regressor, X, y):
        model.fit(X, y)
        preds = model.predict(X)
        mae = statistics.mean(abs(p - t) for p, t in zip(preds, y))
        self.scores[type(model).__name__] = mae

X = [[i] for i in range(10)]
y = [3.0, 3.1, 2.9, 3.0, 3.1, 2.8, 3.2, 3.0, 2.9, 100.0]   # last is an outlier
bench = Benchmark("toy")
bench.score(MeanRegressor(), X, y)
bench.score(MedianRegressor(), X, y)
print(bench)
# MeanRegressor MAE is pulled up by the 100.0 outlier;
# MedianRegressor is essentially untouched by it — a real difference in learning`;

<Motivation title="Modern Python OOP is 3 tools, not 30">

Java-style OOP means inheritance hierarchies, abstract base classes, mixins, and a taxonomy diagram. Do that in Python and your code becomes a maze in six months. The Python community has quietly converged on a much smaller kit:

1. **`@dataclass`** for **value objects** — declare the fields, Python writes `__init__`, `__repr__`, `__eq__`, and (with `frozen=True`) `__hash__` for free.
2. **`Protocol`** for **structural typing** — say what an object should be *shaped like* ("has a `.predict(X)` method"), not what it must inherit from. Duck typing a type checker can verify.
3. **Dunder methods** for making your class cooperate with Python's operators and built-ins — `__add__` gives you `+`, `__iter__` gives you `for`, `__len__` gives you `len()`.

That's 90% of modern Python OOP, tied together by **composition** ("has-a") instead of inheritance ("is-a"). PyTorch, HuggingFace, scikit-learn, LangChain — all built on this minimalism: dataclass configs, protocol-typed strategies, small classes that hold other objects. Learn the kit and you read their source like prose.

</Motivation>

<Intuition title="Value objects vs behavior objects">

Two kinds of classes cover most needs, and knowing which one you're writing makes every design decision easier.

- **Value objects** hold *data*: a `Point`, a `Config`, an `Embedding`. They should be immutable (safe to share and hash), comparable by *value*, and printable. **`@dataclass(frozen=True)` gives you all three for free.**
- **Behavior objects** *do* things: a `DataLoader`, an `Optimizer`, a `TrainingRun`. Their methods mutate state and orchestrate work; their identity matters (two `DataLoader`s configured the same are still different objects). **Write these as plain classes** whose `__init__` receives their dependencies. If a class is 90% methods and 10% state, don't `@dataclass` it.

When in doubt, ask: *would it be weird if two instances with the same fields were "equal"?* Yes → behavior object. No → value object.

</Intuition>

<Recall term="`self`, and why methods take it">

Methods in Python are functions with a first parameter conventionally called `self`. When you call `obj.method(arg)`, Python is *actually* calling `ClassName.method(obj, arg)` — passing the instance in as `self`. Every method is just a function that happens to be defined inside a class body; all the scope and closure rules from the last chapter apply unchanged.

</Recall>

## @dataclass — value objects with no ceremony

<RunnableCode title="frozen + slots, hashability, methods still work" code={dataclassCode} />

Without `@dataclass`, `Point` would take twenty lines of boilerplate. The decorator reads the annotated field list and generates: `__init__` (one parameter per field, defaults honored), `__repr__` (`Point(x=1, y=2, label='origin-ish')` — what makes debugging bearable), and `__eq__` (field-tuple comparison — equality by value). The annotations name the fields; the types themselves are checked by mypy/pyright, not enforced at runtime.

Three switches worth learning:

- **`frozen=True`** — assignment raises, and because state can't change, `__hash__` is auto-generated (non-frozen dataclasses are unhashable by default). The default for config-style objects.
- **`slots=True`** — fixed named slots instead of a per-instance `__dict__`: ~40% memory savings, and typos like `p.xx = 5` raise instead of silently creating an attribute. One limit: no dynamic attributes, and mixing slots with multiple inheritance is subtle.
- **`order=True`** — generates `<`, `<=`, `>`, `>=` by field-tuple comparison, for sortable records.

Need validation or derived fields? `__post_init__` runs right after the generated `__init__` — `if self.lr <= 0: raise ValueError(...)` lives there. Every "config" object and every record you'd otherwise stuff in a `dict[str, Any]` should be a dataclass — HuggingFace's `TrainingArguments` and Pydantic's `BaseModel` are essentially this pattern.

## field() — factories for mutable defaults

<RunnableCode title="field(default_factory=list) and other tricks" code={dataclassFieldsCode} />

The mutable-default trap strikes classes twice. Inside a dataclass, `metrics: list = []` would share one list across every instance — so Python 3.11+ **refuses it with a `ValueError`** at class-creation time. The fix: `field(default_factory=list)` — the generated `__init__` calls the factory once per instance.

The same trap exists in *plain* classes with no guardrail: `class Team: members = []` creates a **class-level attribute** — one list shared by all instances, mutated by all. Per-instance state belongs in `__init__` (`self.members = []`).

Other `field(...)` knobs you'll actually use: `repr=False` (hide noisy fields from `__repr__`), `compare=False` (exclude from `__eq__`/`__hash__`), `init=False` (for fields computed in `__post_init__`), `kw_only=True` (keyword-only in the generated `__init__`).

## Protocol — structural, not nominal, typing

<RunnableCode title="duck typing that a type checker can verify" code={protocolCode} />

<Recall term="Structural vs nominal typing">

**Nominal typing** (Java, C#) asks: "does this object's *declared class* inherit from the required interface?" **Structural typing** (Go, TypeScript, Python's `Protocol`) asks: "does this object have the *methods and attributes* the interface requires?" Structural is friendlier — a third-party class can satisfy your interface without modification — and it's exactly what "duck typing" has always meant. `Protocol` is the machine-checkable version.

</Recall>

This is what makes scikit-learn pipelines accept your custom estimator without inheriting from `sklearn.base.BaseEstimator` — anything with `.fit()` and `.predict()` in the right shape counts. Two ways to use a Protocol:

- **As a type hint**: `def print_length(x: HasLen):` — mypy/pyright accept anything with `__len__`. Static-only, zero runtime cost.
- **As a runtime check**: decorate with `@runtime_checkable`, and `isinstance(obj, HasLen)` inspects the object's methods. Coarser (checks *presence*, not signatures) but useful for dispatch and clear error messages.

**The rule:** `Protocol` for open contracts anyone can satisfy; `ABC` (abstract base class, enforcing inheritance and refusing to instantiate incomplete subclasses) for closed hierarchies where you own every implementation. Modern Python leans Protocol.

## Dunder methods — make your class feel native

<RunnableCode title="__add__, __mul__, __rmul__, __iter__, __len__, __eq__, __hash__" code={dunderCode} />

The pattern repeats endlessly: **pick a piece of syntax you want, look up its dunder, add a method with that name.** `+` → `__add__` (and `__rmul__`/`__radd__` for when your object is on the *right*), `for x in obj` → `__iter__`, `len(obj)` → `__len__`, `obj[k]` → `__getitem__`, `with obj:` → `__enter__`/`__exit__`, `obj()` → `__call__` (which is how `model(batch)` runs a PyTorch forward pass). That's *all* of operator overloading — no declarations, just method names.

Two rules when hand-writing dunders:

- **`__eq__` and `__hash__` must agree**: if `a == b`, then `hash(a) == hash(b)`. Override `__eq__` alone and Python disables hashing on your class. `@dataclass(frozen=True)` handles this; by hand, you must remember (`__hash__ = hash(self.xs)`-style, from the same fields `__eq__` compares).
- **`__repr__` is for developers** (unambiguous, ideally a reconstructor string like `Vector(1, 2, 3)`); `__str__` is for `print()` and may be prettier. Define `__repr__` first — `str()` falls back to it.

## Composition over inheritance

<RunnableCode title="a class that owns its dependencies, not one that inherits their behaviors" code={compositionCode} />

The biggest design taste in modern Python: instead of `class GPUSampler(Sampler, GPUMixin, LoggingMixin)` — three parents, method resolution by C3 linearization, behaviors bleeding across the hierarchy — write a class that *has-a* logger and *has-a* model. Three wins:

1. **Readability.** `run.logger.info(...)` says where logging goes; `self.info(...)` from a mixin makes you trace the MRO.
2. **Testability.** Pass fakes in: `TrainingRun(model=ConstantPredictor(0.5), logger=FakeLogger())` — fully testable, no mocking framework.
3. **Flexibility.** Reconfigure at runtime (`run.logger = FileLogger(...)`); inheritance is fixed at class-creation time.

This is how real ML libraries are shaped: `Adam(model.parameters(), lr=1e-3)` composes parameters with an optimizer rather than inheriting from Model; `Trainer(model=..., args=..., train_dataset=...)` is pure composition. Inheritance isn't evil — it's right for true *is-a* relationships (`SGDMomentum(SGD)`) — but if you're inheriting just to share code, a composed object or a plain function is cleaner.

## Putting it together — a benchmark harness

<RunnableCode title="a benchmark harness — protocol-typed strategy, composed by field" code={modelZooCode} />

All three tools in one design: `Regressor` is a **Protocol** (behavioral contract, no inheritance); `MeanRegressor`/`MedianRegressor` implement it without declaring anything; `Benchmark` is a **dataclass** holding a name and a `field(default_factory=dict)` of scores; `bench.score(model, X, y)` accepts anything protocol-compatible — including a real `sklearn.linear_model.LinearRegression`, unchanged.

The output also teaches a real ML lesson: the mean regressor's error is dragged up by the `100.0` outlier; the median regressor barely notices. Robust statistics, demonstrated by a design pattern.

<RealWorldML title="A model registry with protocol dispatch">

Every serious ML platform has a **model registry**: many kinds of models (sklearn, XGBoost, PyTorch — wildly different classes), loaded by name, served through one uniform surface: `predict(X) -> y`. Protocol + dataclass + composition solve it cleanly.

<RunnableCode title="a tiny model registry using Protocol dispatch" code={`from dataclasses import dataclass, field
from typing import Protocol, runtime_checkable

# 1) The contract every 'model' in our registry must satisfy
@runtime_checkable
class Predictor(Protocol):
    def predict(self, X) -> list: ...

# 2) A dataclass that holds the registry's state
@dataclass
class ModelRegistry:
    _models: dict[str, Predictor] = field(default_factory=dict)

    def register(self, name: str, model: Predictor):
        if not isinstance(model, Predictor):
            raise TypeError(f"{type(model).__name__} does not implement Predictor")
        self._models[name] = model

    def predict(self, name: str, X):
        return self._models[name].predict(X)

    def list_models(self) -> list[str]:
        return list(self._models)

# 3) Any class satisfying Predictor can be registered — no inheritance
class MeanModel:
    def __init__(self, val): self.val = val
    def predict(self, X): return [self.val] * len(X)

class FirstFeatureModel:
    def predict(self, X): return [row[0] for row in X]

reg = ModelRegistry()
reg.register("mean-3", MeanModel(3.0))
reg.register("first", FirstFeatureModel())
print("registered:", reg.list_models())
print("mean-3:", reg.predict("mean-3", [[1,2], [3,4]]))
print("first :", reg.predict("first",  [[1,2], [3,4]]))`} />

**Line by line:**

1. `@runtime_checkable class Predictor(Protocol)` declares the smallest possible contract: anything with `.predict(X)` counts.
2. `@dataclass class ModelRegistry` — free `__init__`/`__repr__`, and `field(default_factory=dict)` dodges the shared-mutable-default bug.
3. `register(...)` uses `isinstance(model, Predictor)` to fail fast with a clear error instead of an `AttributeError` deep inside predict-time.
4. `MeanModel` and `FirstFeatureModel` share no ancestry — both work because both have `.predict`. A real sklearn model drops in identically.

MLflow's registry, HuggingFace's `pipeline("...", model="...")`, and Ray Serve deployments are this exact shape: protocol contract, dict of named models, dispatch by name.

</RealWorldML>

<Summary>

- **`@dataclass(frozen=True, slots=True)`** is the default for value objects — free `__init__`, `__repr__`, `__eq__`, `__hash__`; slots save memory and catch typos; `__post_init__` for validation.
- **Mutable defaults need `field(default_factory=list)`** — a literal `[]` is shared (and rejected by 3.11+). In plain classes, `class C: items = []` is the same bug with no guardrail: class-level = shared by every instance.
- **`Protocol` = structural typing**: describe what an object should *do*, not what it inherits. `@runtime_checkable` enables `isinstance` dispatch; `ABC` only for closed hierarchies you fully own.
- **Dunders wire your class into the language**: `__add__`/`__rmul__`, `__iter__`, `__len__`, `__getitem__`, `__call__`. Keep `__eq__` and `__hash__` consistent or sets break.
- **Composition beats inheritance**: pass collaborators into `__init__`, hold them as fields — easier to read, test with fakes, and swap at runtime.
- **In ML**: configs = dataclasses, model interfaces = protocols, trainers/optimizers = composition. Read PyTorch and HuggingFace with this lens and the source opens up.

</Summary>
