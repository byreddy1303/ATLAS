export const trainLogRegCode = `# Train logistic regression from scratch via gradient descent on the log-loss.
# Same math as sklearn.LogisticRegression -- reduced to bare numpy.
import numpy as np, matplotlib.pyplot as plt

np.random.seed(0)

# Two-cluster 2D classification -- linearly separable with a small overlap
N_per = 150
X_pos = np.random.randn(N_per, 2) + np.array([+1.8, +1.5])
X_neg = np.random.randn(N_per, 2) + np.array([-1.5, -1.2])
X = np.vstack([X_pos, X_neg])
X = np.hstack([X, np.ones((2 * N_per, 1))])                # bias column
y = np.concatenate([np.ones(N_per), np.zeros(N_per)])

def sigmoid(z): return 1 / (1 + np.exp(-z))

def log_loss(w, X, y, eps=1e-12):
    "Binary cross-entropy on the whole dataset."
    p = sigmoid(X @ w)
    return -np.mean(y * np.log(p + eps) + (1 - y) * np.log(1 - p + eps))

def grad_log_loss(w, X, y):
    "d/dw of log-loss is X.T @ (p - y) / N -- the SAME shape as linear regression's gradient."
    p = sigmoid(X @ w)
    return X.T @ (p - y) / len(y)

# Gradient descent
w = np.zeros(3)
lr = 0.5
losses = []
for step in range(500):
    losses.append(log_loss(w, X, y))
    w = w - lr * grad_log_loss(w, X, y)

print(f"Final loss: {log_loss(w, X, y):.4f}")
print(f"Fitted weights: {w.round(3)}")
preds = (sigmoid(X @ w) > 0.5).astype(int)
print(f"Training accuracy: {(preds == y).mean():.4f}")

# Plot 1: decision boundary
fig, axes = plt.subplots(1, 2, figsize=(11, 4.4))
xx, yy = np.meshgrid(np.linspace(-5, 5, 200), np.linspace(-5, 5, 200))
grid = np.hstack([xx.reshape(-1, 1), yy.reshape(-1, 1), np.ones((xx.size, 1))])
probs = sigmoid(grid @ w).reshape(xx.shape)
axes[0].contourf(xx, yy, probs, levels=20, cmap="RdBu", alpha=0.6)
axes[0].contour(xx, yy, probs, levels=[0.5], colors="k", linewidths=2)
axes[0].plot(X_pos[:, 0], X_pos[:, 1], "o", color="crimson", markersize=4, alpha=0.6)
axes[0].plot(X_neg[:, 0], X_neg[:, 1], "s", color="royalblue", markersize=4, alpha=0.6)
axes[0].set_title("Logistic regression decision boundary")
axes[0].set_xlabel("x_1"); axes[0].set_ylabel("x_2"); axes[0].grid(alpha=0.3)

# Plot 2: loss curve
axes[1].semilogy(losses, color="crimson", lw=2)
axes[1].set_xlabel("GD step"); axes[1].set_ylabel("log-loss (log scale)")
axes[1].set_title("Training loss decreases monotonically (convex loss)")
axes[1].grid(alpha=0.3)
plt.tight_layout()`;

export const mleDerivationCode = `# Verify: minimizing log-loss = maximum likelihood under Bernoulli.
# Given y_i ~ Bernoulli(p_i) where p_i = sigmoid(w · x_i), the likelihood is
#     L(w) = prod_i p_i^{y_i} (1 - p_i)^{1 - y_i}
# Take log and negate to get NLL = - sum log likelihood terms = log-loss.
import numpy as np

np.random.seed(0)
N = 100
X = np.random.randn(N, 2)
X = np.hstack([X, np.ones((N, 1))])
w_true = np.array([1.5, -1.2, 0.4])
p_true = 1 / (1 + np.exp(-X @ w_true))
y = (np.random.rand(N) < p_true).astype(int)      # sample Bernoulli(p_true)

def log_lik(w, X, y, eps=1e-12):
    p = 1 / (1 + np.exp(-X @ w))
    return np.sum(y * np.log(p + eps) + (1 - y) * np.log(1 - p + eps))

# Grid over w0 (keeping w1, w2 at their true values) to see the likelihood surface
w0_range = np.linspace(-1, 4, 100)
lls = []
for w0 in w0_range:
    w = np.array([w0, w_true[1], w_true[2]])
    lls.append(log_lik(w, X, y))
lls = np.array(lls)

print(f"True w0 = {w_true[0]}")
print(f"Grid-search argmax (w0 that maximizes likelihood): {w0_range[np.argmax(lls)]:.3f}")
print()
print("The likelihood peaks near the true w0. Minimizing log-loss = maximizing")
print("log-likelihood = maximum likelihood estimation. It's the same objective")
print("stated three ways, which is why they always agree on the optimal w*.")`;

export const softmaxCode = `# Multiclass logistic regression = softmax regression.
# Each class gets its own weight vector w_k; probabilities are given by softmax.
import numpy as np, matplotlib.pyplot as plt

def softmax(Z):
    "Row-wise softmax with numerically-stable form (subtract max for stability)."
    Z_shift = Z - Z.max(axis=1, keepdims=True)
    exp = np.exp(Z_shift)
    return exp / exp.sum(axis=1, keepdims=True)

# 3-class synthetic 2D data
np.random.seed(0)
N_per = 100
centers = [np.array([2, 2]), np.array([-2, 2]), np.array([0, -2])]
X_list, y_list = [], []
for c, center in enumerate(centers):
    X_list.append(np.random.randn(N_per, 2) + center)
    y_list.append(np.full(N_per, c))
X = np.vstack(X_list)
X = np.hstack([X, np.ones((3 * N_per, 1))])
y = np.concatenate(y_list)
K = 3    # 3 classes

# One-hot the labels for the loss computation
Y_onehot = np.eye(K)[y]

def train_softmax(X, Y, lr=0.5, steps=200):
    W = np.zeros((X.shape[1], K))
    for _ in range(steps):
        Z = X @ W
        P = softmax(Z)
        grad = X.T @ (P - Y) / len(X)
        W = W - lr * grad
    return W

W = train_softmax(X, Y_onehot)
Z = X @ W
P = softmax(Z)
preds = P.argmax(axis=1)
print(f"Training accuracy: {(preds == y).mean():.4f}")
print(f"Class-3 weight vectors (each column):\\n{W.round(3)}")

# Plot decision regions
fig, ax = plt.subplots(figsize=(6.5, 5.5))
xx, yy = np.meshgrid(np.linspace(-5, 5, 200), np.linspace(-5, 5, 200))
grid = np.hstack([xx.reshape(-1, 1), yy.reshape(-1, 1), np.ones((xx.size, 1))])
grid_preds = softmax(grid @ W).argmax(axis=1).reshape(xx.shape)
ax.contourf(xx, yy, grid_preds, levels=[-0.5, 0.5, 1.5, 2.5], cmap="RdBu", alpha=0.4)
colors = ["crimson", "royalblue", "seagreen"]
for c in range(K):
    ax.plot(X[y == c, 0], X[y == c, 1], "o", color=colors[c], markersize=5, alpha=0.7, label=f"class {c}")
ax.set_title("Softmax regression — 3-class decision regions")
ax.set_xlabel("x_1"); ax.set_ylabel("x_2"); ax.legend(); ax.grid(alpha=0.3)
plt.tight_layout()`;

<Motivation title="Logistic regression is the bridge from regression to classification — and to neural nets">

Take linear regression, but instead of predicting a continuous number, predict a **class label** (spam or ham, positive or negative review, click or no click). The natural first attempt — "fit a linear regression to 0/1 labels" — doesn't work well: predictions escape the [0, 1] range, the loss surface is bad-shaped, and there's no principled way to threshold.

**Logistic regression fixes this** by passing the linear score through the **sigmoid** function `σ(z) = 1/(1+exp(-z))`, which squishes any real number into a probability. The training loss becomes **binary cross-entropy** (aka log-loss), which is convex, has a clean gradient, and corresponds exactly to maximum-likelihood estimation under a Bernoulli model.

For a GenAI engineer, logistic regression is important because:

1. **It's the last layer of every classifier.** Every softmax classifier — from ResNet to a transformer's next-token head — is logistic regression on top of learned features. Understanding LR gives you the mental model for every model output that's a distribution over classes.
2. **The log-loss you'll optimize is EVERYWHERE.** Cross-entropy loss for classification, language modeling, RLHF reward modeling — all of them are logistic regression's log-loss, generalized. Understanding it well here means you understand it everywhere.
3. **The softmax + cross-entropy pair is one of the most important formulas in DL.** Understanding why softmax is the "differentiable argmax" and why cross-entropy has clean gradients (`p - y`) unlocks the mechanics of how classification networks train.
4. **Linear probing.** Same trick as linear regression: fit LR on top of frozen pretrained embeddings to evaluate representation quality. `sklearn.LogisticRegression` on top of a CLIP embedding IS how a lot of vision-language transfer research works.

</Motivation>

<Intuition title="Fit a linear score, squish it through sigmoid to get a probability, threshold at 0.5">

Given features `x`, compute a **linear score** `z = w · x + b`. Interpret z as "how much evidence do we have for class 1?" — positive z favors class 1, negative z favors class 0, and z near zero means "on the fence."

Now feed z through the **sigmoid** function to get a probability:

$$
p = \sigma(z) = \frac{1}{1 + e^{-z}}
$$

`σ(z) → 0` as z → -∞, `σ(z) → 1` as z → +∞, and `σ(0) = 0.5`. The output is a valid probability, and a threshold of 0.5 (equivalent to z > 0) gives the discrete prediction.

**The training objective** picks weights that make each true label as likely as possible under this probabilistic model. If the true class is 1 and we predict p, the log-likelihood contribution is log(p). If the true class is 0, it's log(1 - p). Summed over the dataset and negated, this becomes the **log-loss** (or **binary cross-entropy**):

$$
\mathcal{L}(w) = -\frac{1}{N} \sum_i \left[ y_i \log p_i + (1 - y_i) \log (1 - p_i) \right]
$$

Optimize with gradient descent (loss is convex, so any GD converges). Softmax + cross-entropy generalizes the same recipe to multi-class classification.

</Intuition>

<Recall term="Sigmoid function">

The sigmoid is `σ(z) = 1 / (1 + exp(-z))`. Properties: (1) maps `R → (0, 1)`, so its output can be interpreted as a probability. (2) monotonically increasing. (3) `σ(0) = 0.5`. (4) derivative `σ'(z) = σ(z)(1 - σ(z))`, which is easy to compute given σ(z) itself — this is why sigmoids were popular in early neural nets. (5) saturates at both ends (`σ'(±10) ≈ 0`), which is why deep networks with sigmoid activations had vanishing-gradient problems — ReLU replaced sigmoid inside deep nets, but sigmoid is still the standard output for binary classification.

</Recall>

## The model and the sigmoid

**Linear score:** for features `x ∈ R^D` with weights `w ∈ R^D` and bias `b`:

$$
z = w \cdot x + b
$$

**Probability via sigmoid:**

$$
\hat{p} = \sigma(z) = \frac{1}{1 + e^{-z}}
$$

**Prediction:** if `p̂ ≥ 0.5`, predict class 1; else class 0. This is equivalent to `z ≥ 0`, so the **decision boundary** is the hyperplane `w · x + b = 0` — the same as for the perceptron, the same as for SVMs. Everything logistic regression does downstream of picking that hyperplane is *probabilistic reasoning*, not a fundamentally different geometry.

**The bias trick:** absorb b into w by appending a column of 1s to X — same as in linear regression.

## The log-loss and its MLE origin

Given a dataset of `(x_i, y_i)` with `y_i ∈ {0, 1}`, we model each label as an **independent Bernoulli trial**: `y_i ~ Bernoulli(p_i)` where `p_i = σ(w · x_i)`. Likelihood:

$$
L(w) = \prod_i p_i^{y_i} (1 - p_i)^{1 - y_i}
$$

The maximum-likelihood estimator picks w to maximize this. Taking log and negating gives the equivalent minimization:

$$
\mathcal{L}(w) = -\frac{1}{N} \sum_i \left[ y_i \log p_i + (1 - y_i) \log (1 - p_i) \right]
$$

**This is the binary cross-entropy** between the true (one-hot) label and the predicted probability. Two names for the same object: "log-loss" (ML tradition) and "binary cross-entropy" (information theory).

**Convexity.** The log-loss is convex in w. Proof: the Hessian is `X^T · D · X` where `D` is a diagonal matrix with entries `p_i (1 - p_i) > 0`. This is a positive semi-definite matrix, so the loss is convex. Convex ⇒ every local minimum is global ⇒ gradient descent from any starting point converges to `w*`.

**Gradient — clean and identical in shape to linear regression:**

$$
\nabla_w \mathcal{L}(w) = \frac{1}{N} X^\top (\sigma(Xw) - y) = \frac{1}{N} X^\top (\hat{p} - y)
$$

This gradient has the same form as linear regression's gradient! `X^T · (predicted − actual)`. The only difference is that "predicted" here is σ(Xw), not Xw. This is a deep pattern: **for most classification losses paired with the "right" output activation, the gradient is X^T · (p - y).**

## Multiclass — softmax regression

For K > 2 classes, generalize sigmoid to the **softmax function**. Each class gets its own weight vector `w_k ∈ R^D`, and the probability of class k is:

$$
p_k = \mathrm{softmax}(z)_k = \frac{e^{z_k}}{\sum_j e^{z_j}} \quad \text{where } z_k = w_k \cdot x
$$

Properties: `p_k > 0`, `Σ p_k = 1`. Multiclass log-loss (categorical cross-entropy):

$$
\mathcal{L}(W) = -\frac{1}{N} \sum_i \log p_{y_i}(x_i)
$$

For a one-hot encoded label vector `y ∈ R^K`, this is `-Σ y_k log p_k`. Same "expected surprise" story as binary cross-entropy.

**Gradient is stunningly clean:** for the categorical cross-entropy with softmax activation,

$$
\frac{\partial \mathcal{L}}{\partial z_k} = p_k - y_k
$$

That is, the gradient with respect to the logits is just `(predicted probability - one-hot label)` — the same `p - y` pattern. This is why the softmax + cross-entropy pair is the standard classifier output: the gradient is trivial to compute and numerically well-behaved.

**Numerical stability:** never compute softmax naively. Subtract the max first:

```python
def softmax(z):
    z = z - z.max(axis=-1, keepdims=True)
    e = np.exp(z)
    return e / e.sum(axis=-1, keepdims=True)
```

Otherwise `exp(z_k)` overflows to `inf` for `z_k > 700` in float64. The max-subtraction is mathematically equivalent (cancels in the ratio) but keeps every exp argument non-positive.

## Decision boundary geometry

The decision boundary of binary LR is the hyperplane `w · x + b = 0`. On one side of this plane, `z > 0`, so σ(z) > 0.5, so predict class 1. On the other side, predict class 0. **Distance from x to the boundary** is `|w · x + b| / ‖w‖`.

For multiclass softmax regression, the boundary between class j and class k is the hyperplane `(w_j − w_k) · x + (b_j − b_k) = 0` — the set of points where the two classes have equal probability. K classes give K·(K−1)/2 pairwise boundaries; each point in feature space falls into whichever class has the highest logit.

**Practical implications:**

- **LR only draws linear boundaries** in feature space. Nonlinear boundaries require nonlinear feature engineering (polynomial features, kernel trick) or a nonlinear model.
- **LR's decision boundary is a global property** — it depends on all training data, not just "boundary points" like SVM does.
- **Adding features shifts and reshapes the boundary.** Adding `x_1²` and `x_2²` as features lets LR draw ellipses. Adding `x_1 · x_2` lets it draw hyperbolas. This is the same feature-engineering idea as polynomial regression.

<RealWorldML title="Logistic regression is the last layer of every classification network">

The final layer of every neural-net classifier — from ResNet to CLIP's text head to a transformer's next-token logit head — is a **linear layer followed by softmax**. That's exactly softmax regression on the "features" (embeddings) produced by the encoder.

- **Vision:** ResNet encodes an image into a 2048-dim embedding; then a `W ∈ R^{2048 × 1000}` linear layer + softmax over 1000 ImageNet classes.
- **NLP:** A transformer encodes a token into a d_model-dim vector; then a `W ∈ R^{d_model × V}` (V = vocab size) linear layer + softmax gives next-token probabilities.
- **CLIP-style multimodal:** encode image → image embedding; encode text → text embedding; the "classifier" is just cosine similarity, but that IS a linear-in-embeddings score followed by softmax over candidate texts.

**Why does this matter for a GenAI engineer?**

1. **Linear probing** (evaluating pretrained embeddings) really means: freeze the encoder, replace the final softmax head with a NEW softmax head, retrain just that head on a labeled task. If it fits well, the embeddings are good for that task.
2. **Fine-tuning is 'linear probing PLUS updating the encoder.'** The gradient of the classification loss flows back through the softmax head, into the encoder, and updates its parameters. Every DL paper about "fine-tuning ImageNet" or "SFT for LLMs" is this.
3. **Calibration.** The raw softmax outputs of a big classifier are often mis-calibrated — the model reports 90% confidence for events that happen 60% of the time. Techniques (temperature scaling, Platt scaling, isotonic regression) tweak the last logistic layer's output to fix this. Understanding logistic regression well is the foundation for calibration work.
4. **Softmax's gradient p - y is the "backward pass" of every classification network.** Understanding this scalar identity here means you understand the last layer's backward pass in every model you'll ever train.

</RealWorldML>
