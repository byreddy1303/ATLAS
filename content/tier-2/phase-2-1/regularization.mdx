# See how L2 tames the wild oscillations.
import numpy as np, matplotlib.pyplot as plt

np.random.seed(0)
N_train = 15
x_train = np.linspace(-3, 3, N_train)

# Polynomial feature matrix
def poly_features(x, degree):
    return np.stack([x**k for k in range(degree + 1)], axis=1)

X_test  = poly_features(x_test,  degree)

export const l1VsL2Code = `# L1 vs L2: same optimization framework, wildly different behavior.
# L2 shrinks all weights toward zero; L1 sets many weights exactly to zero.
import numpy as np, matplotlib.pyplot as plt

np.random.seed(0)

# Synthetic problem with a mix of important and irrelevant features
N, D = 100, 20
X = np.random.randn(N, D)
w_true = np.zeros(D)
w_true[:5] = np.array([2.0, -1.5, 3.0, 0.5, -2.0])       # only first 5 features matter
y = X @ w_true + 0.3 * np.random.randn(N)

def ridge(X, y, lam):
    D = X.shape[1]
    return np.linalg.solve(X.T @ X + lam * np.eye(D), X.T @ y)

def soft_threshold(w, thresh):
    "Soft-threshold operator: pull each entry toward 0 by 'thresh', clip at 0."
    return np.sign(w) * np.maximum(np.abs(w) - thresh, 0)

def lasso_cd(X, y, lam, steps=500):
    "Lasso via coordinate descent — the standard L1 solver."
    N, D = X.shape
    w = np.zeros(D)
    xj_sq = np.sum(X ** 2, axis=0)
    for _ in range(steps):
        for j in range(D):
            residual = y - X @ w + X[:, j] * w[j]
            rho = X[:, j] @ residual
            w[j] = soft_threshold(rho / xj_sq[j], lam * N / xj_sq[j])
    return w

# Fit both at similar effective regularization
w_ridge = ridge(X, y, lam=5.0)
w_lasso = lasso_cd(X, y, lam=0.1)

print("True weights (only first 5 nonzero):")
print(w_true.round(2))
print()
print("Ridge fit — shrinks ALL weights toward 0 but leaves them nonzero:")
print(w_ridge.round(2))
print(f"  nonzero-count in ridge: {(np.abs(w_ridge) > 0.01).sum()} / {D}")
print()
print("Lasso fit — sets many weights EXACTLY to zero:")
print(w_lasso.round(2))
print(f"  nonzero-count in lasso: {(np.abs(w_lasso) > 0.01).sum()} / {D}")

# Visualize the difference
fig, axes = plt.subplots(1, 2, figsize=(11, 3.5))
axes[0].bar(range(D), w_ridge, color="teal");   axes[0].set_title(f"Ridge (L2): {(np.abs(w_ridge) > 0.01).sum()} nonzero")
axes[0].axhline(0, color="k", lw=0.5); axes[0].grid(alpha=0.3)
axes[1].bar(range(D), w_lasso, color="crimson"); axes[1].set_title(f"Lasso (L1): {(np.abs(w_lasso) > 0.01).sum()} nonzero")
axes[1].axhline(0, color="k", lw=0.5); axes[1].grid(alpha=0.3)
for ax in axes: ax.set_xlabel("feature index"); ax.set_ylabel("coefficient")
plt.tight_layout()`;

export const cvLambdaCode = `# Choose lambda via K-fold cross-validation.
# For each lambda, average validation error over K folds; pick the smallest.
import numpy as np

np.random.seed(0)
N, D = 200, 30
X = np.random.randn(N, D)
w_true = np.zeros(D)
w_true[:8] = np.random.randn(8) * 2
y = X @ w_true + 0.5 * np.random.randn(N)

def ridge(X, y, lam):
    return np.linalg.solve(X.T @ X + lam * np.eye(X.shape[1]), X.T @ y)

def kfold_ridge_cv(X, y, lam, K=5, seed=1):
    rng = np.random.default_rng(seed)
    N = len(y)
    idx = rng.permutation(N)
    fold_sizes = np.array_split(idx, K)
    errors = []
    for k in range(K):
        val_idx = fold_sizes[k]
        train_idx = np.concatenate([f for i, f in enumerate(fold_sizes) if i != k])
        w = ridge(X[train_idx], y[train_idx], lam)
        err = np.mean((X[val_idx] @ w - y[val_idx]) ** 2)
        errors.append(err)
    return np.mean(errors), np.std(errors)

lams = np.logspace(-3, 3, 30)
print(f"{'lambda':>8}  {'CV MSE':>10}  {'std':>10}")
print("-" * 32)
best_lam, best_mse = None, np.inf
for lam in lams:
    mse, sd = kfold_ridge_cv(X, y, lam)
    if mse < best_mse:
        best_lam, best_mse = lam, mse
    if lam in [1e-3, 1e-2, 1e-1, 1.0, 10.0, 100.0, 1000.0]:
        print(f"{lam:>8.3f}  {mse:>10.4f}  {sd:>10.4f}")
print()
print(f"Best lambda: {best_lam:.4f}   with CV MSE {best_mse:.4f}")
print()
print("This is exactly what sklearn's RidgeCV does under the hood.")
print("Standard rule: pick the lambda that minimizes CV error, or")
print("the 'one-standard-error' rule: pick the LARGEST lambda whose CV MSE")

<Motivation title="Regularization is the difference between a model that memorizes and one that generalizes">

Every supervised model has a fundamental tension: fit the training data (low training loss) vs. generalize to new data (low test loss). Without any constraint, a flexible model (high polynomial degree, deep network, large tree) can achieve near-zero training loss by memorizing every noisy detail — and then perform terribly on new examples. This is **overfitting**.

**Regularization** is the toolkit for fighting overfitting: add a penalty on model complexity to the loss, so training rewards simple models unless the data really demands complexity. It's the single most widely-applicable technique in classical ML.

For a GenAI engineer, regularization matters everywhere:

1. **Every transformer training config includes `weight_decay=0.1` — that's L2 regularization.** Understanding what L2 does (Bayesian: puts a Gaussian prior on weights; geometric: shrinks weights toward zero) is essential for reading and modifying training configs.
2. **LoRA is implicitly regularizing.** Constraining fine-tuning to a low-rank update is a form of complexity control — you literally forbid full-rank updates, which is regularization by architectural choice.
3. **RLHF's KL penalty is regularization on the policy.** It penalizes moving too far from the reference policy — analogous to L2 on the KL divergence between current and reference distributions.
4. **Sparse fine-tuning (BitFit, prompt tuning) uses L1-style sparsity.** These methods work because most of the parameters in an LLM don't need to change for most downstream tasks — sparsity constraints let the fine-tune focus on the few that do.
5. **Model compression (pruning, quantization).** Trimming weights to zero (pruning) or quantizing to low precision are aggressive regularizations that maintain accuracy while shrinking model size.

This chapter covers the three classical regularizers — L2 (ridge), L1 (lasso), elastic net — and how to pick the regularization strength via cross-validation.

</Motivation>

<Intuition title="Penalize weights for being big — L2 penalizes squared magnitude, L1 penalizes absolute magnitude">

The vanilla loss for linear regression is `L(w) = ½ ‖Xw − y‖²`. Add a **regularization term** that grows with the size of w:

$$
L_{\text{reg}}(w) = \frac{1}{2} \|Xw - y\|^2 + \lambda \cdot \text{penalty}(w)
$$

where λ > 0 controls how much you care about weight-size versus fit-quality. Two common choices:

- **L2 (ridge):** `penalty(w) = ‖w‖² = Σ w_i²`. Punishes big weights quadratically. Shrinks all weights uniformly toward zero (but never exactly to zero unless the data is perfectly noiseless).
- **L1 (lasso):** `penalty(w) = ‖w‖₁ = Σ |w_i|`. Punishes big weights linearly. Has a **kink at zero** that produces **sparse** solutions — many weights get set exactly to zero. This gives implicit feature selection: features whose coefficient is zero are effectively ignored.

**Geometric picture.** The regularized optimum lies inside a "ball" whose radius shrinks as λ grows. For L2, the ball is round (Euclidean ball); for L1, the ball is a diamond (with corners on the axes). The corners of the L1 diamond touch the coordinate axes — that's why L1 tends to push weights onto axes (i.e., to zero).

**λ (lambda) is the regularization strength.** Small λ = weak regularization = model close to unregularized fit. Large λ = strong regularization = weights pushed to zero. λ is a hyperparameter chosen by cross-validation.

</Intuition>

<Recall term="MSE loss">

Mean squared error `L(w) = (1/N) · Σ (ŷ_i − y_i)² = (1/N) · ‖Xw − y‖²`. This is the standard linear-regression loss, differentiable everywhere, corresponding to MLE under Gaussian noise. All regularizers in this chapter modify MSE by ADDING a penalty term — the base loss stays the same.

</Recall>

## L2 (ridge regression)

**Penalized loss:**

$$
L_{\text{ridge}}(w) = \frac{1}{2} \|Xw - y\|^2 + \frac{\lambda}{2} \|w\|^2
$$

**Gradient:**

$$
\nabla L_{\text{ridge}}(w) = X^\top (Xw - y) + \lambda w
$$

**Closed-form solution:** set gradient to zero:

$$
(X^\top X + \lambda I) w = X^\top y \quad \Longrightarrow \quad w^*_{\text{ridge}} = (X^\top X + \lambda I)^{-1} X^\top y
$$

**Two consequences of adding `λI`:**

1. **Numerical stability.** Even if `X^T X` is nearly singular (multicollinearity), `X^T X + λI` is well-conditioned because we've added λ to every eigenvalue. Ridge is the standard fix for multicollinearity.
2. **Bias-variance trade-off.** Ridge introduces a small bias (the estimator no longer equals `w_true` in expectation) but dramatically reduces variance. For high-noise or high-dimensional problems, the bias-variance sum is smaller with ridge than OLS.

**Bayesian interpretation.** Adding L2 is equivalent to putting a **Gaussian prior** on w: `w ~ N(0, σ²_w · I)`. Then MAP estimation gives the ridge solution with `λ = σ²_noise / σ²_w`. See [MAP estimation & regularization](/learn/mathematics/statistics/map-and-priors) for the full derivation.

**Standardize features first.** Ridge penalizes `Σ w_i²`. If feature `x_1` is in dollars and `x_2` in cents, the coefficients are on wildly different scales — the ridge penalty unfairly punishes `x_2`'s huge coefficient. Fix: subtract mean and divide by std for each feature before fitting; the same penalty then applies fairly.

## L1 (lasso regression)

**Penalized loss:**

$$
L_{\text{lasso}}(w) = \frac{1}{2} \|Xw - y\|^2 + \lambda \|w\|_1
$$

where `‖w‖₁ = Σ|w_i|`. The absolute value has a **kink at zero** where the gradient is not defined — the "subgradient" is [−1, 1]. This kink produces sparsity: the optimizer prefers weights that sit at zero rather than at some tiny positive value, because zero has zero contribution to the penalty and any tiny perturbation increases the penalty by |perturbation|.

**No closed form.** Lasso is solved by **coordinate descent**: iterate over coordinates, and for each one, minimize the loss with respect to that coordinate while holding others fixed. The per-coordinate update turns out to be a **soft-threshold**:

$$
w_j \leftarrow \text{sgn}(\rho_j) \cdot \max(|\rho_j| - \lambda, 0), \quad \rho_j = \text{residual correlation with feature } j
$$

If `|ρ_j| < λ`, then `w_j` is set to exactly zero — that's where sparsity comes from. Larger λ = stricter threshold = more zeros.

**Bayesian interpretation.** L1 corresponds to a **Laplace prior** on w: `w_i ~ Laplace(0, b)` with `λ = 1/b`. The Laplace distribution has a sharp peak at zero and heavy tails, unlike Gaussian's soft peak. This is why L1 pushes small weights to exactly zero.

**When to use:**

- **High-dimensional problems where you believe only a few features matter.** L1 automatically selects them.
- **Interpretability.** A model with 5 nonzero weights is much easier to explain than one with 500.
- **Storage/inference cost.** Sparse models can be stored and computed on more efficiently.

**When NOT to use:**

- **Correlated features.** L1 tends to arbitrarily pick one of a group of correlated features and zero out the others — the picked one is unstable. Elastic net (below) handles this better.
- **Every feature genuinely contributes a little.** L2 is a better fit — L1 will artificially zero out some features that actually help.

## Elastic net — L1 + L2 combined

**Penalized loss:**

$$
L_{\text{en}}(w) = \frac{1}{2} \|Xw - y\|^2 + \lambda \left[ \alpha \|w\|_1 + \frac{1 - \alpha}{2} \|w\|^2 \right]
$$

where `α ∈ [0, 1]` controls the L1-vs-L2 mix. `α = 1` is pure lasso; `α = 0` is pure ridge; `α = 0.5` is even mix.

**Why elastic net exists:** pure L1 has two issues that L2 doesn't share:

1. **Instability with correlated features.** L1 arbitrarily picks one of a correlated group and zeros the rest. Adding a small L2 term pulls correlated features together — the "group" of correlated features gets non-zero weights.
2. **Only D nonzero weights possible when N < D.** In high-dim regression, pure L1 can select at most N features (equal to the number of examples). L2 has no such limit.

Elastic net's L2 component fixes both. sklearn's `ElasticNet` uses this. Typical α values in production: 0.1 to 0.9 depending on how much sparsity you want.

## The regularization path and choosing λ

The **regularization path** shows how each coefficient evolves as λ ranges over `[0, ∞)`. Two extremes:

- **λ = 0:** coefficients equal the unregularized (OLS or logistic regression) solution.
- **λ → ∞:** coefficients all go to zero.

For a well-chosen λ in the middle, coefficients are neither wildly overfit (λ = 0) nor uniformly squashed (λ = ∞).

<RunnableCode title="Regularization paths — ridge vs lasso" code={regPathCode} packages={["numpy", "matplotlib"]} captureFig={true} />

**Choosing λ by k-fold cross-validation.** Standard recipe:

1. Split training data into K folds (typically K = 5 or 10).
2. For each candidate λ: fit K models (each on K-1 folds), evaluate on the held-out fold.
3. Average the K validation errors.
4. Pick the λ that minimizes the average error.

**One-standard-error rule (optional):** instead of picking the minimum-CV-error λ, pick the LARGEST λ whose CV error is within 1 standard deviation of the minimum. Rationale: the minimum is a noisy estimator, and picking slightly-more-regularization protects against overfitting the CV score.

<RealWorldML title="Weight decay in transformer training — L2 in disguise">

Every modern LLM training config includes a line like:

```yaml
weight_decay: 0.1
```

This IS L2 regularization. The `optimizer.step()` after computing gradients applies:

$$
w \leftarrow w - \eta \cdot \nabla L(w) - \eta \cdot \text{wd} \cdot w
$$

The second term `−η · wd · w` shrinks every parameter toward zero by a small amount each step. That's exactly the effect of an L2 penalty `½ · wd · ‖w‖²` in the loss (gradient of that penalty w.r.t. w is `wd · w`).

**Why is wd = 0.1 a lot?** In classical ML, ridge `λ` is chosen by CV and typically small (say 0.001–1). In LLM training, "weight decay 0.1" seems large — but it's applied per step, and there are hundreds of thousands of steps. The cumulative shrinkage is enormous. Empirical tuning at scale finds this level of wd is optimal for transformer generalization.

**AdamW vs Adam:** the "W" in AdamW is precisely the correct application of weight decay. Original Adam applied weight decay to the ADAPTIVE step (divided by √v), which meant heavily-updated parameters got extra shrinkage. AdamW decouples wd from the adaptive step, applying it after the Adam update. This is why AdamW is the standard for LLM training and plain Adam is deprecated for transformer work.

**LoRA as regularization by architecture.** LoRA constrains fine-tuning to `W_frozen + B·A` where `B, A` are low-rank. The rank-r constraint IS a form of structural regularization: you literally forbid full-rank updates. This works because most of a pretrained LLM's parameters are already "close to good" for many downstream tasks — you just need small directional nudges, not full re-training.

**Pruning as extreme L1.** Once you've trained a model, you can "prune" (set to exactly zero) the smallest-magnitude weights. Studies show 90%+ of transformer weights can be pruned with minimal accuracy loss — evidence that L1-flavored sparsity is inherent in over-parameterized nets. Structured pruning (removing entire rows/columns) is standard for inference-time model compression.

**Quantization as very coarse regularization.** Reducing weights to 4-bit or 8-bit precision constrains their representable values. This is a discrete regularizer that trades exact precision for storage and compute efficiency. Standard for deployment of LLMs on-device.

</RealWorldML>

<Summary>

- **Regularization** = add a penalty on model complexity to the training loss. Prevents overfitting.
- **L2 (ridge):** penalty is (λ/2)‖w‖². Shrinks all weights toward zero, closed-form solvable, Gaussian-prior interpretation.
- **L1 (lasso):** penalty is λ‖w‖₁. Produces SPARSE solutions (implicit feature selection), no closed form, coordinate-descent solvable, Laplace-prior interpretation.

- **Choose λ by k-fold cross-validation.** Use one-standard-error rule for extra safety.
- **Standardize features first!** Without standardization, the regularizer unfairly penalizes features with larger units.
- **Every transformer's weight_decay parameter IS L2 regularization.** Understanding L2 in classical ML directly transfers to modern deep learning training.
- **AdamW = Adam + correctly-applied weight decay.** The "W" is what makes weight decay work with adaptive optimizers.
