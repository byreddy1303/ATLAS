export const overfitDemoCode = `# Overfitting demo: fit high-degree polynomial regression WITHOUT and WITH ridge.
# See how L2 tames the wild oscillations.
import numpy as np, matplotlib.pyplot as plt

np.random.seed(0)
N_train = 15
x_train = np.linspace(-3, 3, N_train)
y_train = np.sin(x_train) + 0.15 * np.random.randn(N_train)
x_test  = np.linspace(-3, 3, 200)
y_test  = np.sin(x_test)                                 # noiseless truth

# Polynomial feature matrix
def poly_features(x, degree):
    return np.stack([x**k for k in range(degree + 1)], axis=1)

degree = 12
X_train = poly_features(x_train, degree)
X_test  = poly_features(x_test,  degree)

def fit(X, y, lam):
    "Ridge regression: (X.T X + lam I) w = X.T y."
    D = X.shape[1]
    return np.linalg.solve(X.T @ X + lam * np.eye(D), X.T @ y)

lambdas = [0.0, 0.01, 1.0, 100.0]
fig, axes = plt.subplots(1, 4, figsize=(15, 3.6))
for ax, lam in zip(axes, lambdas):
    w = fit(X_train, y_train, lam)
    y_pred = X_test @ w
    ax.plot(x_test, y_test, "-", color="teal", lw=2, label="true sin(x)")
    ax.plot(x_test, y_pred, "-", color="crimson", lw=2, label=f"ridge (lam = {lam})")
    ax.plot(x_train, y_train, "o", color="royalblue", markersize=6, label="noisy train points")
    ax.set_ylim(-2, 2); ax.set_title(f"lam = {lam}")
    ax.legend(fontsize=8); ax.grid(alpha=0.3)
plt.tight_layout()

# Report weight norms
print("Weight norms (L2) — bigger = model fits noise via wild coefficients")
for lam in lambdas:
    w = fit(X_train, y_train, lam)

export const l1VsL2Code = `# L1 vs L2: same optimization framework, wildly different behavior.
# L2 shrinks all weights toward zero; L1 sets many weights exactly to zero.
import numpy as np, matplotlib.pyplot as plt

np.random.seed(0)

# Synthetic problem with a mix of important and irrelevant features
N, D = 100, 20
X = np.random.randn(N, D)
w_true = np.zeros(D)
w_true[:5] = np.array([2.0, -1.5, 3.0, 0.5, -2.0])       # only first 5 features matter
y = X @ w_true + 0.3 * np.random.randn(N)

def ridge(X, y, lam):
    D = X.shape[1]
    return np.linalg.solve(X.T @ X + lam * np.eye(D), X.T @ y)

def soft_threshold(w, thresh):
    "Soft-threshold operator: pull each entry toward 0 by 'thresh', clip at 0."
    return np.sign(w) * np.maximum(np.abs(w) - thresh, 0)

def lasso_cd(X, y, lam, steps=500):
    "Lasso via coordinate descent — the standard L1 solver."
    N, D = X.shape
    w = np.zeros(D)
    xj_sq = np.sum(X ** 2, axis=0)
    for _ in range(steps):
        for j in range(D):
            residual = y - X @ w + X[:, j] * w[j]
            rho = X[:, j] @ residual
            w[j] = soft_threshold(rho / xj_sq[j], lam * N / xj_sq[j])
    return w

# Fit both at similar effective regularization
w_ridge = ridge(X, y, lam=5.0)
w_lasso = lasso_cd(X, y, lam=0.1)

print("True weights (only first 5 nonzero):")
print(w_true.round(2))
print()
print("Ridge fit — shrinks ALL weights toward 0 but leaves them nonzero:")
print(w_ridge.round(2))
print(f"  nonzero-count in ridge: {(np.abs(w_ridge) > 0.01).sum()} / {D}")
print()
print("Lasso fit — sets many weights EXACTLY to zero:")
print(w_lasso.round(2))
print(f"  nonzero-count in lasso: {(np.abs(w_lasso) > 0.01).sum()} / {D}")

# Visualize the difference
fig, axes = plt.subplots(1, 2, figsize=(11, 3.5))
axes[0].bar(range(D), w_ridge, color="teal");   axes[0].set_title(f"Ridge (L2): {(np.abs(w_ridge) > 0.01).sum()} nonzero")
axes[0].axhline(0, color="k", lw=0.5); axes[0].grid(alpha=0.3)
axes[1].bar(range(D), w_lasso, color="crimson"); axes[1].set_title(f"Lasso (L1): {(np.abs(w_lasso) > 0.01).sum()} nonzero")
axes[1].axhline(0, color="k", lw=0.5); axes[1].grid(alpha=0.3)
for ax in axes: ax.set_xlabel("feature index"); ax.set_ylabel("coefficient")
plt.tight_layout()`;

export const cvLambdaCode = `# Choose lambda via K-fold cross-validation.
# For each lambda, average validation error over K folds; pick the smallest.
import numpy as np

np.random.seed(0)
N, D = 200, 30
X = np.random.randn(N, D)
w_true = np.zeros(D)
w_true[:8] = np.random.randn(8) * 2
y = X @ w_true + 0.5 * np.random.randn(N)

def ridge(X, y, lam):
    return np.linalg.solve(X.T @ X + lam * np.eye(X.shape[1]), X.T @ y)

def kfold_ridge_cv(X, y, lam, K=5, seed=1):
    rng = np.random.default_rng(seed)
    N = len(y)
    idx = rng.permutation(N)
    fold_sizes = np.array_split(idx, K)
    errors = []
    for k in range(K):
        val_idx = fold_sizes[k]
        train_idx = np.concatenate([f for i, f in enumerate(fold_sizes) if i != k])
        w = ridge(X[train_idx], y[train_idx], lam)
        err = np.mean((X[val_idx] @ w - y[val_idx]) ** 2)
        errors.append(err)
    return np.mean(errors), np.std(errors)

lams = np.logspace(-3, 3, 30)
print(f"{'lambda':>8}  {'CV MSE':>10}  {'std':>10}")
print("-" * 32)
best_lam, best_mse = None, np.inf
for lam in lams:
    mse, sd = kfold_ridge_cv(X, y, lam)
    if mse < best_mse:
        best_lam, best_mse = lam, mse
    if lam in [1e-3, 1e-2, 1e-1, 1.0, 10.0, 100.0, 1000.0]:
        print(f"{lam:>8.3f}  {mse:>10.4f}  {sd:>10.4f}")
print()
print(f"Best lambda: {best_lam:.4f}   with CV MSE {best_mse:.4f}")
print()
print("This is exactly what sklearn's RidgeCV does under the hood.")
print("Standard rule: pick the lambda that minimizes CV error, or")
print("the 'one-standard-error' rule: pick the LARGEST lambda whose CV MSE")
print("is within 1 SD of the minimum, for extra regularization.")`;

<Motivation title="Regularization — the trick that turns memorizers into generalizers">

Every supervised model has one fundamental tension: **fit the training data well** vs. **generalize to new data.** These are not the same thing. In fact they're often at war.

Give a flexible model (a high-degree polynomial, a deep network, a big decision tree) enough freedom and it can achieve near-zero training loss by *memorizing* every point — including all the noise. Then you show it a new example and it flails. That failure mode has a name: **overfitting** — the model learned the training set instead of the underlying pattern.

**Regularization** is the toolkit for fighting overfitting. The core idea in one sentence: **add a penalty on model complexity to the loss, so training rewards simplicity unless the data really demands otherwise.** The model is now optimizing "fit the data *and* stay simple" instead of just "fit the data." That trade-off, controlled by a single knob called `λ` (Greek "lambda"), does more to protect real-world ML models than almost anything else.

For a GenAI engineer, regularization is *everywhere*:

1. **Every transformer training config has a `weight_decay=0.1` line — that's L2 regularization.** Reading and tuning training configs requires knowing what this does. (Spoiler: it shrinks weights toward zero every step.)
2. **LoRA is regularization by architecture.** Constraining fine-tuning to a **low-rank** update — "you can only change the weights in this restricted way" — is a strong form of complexity control. You forbid full-rank updates entirely.
3. **RLHF's KL penalty is regularization on the policy.** Penalizing how far the new model drifts from a reference model is analogous to L2 on the *distance between distributions*.
4. **Sparse fine-tuning methods** like **BitFit** (only train the bias terms) and **prompt tuning** (only train the prompt embedding) use L1-style sparsity — they work because most parameters don't need to change for most tasks.
5. **Model compression.** Pruning (setting small weights to zero) and quantization (using 4-bit or 8-bit numbers instead of 32-bit) are aggressive regularizers applied *after* training to shrink deployment size.

This chapter covers the three classical regularizers — **L2 (ridge)**, **L1 (lasso)**, and **elastic net** — and how to pick the strength `λ` using cross-validation. Everything you learn here transfers directly to modern deep learning.

</Motivation>

<Intuition title="Add a fee for using big weights — the shape of the fee decides everything">

Start with plain linear regression's loss: `L(w) = ½ ‖Xw − y‖²`. Now add a **regularization term** — a penalty that gets bigger when the weights get bigger:

$$
L_{\text{reg}}(w) = \frac{1}{2} \|Xw - y\|^2 + \lambda \cdot \text{penalty}(w)
$$

Read it: "minimize the fit error *plus* a fee for using big weights." The knob `λ` (lambda) sets how much you care about the fee vs. the fit — a **hyperparameter** you'll tune with cross-validation.

The two most common penalty shapes give wildly different behavior:

- **L2 penalty (ridge)** — `penalty(w) = Σ wᵢ² = ‖w‖²`. Big weights get punished *quadratically* (weight of 10 → penalty of 100). Effect: **shrinks all weights uniformly toward zero**, but never *exactly* to zero. Every feature still shows up in the model, just with smaller coefficients.
- **L1 penalty (lasso)** — `penalty(w) = Σ |wᵢ| = ‖w‖₁`. Big weights get punished *linearly*. Because absolute value has a sharp **kink at zero** (a non-smooth point where the slope changes from −1 to +1), this penalty produces **sparse** solutions — many weights get pushed to *exactly* zero. Any feature whose coefficient is zero is effectively removed from the model. That's why L1 is called **implicit feature selection**.

**A picture worth carrying around.** Imagine your loss surface, and now say the optimizer is only allowed to search inside a "ball" of allowed weights (the shape of the ball depends on which penalty you chose):

- **L2's ball is a round sphere.** The best point tends to be somewhere on its smooth surface — usually not on any axis.
- **L1's "ball" is a diamond** (rotated square in 2D). Its corners sit exactly on the coordinate axes. The best point often lands on a corner — which forces one or more coordinates to be zero. That's the geometric reason for sparsity.

**About that λ knob.** Small `λ` → weak fee → answer close to plain regression, potentially overfit. Large `λ` → heavy fee → weights crushed toward zero, model too simple. Somewhere in the middle sits the "just right" value that generalizes best — and you find it by trying several and using cross-validation to score them.

</Intuition>

<Recall term="MSE loss">

Mean squared error `L(w) = (1/N) · Σ (ŷ_i − y_i)² = (1/N) · ‖Xw − y‖²`. This is the standard linear-regression loss, differentiable everywhere, corresponding to MLE under Gaussian noise. All regularizers in this chapter modify MSE by ADDING a penalty term — the base loss stays the same.

</Recall>

## L2 (ridge regression) — the smooth, everywhere-applicable regularizer

**The loss:**

$$
L_{\text{ridge}}(w) = \frac{1}{2} \|Xw - y\|^2 + \frac{\lambda}{2} \|w\|^2
$$

Same fit-error term as linear regression, plus `(λ/2) · ‖w‖²` — half the sum of squared weights, scaled by `λ`. (The `1/2` is just there so the gradient comes out clean.)

**The gradient** looks almost like linear regression's, with one extra `λw` term:

$$
\nabla L_{\text{ridge}}(w) = X^\top (Xw - y) + \lambda w
$$

**Closed-form solution.** Set the gradient to zero, do a bit of algebra:

$$
(X^\top X + \lambda I) w = X^\top y \quad \Longrightarrow \quad w^*_{\text{ridge}} = (X^\top X + \lambda I)^{-1} X^\top y
$$

Notice: it's the linear-regression formula with one small change — an extra `λI` (`I` = identity matrix, "diagonal of 1s") added to `Xᵀ X`. That one added term does two very good things:

1. **Fixes the multicollinearity problem.** Remember from last chapter — when features are correlated, `Xᵀ X` becomes near-singular and coefficients blow up. Adding `λ · I` shifts every eigenvalue up by `λ`, guaranteeing invertibility. Ridge is *the* standard fix for multicollinearity.
2. **Trades a little bias for a lot less variance.** The ridge estimator is *slightly biased* (on average, doesn't return the exact true weights) but has *much lower variance* (much less sensitive to the specific noise in your training data). For noisy or high-dimensional data, this trade almost always wins.

**Why ridge is the same thing as putting a "Gaussian prior" on weights.** If you believe (before seeing any data) that each weight is drawn from a bell curve centered at zero — call it `w ~ N(0, σ²ᵥᵥ · I)` — and you do **MAP estimation** (maximum a posteriori — the Bayesian version of MLE that also accounts for a prior), you get *exactly* the ridge formula, with `λ = σ²ₙₒᵢₛₑ / σ²ᵥᵥ`. So ridge is literally "assume weights are probably small before I see any data" translated into math. See [MAP estimation & regularization](/learn/mathematics/statistics/map-and-priors) for the full derivation.

**Standardize features before regularizing — this is critical.** The ridge penalty punishes each `wᵢ²`, treating every coefficient equally. If `x₁` is in dollars and `x₂` is in cents, the two coefficients need to differ by 100× to have equivalent effects on `y`. Ridge doesn't know that — it'll unfairly crush the huge coefficient. Fix: subtract the mean and divide by the standard deviation for each feature before fitting. Now everything is on the same scale and the penalty is fair.

## L1 (lasso regression) — the feature-selecting sibling

**The loss:**

$$
L_{\text{lasso}}(w) = \frac{1}{2} \|Xw - y\|^2 + \lambda \|w\|_1
$$

where `‖w‖₁ = Σ |wᵢ|` — the sum of absolute values. That absolute value is the whole story. It has a **kink at zero** where the derivative jumps from `−1` to `+1` — technically the derivative isn't defined *at* zero (something math people call a **subgradient** — the interval `[−1, +1]` of valid slopes at that point).

Here's why the kink matters. When the fit-term gradient is small (a nearly-inactive feature), L1's constant `±λ` slope near zero pushes the weight *all the way* to zero — because zero contributes zero to the penalty, and any nonzero value contributes `λ · |value|` to it. Compare to L2, where the penalty gradient is `λ · w` — that goes to zero as `w` goes to zero, so there's no "final push" to snap it to exactly zero. This asymmetry is the mathematical reason for L1's sparsity.

**No closed form — sorry.** Because of the kink, there's no matrix formula. The standard solver is **coordinate descent**: cycle through the coordinates one at a time; for each `wⱼ`, hold every other weight fixed and minimize the loss with respect to just `wⱼ`. That single-coordinate update turns out to have a beautiful form — the **soft-threshold**:

$$
w_j \leftarrow \text{sign}(\rho_j) \cdot \max(|\rho_j| - \lambda, 0), \quad \rho_j = \text{"correlation" of feature } j \text{ with the residual}
$$

Read the soft-threshold: if `|ρⱼ| < λ`, snap `wⱼ` to exactly zero. Otherwise pull it toward zero by `λ`. That's the sparsity mechanism spelled out.

**Bayesian view.** L1 corresponds to a **Laplace prior** on the weights (a distribution shaped like a sharp tent centered at zero — sharp peak, heavy tails). Ridge's Gaussian prior is smooth at the top; Laplace's sharp peak is what encourages the MAP estimator to land exactly at zero.

**When to reach for L1:**

- **Lots of features, only a few actually matter.** L1 will find them for you automatically.
- **You need an interpretable model.** A model with 5 nonzero weights is a lot easier to explain than one with 500.
- **Deployment care about storage or inference cost.** Sparse models are cheaper to store and evaluate.

**When to *not* reach for L1:**

- **Your features are highly correlated.** L1 arbitrarily picks one of a correlated group and zeros the others — the "picked one" is *unstable* (tiny data changes can flip which one gets kept). Elastic net (next) fixes this.
- **Every feature contributes a bit.** L1 will artificially zero out features that actually help. In that case, L2 is a better fit.

## Elastic net — the "best of both worlds" mix

**The loss:**

$$
L_{\text{en}}(w) = \frac{1}{2} \|Xw - y\|^2 + \lambda \left[ \alpha \|w\|_1 + \frac{1 - \alpha}{2} \|w\|^2 \right]
$$

Elastic net is just a weighted combination of L1 and L2. The `α ∈ [0, 1]` knob picks the mix: `α = 1` is pure lasso, `α = 0` is pure ridge, `α = 0.5` is an even blend.

**Why bother?** Because pure L1 has two annoying quirks that pure L2 doesn't:

1. **Unstable with correlated features.** L1 arbitrarily picks one from a group of correlated features and zeros out the rest. Which one gets picked depends on tiny numerical details — flip a bit and the model changes. Not great. Adding even a small L2 term makes it "share weight" among correlated features (the way ridge does).
2. **Capped at `N` nonzero weights when `N < D`** (fewer examples than features — common in genomics, some NLP settings). Pure L1 can only select at most `N` features no matter how many you started with. L2 has no such cap.

Elastic net inherits ridge's stability and lasso's sparsity. `sklearn.linear_model.ElasticNet` uses exactly this formulation. Practical `α` values are typically 0.1 to 0.9 depending on how much sparsity you want (higher `α` → more sparsity).

## The regularization path — watching coefficients evolve

The **regularization path** is a plot of how each coefficient changes as you slide `λ` from 0 up to infinity. It's a great way to *see* what different regularizers actually do. Two extremes:

- **`λ = 0`:** no regularization at all. Coefficients equal the plain OLS (or logistic regression) fit.
- **`λ → ∞`:** the fee for using any weight becomes infinite. Every coefficient goes to zero.

The interesting stuff happens in the middle. A well-chosen `λ` sits at the sweet spot: enough regularization to prevent overfitting, not so much that useful features get squashed.

**How do you actually pick `λ`?** The standard method is **k-fold cross-validation** — a technique you'll see everywhere in ML. Here's the recipe (we cover this in detail in the [cross-validation chapter](/learn/core-ml/supervised-basics/cross-validation)):

1. Split your training data into `K` chunks (folds). `K = 5` or `K = 10` are typical.
2. For each candidate `λ`: train `K` separate models, each on `K−1` folds, and evaluate each on the held-out fold. You end up with `K` validation errors.
3. Average those `K` errors.
4. Pick the `λ` with the smallest average error.

**One useful refinement — the "one-standard-error rule."** Instead of picking the exact minimum, pick the *largest* `λ` whose CV error is within one standard deviation of the minimum. Why? Because the minimum itself is a noisy estimate — the difference between the "best" `λ` and one slightly larger is often within the noise. Picking the larger one gives you extra safety without meaningfully sacrificing accuracy.

<RealWorldML title="Weight decay in transformer training — ridge in disguise">

Open any modern LLM training config and you'll see a line like this:

```yaml
weight_decay: 0.1
```

That is L2 regularization, hiding under a different name. Here's what it actually does. After the optimizer computes its usual gradient step, it also subtracts a small multiple of the weights themselves:

$$
w \leftarrow w - \eta \cdot \nabla L(w) - \eta \cdot \text{wd} \cdot w
$$

That second term `−η · wd · w` is precisely what you'd get if you added an L2 penalty `(wd/2) · ‖w‖²` to the loss and differentiated. Same shrink-toward-zero effect. Different framing, identical math.

**"wd = 0.1" sounds huge — why does it work?** In classical ML, ridge's `λ` is typically small (0.001 to 1) after cross-validation. But LLM training has *hundreds of thousands* of optimizer steps. Applying "shrink by 10%" every step compounds into enormous total shrinkage. Empirical tuning at scale settled on this range as optimal for transformer generalization.

**AdamW vs Adam — the "W" matters.** Original Adam applied weight decay via the adaptive-step formula, which meant heavily-updated parameters got *extra* shrinkage. That was a subtle bug. **AdamW** (the "W" is for weight decay, done right) decouples weight decay from the adaptive step, applying it plainly *after* the Adam update. This one fix dramatically improved transformer training. It's the reason AdamW is the standard optimizer for LLMs today, and plain Adam is discouraged for transformer work.

**LoRA — regularization by architecture.** LoRA constrains fine-tuning to `W_frozen + B · A` where `B` and `A` are **low-rank matrices** (few rows or few columns). By construction, the update `B · A` can only span a small subspace — you literally *cannot* do a full-rank update to the pretrained weights. That's a hard structural regularization. It works because a pretrained LLM's weights are already "mostly right" for many downstream tasks; you only need small nudges in a few directions, not a full retrain.

**Pruning — extreme L1 applied after training.** Once a model is trained, you can **prune** it: set every weight below some magnitude threshold to exactly zero. Studies routinely find 90%+ of transformer weights can be pruned with minimal accuracy loss. That's L1's sparsity story applied *after the fact* — the model was so overparameterized that most weights were doing very little.

**Quantization — coarse regularization for cheap deployment.** **Quantization** means using low-precision numbers for the weights — 8-bit or even 4-bit integers instead of 32-bit floats. That reduces storage 4×-8× and speeds up inference dramatically. Mathematically, quantization constrains weights to a small discrete set of representable values — a form of aggressive regularization. Standard for on-device LLM deployment.

</RealWorldML>

<Summary>

- **Regularization** = add a penalty on model complexity to the training loss. Prevents overfitting.
- **L2 (ridge):** penalty is (λ/2)‖w‖². Shrinks all weights toward zero, closed-form solvable, Gaussian-prior interpretation.
- **L1 (lasso):** penalty is λ‖w‖₁. Produces SPARSE solutions (implicit feature selection), no closed form, coordinate-descent solvable, Laplace-prior interpretation.
- **Elastic net:** α · L1 + (1-α) · L2 combo. Handles correlated features better than pure L1.
- **Choose λ by k-fold cross-validation.** Use one-standard-error rule for extra safety.
- **Standardize features first!** Without standardization, the regularizer unfairly penalizes features with larger units.
- **Every transformer's weight_decay parameter IS L2 regularization.** Understanding L2 in classical ML directly transfers to modern deep learning training.
- **AdamW = Adam + correctly-applied weight decay.** The "W" is what makes weight decay work with adaptive optimizers.

</Summary>
