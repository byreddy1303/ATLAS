export const kfoldCode = `# K-fold CV from scratch on a simple regression problem.
# The API of sklearn.KFold, but explicit so you see what's happening.
import numpy as np
from sklearn.linear_model import Ridge

np.random.seed(0)
N = 200
X = np.random.randn(N, 10)
y = X[:, 0] + 0.5 * X[:, 1] + 0.3 * np.random.randn(N)

def kfold_indices(N, K, seed=0):
    "Return list of (train_idx, val_idx) for K folds."
    rng = np.random.default_rng(seed)
    idx = rng.permutation(N)
    folds = np.array_split(idx, K)
    splits = []
    for k in range(K):
        val_idx = folds[k]
        train_idx = np.concatenate([f for i, f in enumerate(folds) if i != k])
        splits.append((train_idx, val_idx))
    return splits

def kfold_ridge_score(X, y, alpha, K=5):
    scores = []
    for train_idx, val_idx in kfold_indices(len(y), K):
        model = Ridge(alpha=alpha).fit(X[train_idx], y[train_idx])
        scores.append(np.mean((model.predict(X[val_idx]) - y[val_idx]) ** 2))
    return np.mean(scores), np.std(scores)

# Sweep alpha (ridge parameter)
alphas = [0.001, 0.01, 0.1, 1.0, 10.0, 100.0]
print(f"{'alpha':>8}  {'CV MSE':>10}  {'std':>8}")
print("-" * 30)
best_alpha, best_mse = None, np.inf
for a in alphas:
    mse, sd = kfold_ridge_score(X, y, a)
    if mse < best_mse:
        best_alpha, best_mse = a, mse
    print(f"{a:>8.3f}  {mse:>10.4f}  {sd:>8.4f}")

print(f"\\nBest alpha by CV: {best_alpha}   with CV MSE {best_mse:.4f}")
print()
print("Compare to a 'no-CV' approach: pick alpha by minimizing TRAINING MSE.")
train_alpha, train_mse = None, np.inf
for a in alphas:
    m = Ridge(alpha=a).fit(X, y)
    mse = np.mean((m.predict(X) - y) ** 2)
    if mse < train_mse:
        train_alpha, train_mse = a, mse
print(f"Training-loss-picked alpha: {train_alpha}   (always picks smallest alpha -- OVERFITS)")`;

export const leakageCode = `# The subtle-but-common data leakage bug: scale WITHIN a fold, not on the full data.
# Same trick applies to imputation, feature selection, any data-dependent preprocessing.
import numpy as np
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import Ridge
from sklearn.model_selection import KFold

np.random.seed(0)
N = 200
X = np.random.randn(N, 20)
# Add a "leaky" feature: mean of y over the whole dataset (should NOT be usable at prediction time)
y = X[:, 0] + 0.5 * X[:, 1] + 0.3 * np.random.randn(N)

# WRONG: scale on the whole dataset BEFORE splitting.
X_wrong = StandardScaler().fit_transform(X)      # uses ALL rows' mean & std
kf = KFold(n_splits=5, shuffle=True, random_state=0)
wrong_mses = []
for train_idx, val_idx in kf.split(X_wrong):
    m = Ridge(alpha=1.0).fit(X_wrong[train_idx], y[train_idx])
    wrong_mses.append(np.mean((m.predict(X_wrong[val_idx]) - y[val_idx]) ** 2))
print(f"WRONG (scale before split):   CV MSE = {np.mean(wrong_mses):.4f}   <- data leakage!")

# RIGHT: scale INSIDE each fold, using only training stats.
right_mses = []
for train_idx, val_idx in kf.split(X):
    scaler = StandardScaler().fit(X[train_idx])           # uses only training rows' mean & std
    X_tr_scaled  = scaler.transform(X[train_idx])
    X_val_scaled = scaler.transform(X[val_idx])
    m = Ridge(alpha=1.0).fit(X_tr_scaled, y[train_idx])
    right_mses.append(np.mean((m.predict(X_val_scaled) - y[val_idx]) ** 2))
print(f"RIGHT (scale inside each fold): CV MSE = {np.mean(right_mses):.4f}   <- honest estimate")

print()
print("The gap is small here because scaling doesn't leak much info.")
print("But for feature selection, imputation, or PCA: doing them on the WHOLE data before")
print("splitting can inflate 'CV performance' by 5-30 percentage points on real problems.")
print("Always: preprocess inside each fold. Or wrap it in a sklearn Pipeline (does this for you).")`;

export const nestedCvCode = `# Nested cross-validation: choose hyperparameters WITHIN an inner CV,
# report generalization performance on OUTER CV.
# Non-nested CV overestimates performance because you saw val loss for every alpha.
import numpy as np
from sklearn.linear_model import Ridge
from sklearn.model_selection import KFold

np.random.seed(0)
N = 300
X = np.random.randn(N, 30)
y = X[:, 0] + 0.5 * X[:, 1] + 0.3 * np.random.randn(N)

def outer_score(X, y, alpha_grid, K_outer=5, K_inner=5):
    "Nested CV: inner CV picks alpha, outer CV estimates true performance."
    outer_kf = KFold(n_splits=K_outer, shuffle=True, random_state=0)
    outer_scores = []
    picked_alphas = []
    for outer_train, outer_val in outer_kf.split(X):
        # Inner CV on outer_train -- pick best alpha
        inner_kf = KFold(n_splits=K_inner, shuffle=True, random_state=1)
        best_alpha, best_mse = None, np.inf
        for a in alpha_grid:
            fold_mses = []
            for inner_train, inner_val in inner_kf.split(X[outer_train]):
                m = Ridge(alpha=a).fit(X[outer_train][inner_train], y[outer_train][inner_train])
                pred = m.predict(X[outer_train][inner_val])
                fold_mses.append(np.mean((pred - y[outer_train][inner_val]) ** 2))
            if np.mean(fold_mses) < best_mse:
                best_alpha, best_mse = a, np.mean(fold_mses)
        picked_alphas.append(best_alpha)
        # Retrain on full outer_train with best alpha, evaluate on outer_val
        m = Ridge(alpha=best_alpha).fit(X[outer_train], y[outer_train])
        outer_scores.append(np.mean((m.predict(X[outer_val]) - y[outer_val]) ** 2))
    return np.mean(outer_scores), np.std(outer_scores), picked_alphas

alpha_grid = [0.01, 0.1, 1.0, 10.0, 100.0]
outer_mse, outer_sd, picked = outer_score(X, y, alpha_grid)
print(f"NESTED CV (honest generalization estimate):")
print(f"  outer MSE = {outer_mse:.4f} +/- {outer_sd:.4f}")
print(f"  inner CV picked alphas: {picked}")
print()

# Non-nested: pick alpha by CV, report the CV score
def naive_cv(X, y, alpha_grid, K=5):
    kf = KFold(n_splits=K, shuffle=True, random_state=0)
    best_alpha, best_mse = None, np.inf
    for a in alpha_grid:
        fold_mses = []
        for train_idx, val_idx in kf.split(X):
            m = Ridge(alpha=a).fit(X[train_idx], y[train_idx])
            fold_mses.append(np.mean((m.predict(X[val_idx]) - y[val_idx]) ** 2))
        if np.mean(fold_mses) < best_mse:
            best_alpha, best_mse = a, np.mean(fold_mses)
    return best_alpha, best_mse

naive_alpha, naive_mse = naive_cv(X, y, alpha_grid)
print(f"NAIVE CV (biased-optimistic):")
print(f"  best CV MSE = {naive_mse:.4f}  (picking best-of-{len(alpha_grid)} alphas biases downward)")
print(f"  best alpha = {naive_alpha}")
print()
print("Naive CV: takes the min over 5 alphas, so it's biased optimistic by ~sqrt(K)/N.")
print("Nested CV: never lets the outer test set influence hyperparameter choice.")
print("For an HONEST estimate of test performance, use nested CV.")`;

<Motivation title="Honest evaluation is the whole game — a great number that came from cheating is worse than a smaller one that's true">

Every ML pipeline ends with **a reported test number** — the score you show your boss, put in a paper, or use to decide whether to ship. If that number is 5-10 percentage points higher than what will actually happen in production, you've spent months building a lie you didn't know was a lie.

The main way this happens is **data leakage** — some information from the "test" side sneaks into the "train" side, making your model look better than it really is. **Cross-validation** done right is the discipline that keeps you honest.

For a GenAI engineer, this shows up in specific and important ways:

1. **Benchmark contamination.** Modern LLM benchmarks — MMLU, GSM8K, HumanEval, HellaSwag — have their questions (and sometimes their *answers*) scraped into the pretraining data of every new LLM. When a new model reports "94% on MMLU," the honest question is: how much is real capability, and how much is that the model *saw* the benchmark during training? This chapter's principles are what let you evaluate such claims critically.
2. **LLM-as-judge circularity.** Using an LLM to grade another LLM's outputs is a subtle version of the same trap — the judge's opinions were shaped by the same training data. Fine for rough dev signal; risky as the basis for public claims.
3. **Fine-tuning dataset construction.** SFT, DPO, prompt tuning all need labeled data. If your validation set shares underlying structure with your training set (same customers, same product SKUs, same document sources), your CV score won't predict production behavior. **Group k-fold** (later in this chapter) is the fix.
4. **RLHF reward model overfitting.** The reward model itself can overfit to its training preferences, and then RLHF chases that overfit reward — a phenomenon called **reward hacking**. Careful CV of the reward model is one of the guards against it.

This chapter covers the mechanics: k-fold, stratified splits, group splits, time-series splits, nested CV, and the top three ways data leakage sneaks in without anyone noticing.

</Motivation>

<Intuition title="Hide a chunk of your data. Pretend it's the future. Never let training peek.">

Here's the situation you're always in: you have some labeled data. You want to know how your model will do on *future*, unseen data. You can't actually fast-forward to the future — you have this one dataset and that's it.

**The trick.** Split your data into two parts. Train on one part. Evaluate on the other. As long as you keep the second part *strictly hidden* during training *and* during hyperparameter selection, its performance is an honest, unbiased estimate of how you'd do on brand-new data. That's the whole idea in one sentence.

**K-fold cross-validation** just does this trick `K` times — splits your data into `K` equal chunks, uses each chunk as the hidden "held-out" piece once, and averages the `K` scores. Doing it `K` times gives you a much more stable estimate (less noise) than doing it once, at the cost of `K` times as much compute.

**The three splits everyone talks about** — worth memorizing:

- **Training set** — what the model actually fits on. Typically 60-80% of your data.
- **Validation set** (a.k.a. "dev set") — used to *pick hyperparameters* (regularization strength, learning rate, model size, etc.). Different from test — you're *allowed* to look at it many times and make decisions based on it. Typically 10-20%.
- **Test set** — the final "how did we actually do" number. Used **exactly once**, at the very end. Every time you look at test performance and change something, you've *contaminated the test set* — it's now serving as another validation set, and any performance number you report on it is optimistically biased.

**The one commandment** — write it on your monitor: **anything your model or its hyperparameters depend on must not have been influenced by the validation or test data.** The most common way this rule gets broken is subtle: preprocessing steps like scaling, imputation, or feature selection. Fit those on the training portion only, then *apply* the fitted transform to the validation portion. Don't fit them on all your data before splitting.

</Intuition>

<Recall term="Overfitting">

Overfitting is when a model fits noise in the training set that doesn't generalize to new data. Manifests as low training loss and high validation/test loss. The whole point of cross-validation is to give an honest estimate of generalization performance — so you can detect overfitting and pick the model / regularization that generalizes best. Overfit training predictions can be arbitrarily good; overfit validation predictions are the actual measure.

</Recall>

## Train / val / test — the hygiene rules

**The classical three-way split** — a good starting point for any project:

- **Training set (60-80% of your data)** — what the model actually learns from.
- **Validation set (10-20%)** — for picking hyperparameters, early stopping, deciding model architecture. You look at this *many* times during development. That's fine.
- **Test set (10-20%)** — the final "how did we actually do" number. You look at it **once**, at the very end. If you look and then change something, you've turned it into another validation set and you no longer have an honest generalization estimate.

**For small datasets** (say fewer than 1,000 examples), a single train/val split is too noisy — one bad random draw and your score is meaningless. That's exactly when you switch to **k-fold CV**. For larger datasets, a single split often works fine and saves you compute.

**The rules of hygiene** — memorize these:

1. **Never fit *anything* on the validation set.** Not the model, not the scaler, not the imputer, not the feature selector. Nothing.
2. **All preprocessing that depends on data goes inside the CV loop.** Mean/std for scaling, top-k feature selection, PCA — fit them on the training fold only, then *apply* them (without refitting) to the validation fold.
3. **Look at the test set exactly once.** Every extra look creates a subtle "which experiment do I keep?" bias. Report and move on.
4. **The test set is not for iterating.** If test-set results surprise you, save the insight for next time. Don't tweak this model and re-report; you've now overfit to the test set.

## K-fold cross-validation — the standard workhorse

Here's the algorithm in three steps:

1. Shuffle your dataset (unless it's a time series — more on that later). Split it into `K` equal-size chunks (called **folds**).
2. For each fold `k = 1, 2, …, K`:
   - Set that fold aside as the "held-out" validation.
   - Train the model on the *other* `K − 1` folds combined.
   - Evaluate on the held-out fold, record the score.
3. Average the `K` scores. That's your CV estimate of test performance.

**Which `K` should you use?** Three sensible choices:

- **`K = 5`** — the everyday default. Each fold uses 80% for training and 20% for validation. Low variance in the estimate, moderate compute.
- **`K = 10`** — more stable estimate at 2× the cost. Common in academic ML.
- **`K = N`** (called **leave-one-out CV** or **LOOCV**) — each fold holds out exactly one example. Unbiased but extremely noisy at the fold level (each individual score is very variable). Expensive for large `N`. Rarely used in practice.

## The three sneakiest data-leakage bugs

These are the ways CV goes wrong that even careful people miss. Learn to recognize them.

**Bug 1 — Scaling before splitting.** You call `StandardScaler().fit_transform(X)` on the *entire* dataset, then split into train and test. Small bug, right? Not zero: the scaler's `mean_` and `scale_` now depend on every row, including the test rows. You've moved the training data's "coordinate system" using information from the test set. It's a mild leak, but it's still cheating.

**Fix:** fit the scaler inside each fold using only training data, then apply it (via `.transform`) to the validation fold.

**Bug 2 — Feature selection before splitting.** Say you have 10,000 candidate features. You pick the top 100 by "correlation with `y`" on the full dataset, then train models. The selection step *saw the labels of every row* — including the validation labels — when deciding which features to keep. On real problems this can inflate CV scores by **5 to 30 percentage points**, and it's completely invisible unless you know to look for it.

**Fix:** do feature selection inside each fold, using only that fold's training portion.

**Bug 3 — Class-imbalance handling on all the data.** You oversample the minority class (or apply **SMOTE**) to balance the training set. Fine — but if you oversampled *before* splitting, your training fold and validation fold might share duplicated minority-class examples. You'll appear to predict them perfectly.

**Fix:** balance the classes *inside* each fold, on the training portion only.

**The general rule** — one sentence you can tape to your monitor: **any data-dependent transformation must be fit inside the CV loop on training only.** That covers scaling, imputation, PCA, feature selection, class balancing, sample weighting — anything that "learns from the data." The cleanest way to enforce this is to wrap all your preprocessing plus the estimator in a `sklearn.Pipeline` and pass the Pipeline (not the raw estimator) to `cross_val_score`. Pipeline does the right thing automatically.

## Stratified k-fold — for classification with imbalance

Here's a subtle problem. Say your data is 10% positive class, 90% negative — a common setup for spam filters, fraud detection, disease prediction, etc. Use plain `KFold` and different folds will end up with different class ratios purely by chance — some folds might be 7% positive, others 13%. Adds noise to your CV scores. Worse, a training fold might end up with almost *no* positives, in which case the model can't learn the minority class at all.

**`StratifiedKFold`** fixes this by preserving the class distribution in every fold — each fold looks like a mini version of the whole dataset. Standard for any classification problem, especially with imbalance. Bonus: in sklearn, `cross_val_score(classifier, X, y)` uses stratified splits automatically when the estimator is a classifier.

## Group k-fold — for rows that aren't truly independent

Standard k-fold assumes each row is an **independent sample**. In real data that assumption often fails:

- Multiple medical records from **the same patient**.
- Multiple purchases from **the same customer**.
- Multiple frames from **the same video**.
- Multiple sentences from **the same document**.
- Multiple posts from **the same author**.

If you shuffle rows blindly, some rows from patient X land in training and others land in validation. Now the model can "cheat" by inferring patient X's characteristics from the training rows and using them to predict the held-out ones. Your CV score looks great — but it's measuring "generalization to *other rows from known patients*," not "generalization to *new patients*." The latter is what you actually care about.

**`GroupKFold`** partitions the data so that all rows sharing a group ID stay in the *same* fold. In sklearn:

```python
GroupKFold(n_splits=K).split(X, y, groups=patient_ids)
```

Use it for any medical, longitudinal, or nested data. It's a small code change that eliminates a huge class of overoptimistic CV estimates.

## Time-series splits — future can't leak into past

Any **temporal data** — time series, log data, sensor readings, financial prices, user activity — breaks a core assumption of standard k-fold. Standard k-fold shuffles rows, which randomly puts *future* rows in training folds and *past* rows in validation folds. Your model literally "sees the future" to predict "the past." No real-world deployment scenario looks like this, so CV scores from shuffled k-fold on time-series data are wildly optimistic.

**`TimeSeriesSplit`** uses **expanding-window** (or rolling-window) splits: training data always comes *before* validation data in time. Each fold's training set is a prefix of the timeline; each fold's validation is a short window immediately after that prefix.

**Time-series rules:**

- **Always** use TimeSeriesSplit (or a manually-implemented expanding-window split) on temporal data.
- **Never** shuffle rows before splitting on a time series.
- If there's short-term autocorrelation you don't want leaking (e.g., today's stock price is highly correlated with yesterday's), add a **gap** between the training window and the validation window: `TimeSeriesSplit(gap=100)` skips the 100 immediately-preceding rows.

## Nested cross-validation — for honest hyperparameter tuning

Here's a subtle bias that trips up almost every casual CV user. Say you use k-fold CV to try 10 candidate values of `λ`, and you *report* the CV score at the best one. That number is biased *optimistic* — you took the minimum over 10 noisy scores, and by chance, the noise helped one candidate look better than the others. On truly new data, your model won't do as well as the CV score suggests.

**Nested CV** fixes this with two nested loops:

- **Outer loop (`K_outer` folds).** For each outer fold, hold out a "test" chunk that will *never* be used for hyperparameter selection.
- **Inner loop (`K_inner` folds on the outer training set).** Try each candidate hyperparameter; the one with the best inner CV score wins.
- **Refit** with the best hyperparameters on the *full* outer training portion, then evaluate on the held-out outer test chunk.
- Average the outer test scores → your *honest* generalization estimate.

The cost is `K_outer × K_inner × N_hyperparams` total model fits. Expensive but standard for research papers and any deployment where "how well does this actually work" is the critical question.

<RealWorldML title="Benchmark contamination — the LLM-era version of data leakage">

The classical ML story about data leakage has a modern, high-stakes analog: **benchmark contamination.** Popular LLM benchmarks — MMLU, HellaSwag, HumanEval, GSM8K, and dozens more — have their questions (and often their answers) sitting on the public web. LLM pretraining pipelines crawl the public web. Do the math: when a new model reports "94% on MMLU," part of that is genuine capability, and part of it might be that the model *saw the benchmark questions during pretraining*.

**How does this happen?** In four steps:

1. Benchmarks get published — papers, HuggingFace datasets, GitHub repos.
2. LLM pretraining pipelines crawl the web indiscriminately.
3. Even with de-duplication filters, near-duplicate variants (translated, paraphrased, embedded in tutorials) slip through the cracks.
4. At evaluation time, the model sees a question it partially "remembers" from training and answers correctly for the wrong reason.

**How the field is responding:**

- **Fresh, curated benchmarks** — **MMLU-Pro**, **GPQA-Diamond**, **ARC-AGI**, **Big-Bench Hard**. Newer benchmarks, harder problems, and (in some cases) explicit contamination filtering.
- **Contamination detection.** Compare the model's log-probability of the *exact* benchmark string vs. paraphrased versions. If the exact string scores much higher, that's evidence of memorization.
- **Private test sets.** Organizations increasingly maintain internal evaluation datasets that never touch the public internet. Any lab claiming state-of-the-art on public benchmarks should also report on private ones.
- **Live task creation.** Ask domain experts to write brand-new questions *after* the model's training cutoff. Expensive, but the only truly clean signal.

**The rule that emerges:** don't fully trust any benchmark that was publicly available before your model's training cutoff. That's the modern equivalent of "don't trust CV that fit its scaler on the whole dataset before splitting."

**LLM-as-judge circularity.** Using GPT-4 to grade GPT-3.5's outputs is a subtle version of the same trap — the judge's opinions were shaped by the same pretraining data as the models being graded. Fine for rough dev signals. Risky as the basis for public scientific claims.

**Group k-fold applies to fine-tuning too.** If your fine-tuning data comes from a handful of specific customers or product SKUs, and your validation set is from the *same* customers/SKUs, your CV score will wildly overestimate how well the model generalizes to *new* customers. Group by `customer_id` (or whatever the natural unit is), then k-fold. Small change, huge honesty upgrade.

</RealWorldML>

<CodeItYourself title="Three drills — implement k-fold, spot the leakage, nested CV">

1. **Roll your own k-fold.** Given N and K, write a function that returns list of (train_idx, val_idx) tuples. Test on N = 100, K = 5: verify each fold has size 20, no index appears in both train and val, every index appears exactly once across all val sets.

2. **Detect leakage.** For a regression problem, fit a StandardScaler on the whole dataset before splitting. Then repeat, fitting scaler inside each fold. Compare CV scores — you should see the "leaked" one is slightly optimistic.

3. **Nested CV from scratch.** Write nested CV: outer 5-fold, inner 5-fold. For each outer fold, use inner CV to pick alpha from [0.01, 0.1, 1, 10, 100]. Report outer CV MSE. Compare to non-nested CV (which minimizes CV MSE over alpha directly — biased optimistic).

</CodeItYourself>

<Summary>

- **Three splits:** training (fit model), validation (pick hyperparameters), test (single-use, unbiased estimate).
- **K-fold CV:** average of K held-out scores; K = 5 or 10 is standard.
- **Preprocessing must be inside the fold** — otherwise data leakage from validation into training.
- **Stratified k-fold** for classification, **GroupKFold** for non-iid data, **TimeSeriesSplit** for temporal data.
- **Nested CV** = outer loop for honest generalization + inner loop for hyperparameter search. Prevents optimistic bias.
- **The test set is used exactly once.** Every look is a leak.
- **sklearn Pipeline** automates the "preprocess inside CV" rule; use it.
- **Modern LLM analog: benchmark contamination.** Trust benchmarks only when the questions weren't in the pretraining crawl.

</Summary>
