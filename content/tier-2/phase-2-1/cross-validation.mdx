export const kfoldCode = `# K-fold CV from scratch on a simple regression problem.
# The API of sklearn.KFold, but explicit so you see what's happening.
import numpy as np
from sklearn.linear_model import Ridge

np.random.seed(0)
N = 200
X = np.random.randn(N, 10)
y = X[:, 0] + 0.5 * X[:, 1] + 0.3 * np.random.randn(N)

def kfold_indices(N, K, seed=0):
    "Return list of (train_idx, val_idx) for K folds."
    rng = np.random.default_rng(seed)
    idx = rng.permutation(N)
    folds = np.array_split(idx, K)
    splits = []
    for k in range(K):
        val_idx = folds[k]
        train_idx = np.concatenate([f for i, f in enumerate(folds) if i != k])
        splits.append((train_idx, val_idx))
    return splits

def kfold_ridge_score(X, y, alpha, K=5):
    scores = []
    for train_idx, val_idx in kfold_indices(len(y), K):
        model = Ridge(alpha=alpha).fit(X[train_idx], y[train_idx])
        scores.append(np.mean((model.predict(X[val_idx]) - y[val_idx]) ** 2))
    return np.mean(scores), np.std(scores)

# Sweep alpha (ridge parameter)
alphas = [0.001, 0.01, 0.1, 1.0, 10.0, 100.0]
print(f"{'alpha':>8}  {'CV MSE':>10}  {'std':>8}")
print("-" * 30)
best_alpha, best_mse = None, np.inf
for a in alphas:
    mse, sd = kfold_ridge_score(X, y, a)
    if mse < best_mse:
        best_alpha, best_mse = a, mse
    print(f"{a:>8.3f}  {mse:>10.4f}  {sd:>8.4f}")

print(f"\\nBest alpha by CV: {best_alpha}   with CV MSE {best_mse:.4f}")
print()
print("Compare to a 'no-CV' approach: pick alpha by minimizing TRAINING MSE.")
train_alpha, train_mse = None, np.inf
for a in alphas:
    m = Ridge(alpha=a).fit(X, y)
    mse = np.mean((m.predict(X) - y) ** 2)
    if mse < train_mse:
        train_alpha, train_mse = a, mse
print(f"Training-loss-picked alpha: {train_alpha}   (always picks smallest alpha -- OVERFITS)")`;

export const leakageCode = `# The subtle-but-common data leakage bug: scale WITHIN a fold, not on the full data.
# Same trick applies to imputation, feature selection, any data-dependent preprocessing.
import numpy as np
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import Ridge
from sklearn.model_selection import KFold

np.random.seed(0)
N = 200
X = np.random.randn(N, 20)
# Add a "leaky" feature: mean of y over the whole dataset (should NOT be usable at prediction time)
y = X[:, 0] + 0.5 * X[:, 1] + 0.3 * np.random.randn(N)

# WRONG: scale on the whole dataset BEFORE splitting.
X_wrong = StandardScaler().fit_transform(X)      # uses ALL rows' mean & std
kf = KFold(n_splits=5, shuffle=True, random_state=0)
wrong_mses = []
for train_idx, val_idx in kf.split(X_wrong):
    m = Ridge(alpha=1.0).fit(X_wrong[train_idx], y[train_idx])
    wrong_mses.append(np.mean((m.predict(X_wrong[val_idx]) - y[val_idx]) ** 2))
print(f"WRONG (scale before split):   CV MSE = {np.mean(wrong_mses):.4f}   <- data leakage!")

# RIGHT: scale INSIDE each fold, using only training stats.
right_mses = []
for train_idx, val_idx in kf.split(X):
    scaler = StandardScaler().fit(X[train_idx])           # uses only training rows' mean & std
    X_tr_scaled  = scaler.transform(X[train_idx])
    X_val_scaled = scaler.transform(X[val_idx])
    m = Ridge(alpha=1.0).fit(X_tr_scaled, y[train_idx])
    right_mses.append(np.mean((m.predict(X_val_scaled) - y[val_idx]) ** 2))
print(f"RIGHT (scale inside each fold): CV MSE = {np.mean(right_mses):.4f}   <- honest estimate")

print()
print("The gap is small here because scaling doesn't leak much info.")
print("But for feature selection, imputation, or PCA: doing them on the WHOLE data before")
print("splitting can inflate 'CV performance' by 5-30 percentage points on real problems.")
print("Always: preprocess inside each fold. Or wrap it in a sklearn Pipeline (does this for you).")`;

export const nestedCvCode = `# Nested cross-validation: choose hyperparameters WITHIN an inner CV,
# report generalization performance on OUTER CV.
# Non-nested CV overestimates performance because you saw val loss for every alpha.
import numpy as np
from sklearn.linear_model import Ridge
from sklearn.model_selection import KFold

np.random.seed(0)
N = 300
X = np.random.randn(N, 30)
y = X[:, 0] + 0.5 * X[:, 1] + 0.3 * np.random.randn(N)

def outer_score(X, y, alpha_grid, K_outer=5, K_inner=5):
    "Nested CV: inner CV picks alpha, outer CV estimates true performance."
    outer_kf = KFold(n_splits=K_outer, shuffle=True, random_state=0)
    outer_scores = []
    picked_alphas = []
    for outer_train, outer_val in outer_kf.split(X):
        # Inner CV on outer_train -- pick best alpha
        inner_kf = KFold(n_splits=K_inner, shuffle=True, random_state=1)
        best_alpha, best_mse = None, np.inf
        for a in alpha_grid:
            fold_mses = []
            for inner_train, inner_val in inner_kf.split(X[outer_train]):
                m = Ridge(alpha=a).fit(X[outer_train][inner_train], y[outer_train][inner_train])
                pred = m.predict(X[outer_train][inner_val])
                fold_mses.append(np.mean((pred - y[outer_train][inner_val]) ** 2))
            if np.mean(fold_mses) < best_mse:
                best_alpha, best_mse = a, np.mean(fold_mses)
        picked_alphas.append(best_alpha)
        # Retrain on full outer_train with best alpha, evaluate on outer_val
        m = Ridge(alpha=best_alpha).fit(X[outer_train], y[outer_train])
        outer_scores.append(np.mean((m.predict(X[outer_val]) - y[outer_val]) ** 2))
    return np.mean(outer_scores), np.std(outer_scores), picked_alphas

alpha_grid = [0.01, 0.1, 1.0, 10.0, 100.0]
outer_mse, outer_sd, picked = outer_score(X, y, alpha_grid)
print(f"NESTED CV (honest generalization estimate):")
print(f"  outer MSE = {outer_mse:.4f} +/- {outer_sd:.4f}")
print(f"  inner CV picked alphas: {picked}")
print()

# Non-nested: pick alpha by CV, report the CV score
def naive_cv(X, y, alpha_grid, K=5):
    kf = KFold(n_splits=K, shuffle=True, random_state=0)
    best_alpha, best_mse = None, np.inf
    for a in alpha_grid:
        fold_mses = []
        for train_idx, val_idx in kf.split(X):
            m = Ridge(alpha=a).fit(X[train_idx], y[train_idx])
            fold_mses.append(np.mean((m.predict(X[val_idx]) - y[val_idx]) ** 2))
        if np.mean(fold_mses) < best_mse:
            best_alpha, best_mse = a, np.mean(fold_mses)
    return best_alpha, best_mse

naive_alpha, naive_mse = naive_cv(X, y, alpha_grid)
print(f"NAIVE CV (biased-optimistic):")
print(f"  best CV MSE = {naive_mse:.4f}  (picking best-of-{len(alpha_grid)} alphas biases downward)")
print(f"  best alpha = {naive_alpha}")
print()
print("Naive CV: takes the min over 5 alphas, so it's biased optimistic by ~sqrt(K)/N.")
print("Nested CV: never lets the outer test set influence hyperparameter choice.")
print("For an HONEST estimate of test performance, use nested CV.")`;

<Motivation title="Honest evaluation is the whole game — a nice number that comes from cheating is worse than a lower one that's true">

Every ML pipeline produces a **reported test performance** — the number you show your manager, put in a paper, or use to decide whether to ship. If that number is 5-10 percentage points higher than what will actually happen in production, you've spent months building a lie.

The main way this happens is **data leakage** and **improper cross-validation**. The techniques in this chapter are the discipline that prevents it.

For a GenAI engineer, this matters in specific ways:

1. **Benchmark contamination.** Modern LLM benchmarks (MMLU, GSM8K, HumanEval, etc.) have their questions and sometimes their answers scraped into the pretraining data of every LLM. When a new model claims "94% on MMLU," the honest question is: how much of that is because the benchmark leaked into training? Understanding leakage/CV helps you evaluate these claims critically.
2. **LLM-as-judge validation.** Using an LLM to judge outputs of another LLM introduces circularity: the judge's opinions are shaped by the same training data as the models it's evaluating. Standard CV logic doesn't strictly apply, but the "keep your training and evaluation strictly separated" principle does.
3. **Fine-tuning dataset construction.** Prompt-tuning, SFT, DPO — all rely on labeled data. If your validation set has the same underlying task as your training set (same customers, same product SKUs), your CV score won't predict production performance. Group k-fold is the fix.
4. **RLHF reward-model overfitting.** The reward model itself can overfit to preference data, then RLHF chases that overfit reward — a phenomenon called "reward hacking." Careful CV of the reward model is one of the guards against this.

This chapter covers the mechanics: k-fold, stratified splits, group splits, time-series splits, nested CV, and the most common ways data leakage sneaks in.

</Motivation>

<Intuition title="Hold out a slice of your data — pretend it's the future test set — never let training see it">

The setup: you have a labeled dataset. You want to know how well your model will do on **future, unseen data**. But you only have this one dataset — you can't fast-forward to the future.

**The trick:** split your data into a **training set** and a **held-out set**. Train on the training set. Evaluate on the held-out set. As long as you honestly kept the held-out set separate throughout training AND hyperparameter selection, its performance is an unbiased estimate of how you'd do on truly new data.

**K-fold cross-validation** does this K times: split into K equal chunks, use each chunk as the held-out set once. Average the K performance numbers. This gives you a more stable estimate (K-fold averages beat single train/test split at N/K reduction in variance) at K× more compute.

**The three splits people talk about:**

- **Training set:** the data your model actually fits on.
- **Validation set (or "dev set"):** used to pick hyperparameters (λ, learning rate, model architecture). Different from test — you're allowed to make decisions based on validation performance.
- **Test set:** the final "how did we actually do" number. Used ONCE, at the end. Any time you look at test-set performance and change something, the test set has been contaminated and now serves as another validation set.

**The one commandment:** **anything the model or its hyperparameters depend on must not have been influenced by the validation/test data.** Preprocessing (scaling, imputation, feature selection) is often the sneakiest leak — fit those on training only, then apply the fitted transforms to validation.

</Intuition>

<Recall term="Overfitting">

Overfitting is when a model fits noise in the training set that doesn't generalize to new data. Manifests as low training loss and high validation/test loss. The whole point of cross-validation is to give an honest estimate of generalization performance — so you can detect overfitting and pick the model / regularization that generalizes best. Overfit training predictions can be arbitrarily good; overfit validation predictions are the actual measure.

</Recall>

## Train / val / test hygiene

**The classical three-way split:**

- **Training (60-80%):** the model learns from this.
- **Validation (10-20%):** used to pick hyperparameters, decide when to stop training, choose model architecture. You look at this MANY times during development.
- **Test (10-20%):** used ONCE at the end. If you look at it and change something, you've turned it into a validation set — and you no longer have an honest generalization estimate.

**In practice for small datasets**, one splits is too noisy — hence **k-fold** CV. For larger datasets a single split often suffices for speed.

**Rules of hygiene:**

1. **Never fit ANYTHING on the validation set.** Not the model, not the scaler, not the imputer, not the feature-selector.
2. **Do all preprocessing inside the CV loop.** Preprocessing that depends on the data (mean/std for scaling, top-k feature selection, PCA) must be fit on training-only and applied to validation.
3. **Look at test set exactly once.** Every look creates a subtle "which experiment do I keep?" bias. Report performance once and move on.
4. **The test set is not for iteration.** If you find surprises on the test set, note them for the next study — don't tweak this model and re-report.

## K-fold cross-validation

**Standard K-fold**: shuffle the dataset, split into K equal-size chunks. For each fold k = 1..K:

1. Set aside fold k as the validation set.
2. Train the model on the other K-1 folds.
3. Evaluate on fold k, record the score.

Average the K scores → your CV estimate of test performance.

**Typical K values:**

- **K = 5:** the most common choice. 80/20 train/val ratio per fold, low variance in the estimate.
- **K = 10:** more stable estimate, 90/10 ratio. Slower but slightly more accurate.
- **K = N (leave-one-out):** each fold has 1 val example. Highest variance in individual scores but the estimate is unbiased. Expensive for large N.

## The 3 sneakiest data-leakage bugs

**1. Scaling before splitting.** Compute mean and std over the ENTIRE dataset, subtract mean and divide by std, THEN split. Now the training data has been affected by test-set statistics — small effect, but real.

**Fix:** compute the scaler's parameters (`mean_`, `scale_`) using ONLY the training data, then apply to validation.

**2. Feature selection before splitting.** Pick the top 100 features by correlation with y on the full dataset, then train models. The selected features "know" about the test set. In practice this can inflate CV performance by 5-30% depending on how many features you're selecting from.

**Fix:** do feature selection inside each fold on training-only.

**3. Class-imbalance handling that touches all data.** Oversample the minority class to balance the training set — but if you oversampled BEFORE splitting, your training and test set share duplicated minority examples.

**Fix:** balance inside each fold on the training portion only.

**The general rule:** any data-dependent transformation (imputation, PCA, feature selection, class-balancing, sampling weights) must be fit inside the CV loop on training only. Wrap it in a `sklearn.Pipeline` and use that as your estimator — Pipeline does the right thing automatically.

## Stratified k-fold — for classification with imbalance

If your data is **10% positive, 90% negative** and you use plain KFold, you'll get folds with varying class ratios. Some folds might be **7% positive** and others **13%**, purely by chance — that adds noise to your CV scores and can lead to bad splits (a training fold with almost no positives can't fit).

**StratifiedKFold** preserves the class distribution: each fold has the same 10/90 ratio as the whole dataset. Standard for classification. In sklearn, `cross_val_score(model, X, y)` uses stratified splits automatically when the estimator is a classifier.

## Group k-fold — when rows are not iid

Standard k-fold assumes each row is an independent sample. Reality:

- Multiple medical records from the **same patient** — they're not independent.
- Multiple purchases by the **same customer** — same.
- Multiple frames from the **same video** — same.
- Multiple sentences from the **same document** — same.

If you naively shuffle rows and put them in different folds, the training set might contain other examples from the same patient/customer/video/document, giving the model a **cheat sheet** for the validation set.

**GroupKFold** partitions the data so that all rows sharing a group ID land in the SAME fold. sklearn: `GroupKFold(n_splits=K).split(X, y, groups=patient_ids)`. Standard for any medical, longitudinal, or nested data.

## Time-series splits — future must not leak into past

For any temporal data (time series, log data, sensor readings), standard shuffled k-fold is WRONG: it randomly puts future rows in training folds and past rows in validation folds. Your model learns "the future" and predicts "the past" — no real-world scenario looks like this.

**TimeSeriesSplit** uses expanding-window or rolling-window splits: training data always comes BEFORE validation data in time. Each fold's training set is a prefix; each fold's validation is a short window immediately after that prefix.

**Rule for time-series:**

- **Always** use TimeSeriesSplit (or manual expanding-window split) for temporal data.
- **Never** shuffle rows before splitting on time series.
- Consider **gap** between train and val (some setups use `TimeSeriesSplit(gap=100)`) if there's short-term autocorrelation you don't want to leak.

## Nested cross-validation — for honest hyperparameter tuning

If you use k-fold CV to pick alpha, and then REPORT the CV score at the best alpha, that number is **biased optimistic** — you took the min over many alphas, so the noise in the CV scores helps you. On a new test set, performance will be worse than the reported CV score.

**Nested CV** fixes this: two nested loops.

- **Outer loop (K_outer folds):** for each fold, hold out a "test" chunk.
- **Inner loop (K_inner folds on the outer training set):** for each candidate hyperparameter, run K_inner CV, pick the best.
- Retrain with best hyperparameters on the FULL outer training set, evaluate on the outer test chunk.
- Average the outer test scores → your honest generalization estimate.

Total cost: K_outer × K_inner × N_hyperparams model fits. Standard for research papers and cases where "how well does this actually work" is the critical question.

<RealWorldML title="Benchmark contamination in modern LLM evaluation">

The LLM-era version of data leakage is **benchmark contamination**: the test sets of popular benchmarks (MMLU, HellaSwag, HumanEval, GSM8K) get scraped into the pretraining data of every new LLM. When GPT-4 reports 86% on MMLU, part of that is genuine capability — and part might be from having seen the questions during pretraining.

**How does this happen?**

1. Benchmarks are published on the web (papers, HuggingFace, GitHub).
2. LLM pretraining pipelines crawl the web indiscriminately.
3. Even with dedup filters, near-duplicate variants (translated, paraphrased, quiz-style repackaging) slip through.
4. During evaluation, the model sees a question it "remembers" from training and answers correctly.

**Mitigations:**

- **Held-out benchmarks (e.g., MMLU-Pro, Big-Bench Hard):** curated recently, less contamination.
- **Contamination detection:** measure the LLM's log-likelihood of the exact benchmark string vs. paraphrased versions. If the exact string is much higher, likely contamination.
- **Private test sets:** OpenAI's "hidden" MMLU, or organization-internal evaluation datasets that never leave your infra.
- **Live task creation:** ask domain experts to write new questions AFTER the model's training cutoff. GPQA-diamond, ARC-AGI use variants of this.

**The CV analog:** you CANNOT trust benchmark performance on any test set that was in a plausible crawl of the internet before your model's training cutoff. Just as you can't trust CV performance if you scaled your data before splitting.

**LLM-as-judge caveats:** using GPT-4 to judge GPT-3.5 outputs is like using the training-set-influenced model to grade another model — the judge's biases are the same. Fine for rough dev signal, bad for scientific claims.

**Group-k-fold applies to fine-tuning too.** If your fine-tuning data comes from a few specific customers/products, and your test set is from the same customers/products, your CV score will overestimate generalization to new customers. Group by customer_id, then k-fold.

</RealWorldML>

<CodeItYourself title="Three drills — implement k-fold, spot the leakage, nested CV">

1. **Roll your own k-fold.** Given N and K, write a function that returns list of (train_idx, val_idx) tuples. Test on N = 100, K = 5: verify each fold has size 20, no index appears in both train and val, every index appears exactly once across all val sets.

2. **Detect leakage.** For a regression problem, fit a StandardScaler on the whole dataset before splitting. Then repeat, fitting scaler inside each fold. Compare CV scores — you should see the "leaked" one is slightly optimistic.

3. **Nested CV from scratch.** Write nested CV: outer 5-fold, inner 5-fold. For each outer fold, use inner CV to pick alpha from [0.01, 0.1, 1, 10, 100]. Report outer CV MSE. Compare to non-nested CV (which minimizes CV MSE over alpha directly — biased optimistic).

</CodeItYourself>

<Summary>

- **Three splits:** training (fit model), validation (pick hyperparameters), test (single-use, unbiased estimate).
- **K-fold CV:** average of K held-out scores; K = 5 or 10 is standard.
- **Preprocessing must be inside the fold** — otherwise data leakage from validation into training.
- **Stratified k-fold** for classification, **GroupKFold** for non-iid data, **TimeSeriesSplit** for temporal data.
- **Nested CV** = outer loop for honest generalization + inner loop for hyperparameter search. Prevents optimistic bias.
- **The test set is used exactly once.** Every look is a leak.
- **sklearn Pipeline** automates the "preprocess inside CV" rule; use it.
- **Modern LLM analog: benchmark contamination.** Trust benchmarks only when the questions weren't in the pretraining crawl.

</Summary>
