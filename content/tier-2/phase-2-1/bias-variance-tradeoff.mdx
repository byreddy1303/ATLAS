export const biasVarianceEmpiricalCode = `# Empirical bias-variance decomposition on polynomial regression.
# For each model complexity, resample the training set many times, retrain,
# and measure BIAS (average prediction vs truth) and VARIANCE (spread of predictions).
import numpy as np, matplotlib.pyplot as plt

np.random.seed(0)

def true_fn(x):    return np.sin(2 * x) + 0.5 * x
def make_data(N):
    x = np.random.uniform(-2, 2, N)
    y = true_fn(x) + 0.4 * np.random.randn(N)
    return x, y

def poly_features(x, degree):
    return np.stack([x**k for k in range(degree + 1)], axis=1)

def fit_poly(x, y, degree):
    X = poly_features(x, degree)
    return np.linalg.lstsq(X, y, rcond=None)[0]

x_test = np.linspace(-2, 2, 200)
y_true = true_fn(x_test)

# For each degree, resample training data 100 times, compute prediction at each test point
degrees = [1, 3, 8, 20]
N_train = 40
N_resamples = 100

fig, axes = plt.subplots(1, 4, figsize=(15, 3.5))
bias_sq_list, var_list = [], []

for ax, deg in zip(axes, degrees):
    preds = np.zeros((N_resamples, len(x_test)))
    for i in range(N_resamples):
        x_tr, y_tr = make_data(N_train)
        w = fit_poly(x_tr, y_tr, deg)
        preds[i] = poly_features(x_test, deg) @ w

    mean_pred = preds.mean(axis=0)
    bias_sq   = np.mean((mean_pred - y_true) ** 2)
    variance  = np.mean(preds.var(axis=0))
    bias_sq_list.append(bias_sq); var_list.append(variance)

    # Show fanned resamples + mean + truth
    for i in range(min(30, N_resamples)):
        ax.plot(x_test, preds[i], "-", color="rosybrown", lw=0.6, alpha=0.4)
    ax.plot(x_test, mean_pred, "-", color="crimson", lw=2, label=f"mean pred")
    ax.plot(x_test, y_true,   "-", color="teal",    lw=2, label=f"true")
    ax.set_title(f"degree {deg}: bias^2={bias_sq:.3f}, var={variance:.3f}")
    ax.set_ylim(-3, 3); ax.legend(fontsize=8); ax.grid(alpha=0.3)

plt.tight_layout()

print(f"{'degree':>8}  {'bias^2':>10}  {'variance':>10}  {'total':>10}")
for deg, b2, v in zip(degrees, bias_sq_list, var_list):
    print(f"{deg:>8}  {b2:>10.4f}  {v:>10.4f}  {b2+v:>10.4f}")
print()
print("Low degree (underfits): HIGH bias, LOW variance.")
print("High degree (overfits): LOW bias, HIGH variance.")
print("Middle degree: minimizes bias + variance (best generalization).")`;

# Different curves = different diagnoses:
#   HIGH BIAS: both losses converge to a HIGH value -> underfitting
#   HIGH VAR:  large GAP between train and val losses -> overfitting
import numpy as np, matplotlib.pyplot as plt

np.random.seed(0)
def true_fn(x): return 3 * np.sin(x) + 0.5 * x**2 - x

def make_data(N):
    x = np.random.uniform(-3, 3, N)
    y = true_fn(x) + 0.6 * np.random.randn(N)
    return x, y

def poly_features(x, degree):
    return np.stack([x**k for k in range(degree + 1)], axis=1)

def eval_at_size(N, degree, n_trials=20):
    train_losses, val_losses = [], []
    for _ in range(n_trials):
        x_tr, y_tr = make_data(N)
        x_val, y_val = make_data(500)
        X_tr = poly_features(x_tr, degree)
        X_val = poly_features(x_val, degree)
        w = np.linalg.lstsq(X_tr, y_tr, rcond=None)[0]
        train_losses.append(np.mean((X_tr @ w - y_tr) ** 2))
        val_losses.append(np.mean((X_val @ w - y_val) ** 2))
    return np.mean(train_losses), np.mean(val_losses)

Ns = [10, 25, 50, 100, 200, 500]

fig, axes = plt.subplots(1, 3, figsize=(14, 4))
for ax, (deg, label) in zip(axes, [(1, "degree 1 (UNDERFITTING)"),

    train, val = [], []
    for N in Ns:
        t, v = eval_at_size(N, deg)
        train.append(t); val.append(v)
    ax.plot(Ns, train, "o-", color="crimson", lw=2, label="training loss")
    ax.plot(Ns, val,   "o-", color="teal",    lw=2, label="validation loss")

export const doubleDescentCode = `# The double-descent phenomenon: modern deep learning wrinkle in bias-variance.
# For over-parameterized models, test error DECREASES beyond the interpolation
# threshold -- opposite of the classical U-shape.
import numpy as np, matplotlib.pyplot as plt

np.random.seed(0)

# Small linear-regression-with-random-features problem, sweep model width
N_train = 100
N_test  = 500
D_in = 20
x_train = np.random.randn(N_train, D_in)
x_test  = np.random.randn(N_test,  D_in)
w_true  = np.random.randn(D_in)
y_train = x_train @ w_true + 0.5 * np.random.randn(N_train)
y_test  = x_test  @ w_true + 0.5 * np.random.randn(N_test)

def random_features(x, D_feat, seed):
    rng = np.random.default_rng(seed)
    W = rng.normal(0, 1, size=(x.shape[1], D_feat))
    b = rng.normal(0, 1, size=D_feat)
    return np.tanh(x @ W + b)             # random-feature MLP-style

# Sweep number of random features
D_feats = [5, 10, 20, 40, 80, 99, 100, 105, 150, 300, 600, 1000]
train_mse, test_mse = [], []
for D_feat in D_feats:
    Phi_train = random_features(x_train, D_feat, seed=42)
    Phi_test  = random_features(x_test,  D_feat, seed=42)
    # Use lstsq (min-norm solution when over-parameterized)
    w, *_ = np.linalg.lstsq(Phi_train, y_train, rcond=None)
    train_mse.append(np.mean((Phi_train @ w - y_train) ** 2))
    test_mse.append (np.mean((Phi_test  @ w - y_test)  ** 2))

# Print results
print(f"{'D_feat':>8}  {'train MSE':>10}  {'test MSE':>10}")
for D, tr, te in zip(D_feats, train_mse, test_mse):
    marker = " <-- interpolation threshold" if D == N_train else ""
    print(f"{D:>8}  {tr:>10.4f}  {te:>10.4f}{marker}")

fig, ax = plt.subplots(figsize=(8, 4))
ax.plot(D_feats, train_mse, "o-", color="crimson", label="train MSE")
ax.plot(D_feats, test_mse,  "o-", color="teal",    label="test MSE")
ax.axvline(N_train, linestyle="--", color="gold", label="interpolation threshold (D = N)")
ax.set_xscale("log"); ax.set_yscale("log")
ax.set_xlabel("model width (D_feat)"); ax.set_ylabel("MSE")
ax.set_title("Double descent: test MSE peaks at D=N, then DROPS again")
ax.legend(); ax.grid(alpha=0.3, which="both")
plt.tight_layout()

print()
print("Classical bias-variance: test error is a U-shape in complexity.")
print("Modern (large-model) regime: test error has TWO valleys -- one classical,")
print("one at the highly over-parameterized end. Between them, at D ~= N, there's")
print("a PEAK. This is 'double descent' and is why very large transformers work.")`;

<Motivation title="The bias-variance dial — the single most important diagnostic in ML">

Say you've trained a model and its test loss is bad. You need to fix it. Where do you look?

There are basically only two possible reasons your test loss is bad:

1. **The model is too simple to represent the true relationship.** You're bringing a straight line to a curve — no fit is going to work. This kind of failure is called **bias**.
2. **The model has enough flexibility, but the specific fit is chasing the noise in your training set.** If you'd trained on a *different* random sample of data, you'd have gotten a very different model. This kind of failure is called **variance**.

Here's the frustrating part: **the two failures require opposite fixes.**

- **High bias?** Give the model more capacity — more parameters, richer features, less regularization.
- **High variance?** Give the model *less* capacity — fewer parameters, more regularization, or more data.

Do the wrong one and you'll make things worse. Try to fix an underfitting model with more data or heavier regularization → it gets *worse*. Try to fix an overfitting model with more parameters → it overfits *harder*.

**Diagnosing which one you have** is the single most valuable ML-engineering skill covered in this chapter. It's also the difference between "we spent \$50k on a bigger cluster and it didn't help" and "we added a dropout layer and shipped in an afternoon."

For a GenAI engineer, bias-variance judgment is everywhere:

1. **Model selection.** GPT-4 vs Claude Sonnet vs a small local model. Fine-tune 7B vs 70B. LoRA rank 8 vs 64. All bias-variance calls.
2. **Regularization choices.** Weight decay, LoRA rank, dropout rate, RLHF's KL penalty — all dials on the same underlying trade-off.
3. **When to gather more data.** More data reduces variance, not bias. If your validation loss is high *and* close to your training loss, you're bias-limited — more data won't help. You need a better *model*.
4. **The double-descent story.** Modern LLMs live on a very unusual part of the bias-variance curve, where "just make it bigger" keeps working past the point where classical wisdom said it should fail. Knowing why is important for judging scaling claims.

This chapter builds three things: the formal decomposition (so the math is precise), the visual intuition (so you *feel* it), and the modern deep-learning wrinkle (so you can reason about LLMs).

</Motivation>

<Intuition title="Throw darts at a bullseye — bias is your aim, variance is your consistency">

The clearest picture of bias and variance is a target with four different dart-throwers taking shots at the bullseye.

- **Low bias, low variance** — darts land in a tight cluster right on the bullseye. Your aim is right *and* you're consistent. The dream.
- **Low bias, high variance** — darts scatter widely around the bullseye. On average, your throws are aimed at the center — but each individual throw is all over the place. Right idea, sloppy delivery.
- **High bias, low variance** — darts land in a tight cluster in the *wrong* place (say, always up and to the right of the bullseye). Consistent, but consistently off-target.
- **High bias, high variance** — darts scatter widely in the wrong region. Neither aim nor consistency. Everything's broken.

Now translate to ML. "Throwing a dart" = drawing a fresh training set and refitting the model. "The bullseye" = the true function you're trying to learn. So the *average prediction* across many resampled fits tells you the **bias** (where the "aim" is), and the *spread* of predictions tells you the **variance** (how "consistent" your fitting is). Total expected error at any test point = **bias² + variance + irreducible noise**.

**How model complexity moves both dials:**

- **A simple model (few parameters).** Can't represent the true function no matter what data you throw at it → **high bias.** But you'd get essentially the same simple fit every time → **low variance.**
- **A complex model (many parameters).** Can perfectly represent almost any function → **low bias.** But the specific fit closely chases the noise in your training set → **high variance.**
- **Just right, somewhere in the middle.** Bias² + variance sum is smallest. That's the sweet spot cross-validation is looking for.

**How regularization moves both dials:**

- **Weak regularization (small `λ`).** Model uses all its capacity → low bias, high variance. Overfits.
- **Strong regularization (large `λ`).** Model is constrained → higher bias, lower variance. Underfits at the extreme.
- **Just-right `λ`.** Minimizes total error. That's what CV picks.

Same trade-off, different knob. Understanding this equivalence is why "regularization" and "model complexity" are often used interchangeably — they're both dials on the same underlying dial.

</Intuition>

<Recall term="Expectation and variance">

For a random variable X, expectation E[X] is the average, variance Var(X) = E[(X − E[X])²] is the squared spread around the mean. The bias-variance decomposition treats the model's prediction ŷ(x) at a fixed x as a random variable — random because the training data used to fit the model is random. Bias = E[ŷ(x)] − y_true(x). Variance = Var(ŷ(x)).

</Recall>

## The formal decomposition — bias² + variance + noise, exactly

Time to make the story precise. Pick a fixed test point `x*`. The true value at that point is `y* = f(x*) + ε`, where `f` is the (unknown) true function and `ε` is random label noise (zero mean, variance `σ²`).

Your model, trained on some random training set, produces a prediction `ŷ(x*)`. Because the training set was random, `ŷ(x*)` is *itself* a random quantity. The expected squared error at this test point decomposes exactly into three pieces:

$$
\mathbb{E}[(\hat{y}(x^*) - y^*)^2] = \underbrace{(\mathbb{E}[\hat{y}(x^*)] - f(x^*))^2}_{\text{bias}^2} + \underbrace{\text{Var}(\hat{y}(x^*))}_{\text{variance}} + \underbrace{\sigma^2}_{\text{irreducible noise}}
$$

Three terms, each with its own meaning:

1. **Bias²** — how far the *average* prediction is from the truth. Only zero if the model class can contain the true function AND you happened to pick the right one.
2. **Variance** — how much the prediction fluctuates as you change the training set. Only zero for a fully deterministic model that ignores data (like "always predict 0").
3. **Noise** (`σ²`) — the irreducible error baked into the labels themselves. No model can beat this — it's a floor on how good you can be.

**Where does the decomposition come from?** Short derivation, because it's clean and worth seeing once. Let `μ = E[ŷ(x*)]` (the expected prediction, averaged over training sets). Add and subtract `μ` inside the squared error:

$$
\mathbb{E}[(\hat{y} - y^*)^2] = \mathbb{E}[(\hat{y} - \mu + \mu - f - \varepsilon)^2]
$$

Expand the square, take expectations. Three cross-terms show up — and *all three vanish* because (a) `ε` is independent of the training data, (b) `E[ε] = 0`, and (c) `ŷ − μ` has zero mean by definition. What's left:

$$
= \mathbb{E}[(\hat{y} - \mu)^2] + (\mu - f)^2 + \mathbb{E}[\varepsilon^2]
$$

$$
= \text{Var}(\hat{y}) + \text{bias}^2 + \sigma^2 \quad \blacksquare
$$

**The punchline you take away.** Increasing model capacity trades bias for variance. A more complex model can represent more functions (lower bias) but is more sensitive to the specific training data (higher variance). Their sum — the actual test error — is U-shaped in model complexity. That U is what cross-validation is trying to find the bottom of.

## Learning curves — the cheapest diagnostic in ML

A **learning curve** is a plot of training and validation error as you grow the training set size `N`. It's *the* fastest way to figure out whether you have a bias problem or a variance problem.

Three signature patterns to memorize:

- **Both curves converge to a high value.** → **High bias.** The model can't fit even with plenty of data. Fix: more capacity (bigger network, richer features, less regularization). *More data won't help you here.*
- **Big gap between training and validation error, both dropping as data grows.** → **High variance.** The model is overfitting. Fix: more data, more regularization, or less capacity.
- **Both curves are low and close together.** → **Well-tuned.** Ship it.

**The diagnostic algorithm — memorize this:**

1. Plot the learning curve with your current model + training data.
2. Big gap between train and val? → high variance → add data / regularize / shrink model.
3. Both curves high and converged? → high bias → add capacity / richer features.
4. Both curves low? → done, ship it.

Simple, cheap, and used by every serious ML engineer.

## How regularization moves the dial

Regularization is bias-variance control in the most direct form. When you add a regularization penalty:

- **Bias goes up.** The penalty distorts the optimum away from the "best possible data fit."
- **Variance goes down.** The penalty constrains where the model can live, so different training sets can't produce wildly different fits.

**As a function of `λ`:**

- `λ = 0` — no regularization → minimum bias (the model uses all its capacity), maximum variance.
- `λ → ∞` — infinite regularization → maximum bias (model collapses to a constant), zero variance.
- **Optimal `λ`** — somewhere in the middle. Formally, it's where `d(bias²)/dλ + d(variance)/dλ = 0` — the point where bias climbing up perfectly balances variance dropping down. That's what CV finds for you.

## The modern deep-learning wrinkle — double descent

Classical ML wisdom used to say: "test error is U-shaped in model complexity, so pick a moderate-sized model." Beyond the **interpolation threshold** — the model complexity where training loss hits exactly zero — everyone assumed test error would blow up and stay blown up.

**Modern reality is more interesting.** The U-shape holds up to the interpolation threshold. Past that, test error *drops again*. This phenomenon is called **double descent** (Belkin et al. 2019), and it's one of the most important empirical discoveries of the deep-learning era.

Why does the "peak" at the interpolation threshold happen?

- **Just before `D ≈ N`:** the model is *barely* flexible enough to fit the training data. It uses every degree of freedom, so any random perturbation of the data changes the fit dramatically. High variance, ugly test error.
- **Right at `D = N`:** the unique interpolating solution tends to be pathological — the model is stretched to its absolute limit.
- **For `D ≫ N`:** many different fits achieve zero training loss. Gradient descent (and least-squares) reliably pick the **min-norm** interpolant — the one with the smallest possible weights. That "min-norm" bias turns out to be a *very good* implicit regularizer, and generalization gets better again.

**Why this matters for GenAI:**

- **Very big models with tons of parameters can still generalize well** — even *past* the interpolation threshold. That's the theoretical justification for scaling up.
- **Small dataset + medium-sized model = danger zone.** You're sitting right at the peak of the curve. Either shrink the model (climb down the left side) or (better) grow the dataset until you're back in "the model is huge relative to the data" territory.
- **"Grokking"** — the surprising phenomenon where a model that already fits the training data keeps *improving* on the test set for many more epochs — is thought to be another manifestation of the same effect. The model is slowly descending into the over-parameterized valley.

<RealWorldML title="Bias-variance thinking in LLM training and fine-tuning">

Almost every decision in LLM development is secretly a bias-variance call.

**Choosing model size.** GPT-3.5 vs GPT-4, Claude Sonnet vs Claude Opus, 7B vs 70B fine-tune. Bigger = more capacity = lower bias, but also more expensive to run and (until double descent kicks in) higher variance. Rule of thumb: for tasks with lots of labeled examples, bigger models help. For tasks with only a few examples, smaller models often generalize better.

**Prompting vs fine-tuning vs SFT.** Prompting a pretrained model is very-high-bias / very-low-variance — you're using the model exactly as it came off the shelf, and any distribution mismatch between the pretraining data and your task shows up as bias. Fine-tuning cuts the bias (the model adapts) but adds variance (it's now sensitive to your particular dataset). The **FLAN** line of work (instruction tuning on many diverse tasks) landed in an intermediate spot that wins on both dimensions — you get task-adaptation without over-specializing to any one task.

**Choosing LoRA rank.** LoRA constrains fine-tuning updates to a **rank-`r`** subspace. Small `r` (say 4-8) = more bias (can't adapt as much), less variance (the small update is stable). Large `r` (32-64) = closer to full fine-tuning. In practice, `r = 8-32` covers most tasks well.

**RLHF's KL penalty.** The `β` coefficient in **PPO** (proximal policy optimization) controls how far the fine-tuned policy is allowed to drift from the reference (pre-RLHF) model. Small `β` → high-variance policy chases the reward hard and may drift wildly (**reward hacking** or **mode collapse**). Large `β` → high-bias policy stays too close to reference, doesn't fully use the reward signal. Classic bias-variance dial.

**When to add more data.** Classical rule: more data reduces variance, not bias. In LLM land, more *high-quality* data is nearly always positive — because it both reduces variance *and* lets you use a bigger model without overfitting. The **scaling laws** work (Kaplan 2020, Chinchilla 2022) is basically empirical bias-variance curves plotted at enormous scale, discovering the optimal joint scaling of parameters and tokens.

**Double descent at massive scale.** Frontier LLMs live in the "right descent" region — hundreds of billions of parameters, trillions of tokens. Test loss keeps improving as you add parameters past what looks like an interpolation threshold, because (a) the training data is essentially inexhaustible so overfitting is hard, and (b) the min-norm solution generalizes well. This is the mathematical version of "scale is all you need." It's not that scale magically fixes everything — it's that we're on a specific, favorable part of the double-descent curve.

</RealWorldML>
